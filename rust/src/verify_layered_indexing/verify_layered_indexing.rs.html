3use hyperstreamdb::Table;
4use anyhow::Result;

11#[tokio::main]
12async fn main() -> Result<()> {
13    let base_dir = "/tmp/iceberg_test";
14    let _ = std::fs::remove_dir_all(base_dir);
15    std::fs::create_dir_all(format!("{}/metadata", base_dir))?;
16    std::fs::create_dir_all(format!("{}/data", base_dir))?;
17
18    println!("1. Creating dummy data file f1 (Parquet)...");
19    let data_path1 = format!("{}/data/f1.parquet", base_dir);
20    create_dummy_parquet(&data_path1)?;
21
22    println!("2. Creating Iceberg Metadata v1...");
23    create_iceberg_snapshot(base_dir, 1, &data_path1, "m1.avro", "ml1.avro")?;
24    
25    let metadata_path = format!("{}/metadata/v1.metadata.json", base_dir);
26    let metadata_json = iceberg_metadata_json(base_dir, 1, "ml1.avro");
27    std::fs::write(&metadata_path, serde_json::to_string_pretty(&metadata_json)?)?;
28
29    println!("3. Registering external table in HyperStreamDB...");
30    let hdb_uri = "file:///tmp/hdb_shadow";
31    let _ = std::fs::remove_dir_all("/tmp/hdb_shadow");
32    
33    let mut table = Table::register_external(
34        hdb_uri.to_string(), 
35        &format!("file://{}", metadata_path)
36    ).await?;
37
38    println!("4. Verifying initial shadowed entries...");
39    let segments = table.get_snapshot_segments().await?;
40    assert_eq!(segments.len(), 1, "Should have 1 segment initially");
41    println!("Found 1 segment as expected.");
42
43    println!("5. Building Scalar Index for 'id' column...");
44    // This should build a sidecar index in /tmp/hdb_shadow
45    table.add_index_columns_async(vec!["id".to_string()], None).await?;
46    
47    println!("6. Checking for local sidecar index file...");
48    let mut found_idx = false;
49    for entry in std::fs::read_dir("/tmp/hdb_shadow")? {
50        let entry = entry?;
51        let name = entry.file_name().into_string().unwrap();
52        if name.contains(".id.idx") || name.contains(".id.inv.parquet") {
53            println!("Found sidecar index: {}", name);
54            found_idx = true;
55        }
56    }
57    assert!(found_idx, "Sidecar index file was not created in local HDB directory");
58
59    println!("7. Verifying Query using sidecar index...");
60    // The dummy data has id=1..100. Let's filter id=10.
61    // The scalar index (in its current mock-like state) might just filter > 0,
62    // but the point is to see if it executes.
63    let results = table.read_async(Some("id = 10"), None, None).await?;
64    println!("Query results: {} batches", results.len());
65    let mut total_rows = 0;
66    for batch in results {
67        total_rows += batch.num_rows();
68        let id_col = batch.column(0).as_any().downcast_ref::<Int32Array>().unwrap();
69        for i in 0..batch.num_rows() {
70             assert_eq!(id_col.value(i), 10, "Query returned wrong ID");
71        }
72    }
73    assert!(total_rows > 0, "Query should have returned at least one row");
74    println!("Universal Indexing Verified Successfully!");
75
76    Ok(())
77}

79fn create_dummy_parquet(path: &str) -> Result<()> {
80    use parquet::arrow::ArrowWriter;
81    use arrow::datatypes::{DataType, Field, Schema};
82    
83    let schema = Arc::new(Schema::new(vec![
84        Field::new("id", DataType::Int32, false),
85        Field::new("vector", DataType::FixedSizeList(
86            Arc::new(Field::new("item", DataType::Float32, true)),
87            4
88        ), false),
89    ]));
90
91    let file = File::create(path)?;
92    let mut writer = ArrowWriter::try_new(file, schema.clone(), None)?;
93
94    let ids = Int32Array::from((1..=100).collect::<Vec<i32>>());
95    let mut vector_values = Vec::new();
96    for i in 1..=100 {
97        for _ in 0..4 {
98            vector_values.push(i as f32);
99        }
100    }
101    let flattened_vectors = Float32Array::from(vector_values);
102    let vectors = FixedSizeListArray::try_new(
103        Arc::new(Field::new("item", DataType::Float32, true)),
104        4,
105        Arc::new(flattened_vectors),
106        None
107    )?;
108
109    let batch = RecordBatch::try_new(schema, vec![Arc::new(ids), Arc::new(vectors)])?;
110    writer.write(&batch)?;
111    writer.close()?;
112    Ok(())
113}

115fn create_iceberg_snapshot(base_dir: &str, snapshot_id: i64, data_path: &str, m_name: &str, ml_name: &str) -> Result<()> {
116    let manifest_schema_str = r#"{
117        "type": "record",
118        "name": "manifest_entry",
119        "fields": [
120            {"name": "status", "type": "int"},
121            {"name": "snapshot_id", "type": ["null", "long"]},
122            {
123                "name": "data_file",
124                "type": {
125                    "type": "record",
126                    "name": "data_file",
127                    "fields": [
128                        {"name": "file_path", "type": "string"},
129                        {"name": "file_format", "type": "string"},
130                        {"name": "partition", "type": {"type": "record", "name": "partition", "fields": []}},
131                        {"name": "record_count", "type": "long"},
132                        {"name": "file_size_in_bytes", "type": "long"}
133                    ]
134                }
135            }
136        ]
137    }"#;
138    let manifest_schema = AvroSchema::parse_str(manifest_schema_str)?;
139    let manifest_path = format!("{}/metadata/{}", base_dir, m_name);
140    let mut writer = Writer::new(&manifest_schema, File::create(&manifest_path)?);
141
142    let data_file = vec![
143        ("file_path".to_string(), AvroValue::String(data_path.to_string())),
144        ("file_format".to_string(), AvroValue::String("PARQUET".to_string())),
145        ("partition".to_string(), AvroValue::Record(Vec::new())),
146        ("record_count".to_string(), AvroValue::Long(100)),
147        ("file_size_in_bytes".to_string(), AvroValue::Long(1024)),
148    ];
149
150    let entry = vec![
151        ("status".to_string(), AvroValue::Int(1)), // Added
152        ("snapshot_id".to_string(), AvroValue::Union(1, Box::new(AvroValue::Long(snapshot_id)))),
153        ("data_file".to_string(), AvroValue::Record(data_file)),
154    ];
155
156    writer.append(AvroValue::Record(entry))?;
157    writer.flush()?;
158
159    let ml_schema_str = r#"{
160        "type": "record",
161        "name": "manifest_list_entry",
162        "fields": [
163            {"name": "manifest_path", "type": "string"},
164            {"name": "manifest_length", "type": "long"},
165            {"name": "partition_spec_id", "type": "int"},
166            {"name": "added_snapshot_id", "type": "long"}
167        ]
168    }"#;
169    let ml_schema = AvroSchema::parse_str(ml_schema_str)?;
170    let ml_path = format!("{}/metadata/{}", base_dir, ml_name);
171    let mut ml_writer = Writer::new(&ml_schema, File::create(&ml_path)?);
172
173    let ml_entry = vec![
174        ("manifest_path".to_string(), AvroValue::String(m_name.to_string())),
175        ("manifest_length".to_string(), AvroValue::Long(512)),
176        ("partition_spec_id".to_string(), AvroValue::Int(0)),
177        ("added_snapshot_id".to_string(), AvroValue::Long(snapshot_id)),
178    ];
179
180    ml_writer.append(AvroValue::Record(ml_entry))?;
181    ml_writer.flush()?;
182    Ok(())
183}
