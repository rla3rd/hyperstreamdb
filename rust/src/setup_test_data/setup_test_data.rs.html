3use hyperstreamdb::Table;
4use arrow::datatypes::{DataType, Field, Schema};
5use arrow::array::{Int32Array, StringArray};
6use arrow::record_batch::RecordBatch;
7use std::sync::Arc;
8
9#[tokio::main]
10async fn main() -> Result<(), Box<dyn std::error::Error>> {
11    let uri = std::env::var("HYPERSTREAM_STORAGE_URI").unwrap_or_else(|_| "file:///tmp/test_table".to_string());
12    
13    // Clean up previous run
14    let _ = std::fs::remove_dir_all("/tmp/test_table");
15    
16    println!("Creating table at {}", uri);
17    let schema = Arc::new(Schema::new(vec![
18        Field::new("id", DataType::Int32, false),
19        Field::new("name", DataType::Utf8, false),
20    ]));
21    
22    let table = Table::create_async(uri.to_string(), schema.clone()).await?;
23    
24    println!("Inserting data...");
25    let batch = RecordBatch::try_new(
26        schema.clone(),
27        vec![
28            Arc::new(Int32Array::from(vec![1, 2, 3])),
29            Arc::new(StringArray::from(vec!["Alice", "Bob", "Charlie"])),
30        ],
31    )?;
32    
33    table.write_async(vec![batch]).await?;
34    table.commit_async().await?;
35    
36    println!("Data seeded.");
37    Ok(())
38}
