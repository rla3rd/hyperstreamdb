50async fn query_handler(Json(payload): Json<QueryRequest>) -> impl IntoResponse {
51    println!("Received query: filter='{}'", payload.filter);
52    
53    // Demonstrate the "Index-First" Read
54    // Use factory to support s3://, az://, etc.
55    // Ideally this comes from payload or config. defaulting to /tmp for local PoC
56    let uri = std::env::var("HYPERSTREAM_STORAGE_URI").unwrap_or_else(|_| "file:///tmp".to_string());
57    println!("Connecting to storage: {}", uri);
58    let store = hyperstreamdb::core::storage::create_object_store(&uri).expect("Failed to create object store");
59    
60    // Config: path is relative to the store prefix now
61    let config = SegmentConfig::new("", "segment_001");
62    let reader = HybridReader::new(config, store, &uri);
63    
64    let filter = hyperstreamdb::core::planner::QueryFilter::parse(&payload.filter).unwrap();
65    // Gateway queries all columns by default (None = no projection)
66    match reader.query_index_first(&filter, None::<std::sync::Arc<Schema>>).await {
67        Ok(batches) => {
68             let total_rows: usize = batches.iter().map(|b: &arrow::record_batch::RecordBatch| b.num_rows()).sum();
69             println!("Query successful. Read {} rows from Parquet.", total_rows);
70             let response = QueryResponse {
71                rows: vec![format!("{} rows matching index", total_rows)]
72             };
73             (StatusCode::OK, Json(response))
74        },
75        Err(e) => {
76            eprintln!("Query failed: {}", e);
77            let response = QueryResponse {
78                rows: vec![format!("Error: {}", e)]
79            };
80            (StatusCode::INTERNAL_SERVER_ERROR, Json(response))
81        }
82    }
83}

91use arrow::array::Int32Array;
92use arrow::datatypes::{DataType, Field, Schema};
93use arrow::record_batch::RecordBatch;

95
96async fn ingest_handler(Json(_payload): Json<IngestRequest>) -> impl IntoResponse {
97    println!("Received ingest data request");
98
99    // 1. Create Mock Data (Arrow Batch) for PoC
100    let id_array = Int32Array::from(vec![1, 2, 3, 4, 5]);
101    let schema = Schema::new(vec![
102        Field::new("id", DataType::Int32, false),
103    ]);
104    let batch = RecordBatch::try_new(
105        Arc::new(schema),
106        vec![Arc::new(id_array)],
107    ).unwrap();
108
109    // 2. Configure Writer
110    // In a real app, base_path would be S3 bucket or config
111    let config = SegmentConfig::new("/tmp", "segment_001"); 
112    let writer = HybridSegmentWriter::new(config);
113
114    // 3. Write Data & Index
115    match writer.write_batch(&batch) {
116        Ok(_) => (StatusCode::CREATED, "Ingested and Indexed"),
117        Err(e) => {
118            eprintln!("Error writing segment: {}", e);
119            (StatusCode::INTERNAL_SERVER_ERROR, "Failed to write")
120        }
121    }
122}
