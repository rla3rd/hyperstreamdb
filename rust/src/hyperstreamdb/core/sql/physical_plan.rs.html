24#[derive(Debug)]
25pub struct HyperStreamExec {
26    pub table: Arc<Table>,
27    // Partitions: Each partition is a list of segments to read
28    pub partitions: Vec<Vec<ManifestEntry>>,
29    projection: Option<Vec<usize>>,
30    filter: Option<String>,
31    pub vector_params: Option<VectorSearchParams>,
32    limit: Option<usize>,
33    base_schema: SchemaRef,  // Original table schema for projection
34    schema: SchemaRef,        // Projected schema
35    properties: PlanProperties,
36}
37
38impl HyperStreamExec {
39    pub fn new(
40        table: Arc<Table>,
41        partitions: Vec<Vec<ManifestEntry>>,
42        projection: Option<Vec<usize>>,
43        filter: Option<String>,
44        vector_params: Option<VectorSearchParams>,
45        limit: Option<usize>,
46        base_schema: SchemaRef,
47    ) -> Result<Self> {
48        // Calculate projected schema
49        let projected_schema = if let Some(ref proj) = projection {
50            // Validate projection indices
51            if proj.iter().any(|&i| i >= base_schema.fields().len()) {
52                // If projection is invalid, use base schema
53                base_schema.clone()
54            } else {
55                Arc::new(base_schema.project(proj)
56                    .map_err(|e| DataFusionError::from(e))?)
57            }
58        } else {
59            base_schema.clone()
60        };
61
62        let partition_count = partitions.len().max(1);
63
64        let properties = PlanProperties::new(
65            EquivalenceProperties::new(projected_schema.clone()),
66            Partitioning::UnknownPartitioning(partition_count),
67            EmissionType::Final,
68            Boundedness::Bounded,
69        );
70
71        Ok(Self {
72            table,
73            partitions,
74            projection,
75            filter,
76            vector_params,
77            limit,
78            base_schema,
79            schema: projected_schema,
80            properties,
81        })
82    }
83    pub fn projection(&self) -> Option<&Vec<usize>> {
84        self.projection.as_ref()
85    }
86
87    pub fn filter_str(&self) -> Option<&str> {
88        self.filter.as_deref()
89    }
90}
91
92impl DisplayAs for HyperStreamExec {
93    fn fmt_as(
94        &self,
95        t: datafusion::physical_plan::DisplayFormatType,
96        f: &mut std::fmt::Formatter,
97    ) -> std::fmt::Result {
98        match t {
99            datafusion::physical_plan::DisplayFormatType::Default | datafusion::physical_plan::DisplayFormatType::Verbose => {
100                write!(f, "HyperStreamExec: partitions={}, filter={:?}, projection={:?}, limit={:?}", self.partitions.len(), self.filter, self.projection, self.limit)
101            }
102            _ => Ok(()),
103        }
104    }
105}

107impl ExecutionPlan for HyperStreamExec {
108    fn name(&self) -> &str {

110    }

114    }

116    fn schema(&self) -> SchemaRef {

118    }
119
120    fn properties(&self) -> &PlanProperties {
121        &self.properties
122    }
123
124    fn children(&self) -> Vec<&Arc<dyn ExecutionPlan>> {
125        vec![]
126    }
127
128    fn with_new_children(
129        self: Arc<Self>,
130        _: Vec<Arc<dyn ExecutionPlan>>,
131    ) -> Result<Arc<dyn ExecutionPlan>> {
132        Ok(Arc::new(HyperStreamExec::new(
133            self.table.clone(),
134            self.partitions.clone(),
135            self.projection.clone(),
136            self.filter.clone(),
137            self.vector_params.clone(),
138            self.limit,
139            self.base_schema.clone(),  // Use base schema for reprojection
140        )?))
141    }
142
143    fn execute(
144        &self,
145        partition: usize,
146        _context: Arc<TaskContext>,
147    ) -> Result<SendableRecordBatchStream> {
148        if partition >= self.partitions.len() && !self.partitions.is_empty() {
149            return Err(DataFusionError::Internal(format!(
150                "HyperStreamExec invalid partition {} (count {})",
151                partition, self.partitions.len()
152            )));
153        }
154
155        let table = self.table.clone();
156        let filter = self.filter.clone();
157        let vector_params = self.vector_params.clone();
158        
159        // If no partitions (empty table), return empty stream
160        let entries = if self.partitions.is_empty() {
161            Vec::new()
162        } else {
163            self.partitions[partition].clone()
164        };
165
166        // Resolve usage of projection to column names
167        let original_schema = table.arrow_schema();
168        let column_names = if let Some(ref proj) = self.projection {
169             let names: Vec<String> = proj.iter()
170                 .map(|i| original_schema.field(*i).name().clone())
171                 .collect();
172             Some(names)
173        } else {
174             None
175        };
176        
177        // Pre-convert column names to &str slice
178        let col_names_owned = column_names;
179        
180        let expected_schema = self.schema.clone();
181        let expected_schema_inner = expected_schema.clone();
182        use crate::core::planner::{QueryFilter, FilterExpr};
183        
184        let stream = async_stream::stream! {            
185            // For each segment in this partition
186            for entry in entries {
187                let col_refs: Option<Vec<&str>> = col_names_owned.as_ref().map(|v| v.iter().map(|s| s.as_str()).collect());
188                let col_slice = col_refs.as_deref();
189
190                // Apply filter parsing inside the loop or pre-parse?
191                // read_segment handles parsing if we pass QueryFilter.
192                // But here we have string filter.
193                // Better to parse once? 
194                // Table::read_segment takes `query_filter_opt: Option<&QueryFilter>`
195                
196                let filter_expr = if let Some(ref f) = filter {
197                     let filters = QueryFilter::parse_multi(f);
198                     FilterExpr::from_filters(filters)
199                } else {
200                    None
201                };
202                
203                // Read Segment
204                if let Some(ref vp) = vector_params {
205                    // Hybrid Search: Vector + Scalar
206                    use crate::core::query::{execute_vector_search_with_config, VectorSearchRequest};
207                    
208                    let mut request = VectorSearchRequest::new(
209                        vp.column.clone(),
210                        vp.query.clone(),
211                        vp.k,
212                        vp.metric,
213                    )
214                    .with_filter(filter_expr)
215                    .with_config(table.query_config().clone())
216                    .with_ef_search(vp.ef_search);
217
218                    // Pass projected columns if available
219                    if let Some(ref proj_names) = col_names_owned {
220                        request = request.with_columns(Some(proj_names.clone()));
221                    }
222                    
223                    match execute_vector_search_with_config(
224                        vec![entry.clone()],
225                        table.object_store(),
226                        None, // data_store
227                        &table.table_uri(),
228                        request,
229                    ).await {
230                        Ok(batches) => {
231                            for batch in batches {
232                                let mut b = batch;
233                                
234                                // If batch has one extra column (distance) that isn't in expected schema, drop it
235                                if b.num_columns() == expected_schema_inner.fields().len() + 1 && b.schema().fields().last().unwrap().name() == "distance" {
236                                    let mut cols = b.columns().to_vec();
237                                    cols.pop();
238                                    let mut options = datafusion::arrow::record_batch::RecordBatchOptions::default();
239                                    options.row_count = Some(b.num_rows());
240                                    b = datafusion::arrow::record_batch::RecordBatch::try_new_with_options(expected_schema_inner.clone(), cols, &options)
241                                        .map_err(|e| DataFusionError::Execution(format!("Schema mismatch in Hybrid Search: {}", e)))?;
242                                } else if b.num_columns() == expected_schema_inner.fields().len() {
243                                    // Soft-replace schema to ignore metadata mismatches
244                                    let mut options = datafusion::arrow::record_batch::RecordBatchOptions::default();
245                                    options.row_count = Some(b.num_rows());
246                                    let b_new = datafusion::arrow::record_batch::RecordBatch::try_new_with_options(expected_schema_inner.clone(), b.columns().to_vec(), &options)
247                                        .map_err(|e| DataFusionError::Execution(format!("Type mismatch in Hybrid Search: {}. Expected {:?} got {:?}", e, expected_schema_inner, b.schema())))?;
248                                    b = b_new;
249                                } else {
250                                    tracing::error!("Field count mismatch: Expected {:?}, Got {:?}", expected_schema_inner.fields().iter().map(|f| f.name()).collect::<Vec<_>>(), b.schema().fields().iter().map(|f| f.name()).collect::<Vec<_>>());
251                                    yield Err(DataFusionError::Execution(format!("Field count mismatch in Hybrid Search: Expected {} fields, got {}", expected_schema_inner.fields().len(), b.schema().fields().len())));
252                                    return;
253                                }
254                                yield Ok(b);
255                            }
256                        },
257                        Err(e) => yield Err(DataFusionError::Execution(e.to_string())),
258                    }
259                } else {
260                    // Standard scan
261                    let version = 1; 
262                    let query_filter = if let Some(ref f) = filter {
263                         QueryFilter::parse_multi(f).into_iter().next()
264                    } else { None };
265                    
266                    match table.read_segment(&entry, query_filter.as_ref(), version, col_slice).await {
267                        Ok(batches) => {
268                            for batch in batches {
269                                let mut b = batch;
270                                if b.schema().fields().len() == expected_schema_inner.fields().len() {
271                                    // Soft-replace schema to ignore metadata mismatches
272                                    let mut options = datafusion::arrow::record_batch::RecordBatchOptions::default();
273                                    options.row_count = Some(b.num_rows());
274                                    let b_new = datafusion::arrow::record_batch::RecordBatch::try_new_with_options(expected_schema_inner.clone(), b.columns().to_vec(), &options)
275                                        .map_err(|e| DataFusionError::Execution(format!("Type mismatch in standard scan: {}. Expected {:?} got {:?}", e, expected_schema_inner, b.schema())))?;
276                                    b = b_new;
277                                } else {
278                                    yield Err(DataFusionError::Execution(format!("Field count mismatch in standard scan: Expected {} fields, got {}", expected_schema_inner.fields().len(), b.schema().fields().len())));
279                                    return;
280                                }
281                                yield Ok(b);
282                            }
283                        },
284                        Err(e) => yield Err(DataFusionError::Execution(e.to_string())),
285                    }
286                }
287            }
288
289            // If this is partition 0, also include in-memory write buffer data
290            if partition == 0 {
291                let col_refs: Option<Vec<&str>> = col_names_owned.as_ref().map(|v| v.iter().map(|s| s.as_str()).collect());
292                let col_slice = col_refs.as_deref();
293                
294                let query_filter = if let Some(ref f) = filter {
295                    use crate::core::planner::QueryFilter;
296                    QueryFilter::parse(f)
297                } else {
298                    None
299                };
300
301                match table.read_write_buffer(query_filter.as_ref(), col_slice) {
302                    Ok(batches) => {
303                        for batch in batches {
304                            if batch.schema() != expected_schema_inner {
305                                yield Err(DataFusionError::Execution("Write buffer schema mismatch".to_string()));
306                                return;
307                            }
308                            yield Ok(batch);
309                        }
310                    },
311                    Err(e) => yield Err(DataFusionError::Execution(e.to_string())),
312                }
313            }
314        };
315
316        Ok(Box::pin(RecordBatchStreamAdapter::new(
317            expected_schema,
318            Box::pin(stream),
319        )))
320    }
