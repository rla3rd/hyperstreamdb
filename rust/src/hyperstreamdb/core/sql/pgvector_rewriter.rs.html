3/// Expression rewriter for pgvector-compatible syntax
4/// 
5/// This module provides a DataFusion expression rewriter that transforms pgvector
6/// operators and vector literals into UDF calls:
7/// 
8/// - Distance operators: `embedding <-> query` → `dist_l2(embedding, query)`
9/// - Vector literals: Handled through CAST expressions with custom logic
10/// 
11/// This uses DataFusion's TreeNodeRewriter for proper expression tree traversal.
12use datafusion::common::tree_node::{Transformed, TreeNodeRewriter};
13use datafusion::error::{DataFusionError, Result};
14use datafusion::logical_expr::Expr;
15use datafusion::scalar::ScalarValue;
16use std::sync::Arc;
17
18use super::vector_literal::VectorLiteralParser;
19
20/// Mapping of pgvector operators to UDF names
21const OPERATOR_MAPPINGS: &[(&str, &str)] = &[
22    ("<->", "dist_l2"),
23    ("<=>", "dist_cosine"),
24    ("<#>", "dist_ip"),
25    ("<+>", "dist_l1"),
26    ("<~>", "dist_hamming"),
27    ("<%>", "dist_jaccard"),
28];
29
30/// Expression rewriter that converts pgvector syntax to UDF calls
31pub struct PgVectorRewriter;
32
33impl TreeNodeRewriter for PgVectorRewriter {
34    type Node = Expr;
35
36    fn f_down(&mut self, expr: Expr) -> Result<Transformed<Expr>> {
37        match expr {
38            // Handle binary expressions with custom operators (e.g. <->)
39            Expr::BinaryExpr(binary) => {
40                let op_str = binary.op.to_string();
41                
42                // Check if this is a pgvector operator
43                if let Some((_, udf_name)) = OPERATOR_MAPPINGS.iter().find(|(op, _)| op == &op_str) {
44                    let mut left = *binary.left;
45                    let mut right = *binary.right;
46                    
47                    // Check if either side is a string literal starting with '['
48                    // and convert it automatically to a vector literal
49                    for arg in [&mut left, &mut right] {
50                        let literal_text = match arg {
51                            Expr::Literal(ScalarValue::Utf8(Some(s)), _) => Some(s),
52                            Expr::Literal(ScalarValue::LargeUtf8(Some(s)), _) => Some(s),
53                            _ => None,
54                        };
55                        
56                        if let Some(s) = literal_text {
57                            let trimmed = s.trim();
58                            if trimmed.starts_with('[') {
59                                if let Ok(parsed_value) = VectorLiteralParser::parse(trimmed) {
60                                    *arg = Expr::Literal(parsed_value, None);
61                                }
62                            }
63                        }
64                    }
65                    
66                    // Create the UDF call
67                    let udf = get_distance_udf(udf_name)?;
68                    let args = vec![left, right];
69                    let func = datafusion::logical_expr::expr::ScalarFunction::new_udf(
70                        Arc::new(udf),
71                        args,
72                    );
73                    return Ok(Transformed::yes(Expr::ScalarFunction(func)));
74                }
75                
76                Ok(Transformed::no(Expr::BinaryExpr(binary)))
77            }
78            
79            // Handle scalar function calls (e.g., dist_l2)
80            Expr::ScalarFunction(mut func) => {
81                let name = func.name();
82                if matches!(name, "dist_l2" | "dist_cosine" | "dist_ip" | "dist_l1" | "dist_hamming" | "dist_jaccard") {
83                    if func.args.len() == 2 {
84                        // Support both regular and large string literals for the second argument
85                        let literal_text = match &func.args[1] {
86                            Expr::Literal(ScalarValue::Utf8(Some(s)), _) => Some(s.as_str()),
87                            Expr::Literal(ScalarValue::LargeUtf8(Some(s)), _) => Some(s.as_str()),
88                            _ => None,
89                        };
90
91                        if let Some(s) = literal_text {
92                            let trimmed = s.trim();
93                            if trimmed.starts_with('[') {
94                                match VectorLiteralParser::parse(trimmed) {
95                                    Ok(parsed_value) => {
96                                        func.args[1] = Expr::Literal(parsed_value, None);
97                                        return Ok(Transformed::yes(Expr::ScalarFunction(func)));
98                                    }
99                                    Err(e) => {
100                                        return Err(datafusion::error::DataFusionError::Plan(
101                                            format!("Failed to parse vector literal: {}. Input starts with: {}", e, &trimmed[..trimmed.len().min(50)])
102                                        ));
103                                    }
104                                }
105                            }
106                        }
107                    }
108                }
109                Ok(Transformed::no(Expr::ScalarFunction(func)))
110            }
111            
112            // Recurse into aliases to ensure nested distance calls are rewritten
113            Expr::Alias(mut alias) => {
114                use datafusion::common::tree_node::TreeNode;
115                // Move the expression out of the box to rewrite it
116                let expr = std::mem::replace(&mut alias.expr, Box::new(Expr::Literal(ScalarValue::Null, None)));
117                let transformed = expr.rewrite(self)?;
118                
119                alias.expr = Box::new(transformed.data);
120                if transformed.transformed {
121                    return Ok(Transformed::yes(Expr::Alias(alias)));
122                }
123                Ok(Transformed::no(Expr::Alias(alias)))
124            }
125            
126            _ => Ok(Transformed::no(expr)),
127        }
128    }
129}

131/// Helper function to create a new distance UDF
132fn get_distance_udf(name: &str) -> Result<datafusion::logical_expr::ScalarUDF> {
133    use super::vector_udf::*;
134    
135    let udf = match name {
136        "dist_l2" => datafusion::logical_expr::ScalarUDF::new_from_impl(L2DistUDF::new()),
137        "dist_cosine" => datafusion::logical_expr::ScalarUDF::new_from_impl(CosineDistUDF::new()),
138        "dist_ip" => datafusion::logical_expr::ScalarUDF::new_from_impl(IPDistUDF::new()),
139        "dist_l1" => datafusion::logical_expr::ScalarUDF::new_from_impl(L1DistUDF::new()),
140        "dist_hamming" => datafusion::logical_expr::ScalarUDF::new_from_impl(HammingDistUDF::new()),
141        "dist_jaccard" => datafusion::logical_expr::ScalarUDF::new_from_impl(JaccardDistUDF::new()),
142        _ => return Err(DataFusionError::Plan(
143            format!("Unknown distance UDF: {}", name)
144        )),
145    };
146    
147    Ok(udf)
148}

150/// Rewrite a logical plan to convert pgvector syntax
151pub fn rewrite_pgvector_plan(
152    plan: datafusion::logical_expr::LogicalPlan,
153) -> Result<datafusion::logical_expr::LogicalPlan> {
154    use datafusion::common::tree_node::TreeNode;
155    
156    // Use transform to visit all nodes in the logical plan
157    // For each node, we map its expressions through our PgVectorRewriter
158    let result = plan.transform(|node| {
159        let mut rewriter = PgVectorRewriter;
160        node.map_expressions(|expr| {
161            expr.rewrite(&mut rewriter)
162        })
163    })?;
164    
165    Ok(result.data)
166}

168/// Rewrites pgvector-specific SQL syntax into standard DataFusion SQL strings
169/// BEFORE parsing the AST. This is necessary because DataFusion's SQL parser
170/// does not natively map pgvector operators (e.g. `<->`) or `::vector` casts
171/// into internal DataFusion LogicalPlan nodes effectively.
172pub fn rewrite_sql_string(query: &str) -> String {
173    let mut q = query.to_string();
174    
175    // 1. Remove ::vector, ::vector(1536), etc.
176    let re_cast = regex::Regex::new(r"::vector(?:\(\d+\))?").unwrap();
177    q = re_cast.replace_all(&q, "").to_string();
178    
179    // 2. Replace distance operators: A <-> B -> dist_l2(A, B)
180    // We match LHS: an identifier or function call (without nested parens)
181    // RHS: string literal '[...]', ARRAY[...], or generic identifier
182    
183    let operators = [
184        ("<->", "dist_l2"),
185        ("<=>", "dist_cosine"),
186        ("<#>", "dist_ip"),
187        ("<+>", "dist_l1"),
188        ("<~>", "dist_hamming"),
189        ("<%>", "dist_jaccard"),
190    ];
191    
192    for (op, func) in operators {
193        // Simple regex to grab word-character identifiers and simple string literals
194        let pattern = format!(r"(?P<lhs>[a-zA-Z0-9_.]+(?:\([^)]*\))?)\s*{}\s*(?P<rhs>'[^']+'|ARRAY\[[^\]]+\]|[a-zA-Z0-9_.]+)", regex::escape(op));
195        if let Ok(re) = regex::Regex::new(&pattern) {
196            let replacement = format!("{}(${{lhs}}, ${{rhs}})", func);
197            q = re.replace_all(&q, replacement.as_str()).to_string();
198        }
199    }
200    
201    q
202}
