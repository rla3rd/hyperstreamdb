20/// Helper function to check if values in `col` distinct are in `values` set.
21/// Returns a BooleanArray where true means the value is in the set.
22fn check_is_in(col: &arrow::array::ArrayRef, values: &arrow::array::ArrayRef) -> Result<arrow::array::BooleanArray> {
23    use arrow::datatypes::*;
24    use arrow::array::Array;
25    use std::collections::HashSet;
26    
27    match col.data_type() {
28        DataType::Utf8 => {
29             let col_arr = col.as_any().downcast_ref::<arrow::array::StringArray>()
30                 .ok_or_else(|| anyhow::anyhow!("Expected StringArray in column for equality check"))?;
31             let val_arr = arrow::compute::cast(values, &DataType::Utf8)?; // Ensure type match
32             let val_arr = val_arr.as_any().downcast_ref::<arrow::array::StringArray>()
33                 .ok_or_else(|| anyhow::anyhow!("Expected StringArray in values for equality check"))?;
34             
35             let mut set = HashSet::with_capacity(val_arr.len());
36             for i in 0..val_arr.len() {
37                 if !val_arr.is_null(i) {
38                     set.insert(val_arr.value(i));
39                 }
40             }
41             
42             let mut result = arrow::array::BooleanBuilder::with_capacity(col_arr.len());
43             for i in 0..col_arr.len() {
44                 if col_arr.is_null(i) {
45                     result.append_value(false); // Nulls don't match (usually)
46                 } else {
47                     result.append_value(set.contains(col_arr.value(i)));
48                 }
49             }
50             Ok(result.finish())
51        },
52        DataType::Int64 | DataType::Date64 | DataType::Timestamp(_, _) => {
53             // Treat all as Int64 for comparison if possible, or cast
54             let col_arr = arrow::compute::cast(col, &DataType::Int64)?;
55             let col_arr = col_arr.as_any().downcast_ref::<arrow::array::Int64Array>()
56                 .ok_or_else(|| anyhow::anyhow!("Expected Int64Array in column for equality check"))?;
57             
58             let val_arr = arrow::compute::cast(values, &DataType::Int64)?;
59             let val_arr = val_arr.as_any().downcast_ref::<arrow::array::Int64Array>()
60                 .ok_or_else(|| anyhow::anyhow!("Expected Int64Array in values for equality check"))?;
61             
62             let mut set = HashSet::with_capacity(val_arr.len());
63             for i in 0..val_arr.len() {
64                 if !val_arr.is_null(i) {
65                     set.insert(val_arr.value(i));
66                 }
67             }
68             
69             let mut result = arrow::array::BooleanBuilder::with_capacity(col_arr.len());
70             for i in 0..col_arr.len() {
71                  if col_arr.is_null(i) {
72                      result.append_value(false);
73                  } else {
74                      result.append_value(set.contains(&col_arr.value(i)));
75                  }
76             }
77             Ok(result.finish())
78        },
79        DataType::Int32 | DataType::Date32 | DataType::Time32(_) => {
80             let col_arr = arrow::compute::cast(col, &DataType::Int32)?;
81             let col_arr = col_arr.as_any().downcast_ref::<arrow::array::Int32Array>()
82                 .ok_or_else(|| anyhow::anyhow!("Expected Int32Array in column for equality check"))?;
83             
84             let val_arr = arrow::compute::cast(values, &DataType::Int32)?;
85             let val_arr = val_arr.as_any().downcast_ref::<arrow::array::Int32Array>()
86                 .ok_or_else(|| anyhow::anyhow!("Expected Int32Array in values for equality check"))?;
87             
88             let mut set = HashSet::with_capacity(val_arr.len());
89             for i in 0..val_arr.len() {
90                 if !val_arr.is_null(i) {
91                     set.insert(val_arr.value(i));
92                 }
93             }
94             
95             let mut result = arrow::array::BooleanBuilder::with_capacity(col_arr.len());
96             for i in 0..col_arr.len() {
97                  if col_arr.is_null(i) {
98                      result.append_value(false);
99                  } else {
100                      result.append_value(set.contains(&col_arr.value(i)));
101                  }
102             }
103             Ok(result.finish())
104        },
105        _ => {
106             // Fallback or warning
107             tracing::warn!("Unsupported generic equality check for type: {:?}", col.data_type());
108             // Return false (no match) to be safe (don't delete anything)
109             let result = arrow::array::BooleanArray::from(vec![false; col.len()]);
110             Ok(result)
111        }
112    }

165    /// Resolve a path for ObjectStore operations (relative to store root)
166    fn resolve_object_path(&self, extension: &str) -> Path {
167        // 1. Get the base string and determine if it includes the filename
168        let (base, has_filename) = if extension == "parquet" && self.config.parquet_path.is_some() {
169            (self.config.parquet_path.as_ref().map(|p| p.as_str()).unwrap_or(""), true)
170        } else {
171            (self.config.base_path.as_str(), false)
172        };
173
174        let filename = if has_filename {
175            String::new()
176        } else {
177            format!("{}.{}", self.config.segment_id, extension)
178        };
179
180        // 2. Helper to get local path from URI or absolute path
181        fn to_local(s: &str) -> &str {
182            s.strip_prefix("file://").unwrap_or(s)
183        }
184
185        let root_local = to_local(&self.root_uri).trim_end_matches('/');
186        let base_local = to_local(base).trim_end_matches('/');
187
188        // 3. Relativize
189        let mut rel = if !root_local.is_empty() && base_local.starts_with(root_local) {
190            let r = &base_local[root_local.len()..];
191            r.trim_start_matches('/').to_string()
192        } else if base.contains("://") {
193            // If it's a URI but not matching root, try to parse it
194            if let Ok(url) = url::Url::parse(base) {
195                url.path().trim_start_matches('/').to_string()
196            } else {
197                base_local.trim_start_matches('/').to_string()
198            }
199        } else {
200            // Fallback: just use the local part
201            base_local.trim_start_matches('/').to_string()
202        };
203
204        // Append filename if needed
205        if !filename.is_empty() {
206            if !rel.is_empty() {
207                rel.push('/');
208            }
209            rel.push_str(&filename);
210        }
211
212        Path::from(rel)
213    }

215    /// Load and merge all .del files into a single RoaringBitmap
216    pub async fn load_merged_deletes(&self) -> Result<RoaringBitmap> {
217        let mut deleted_bitmap = RoaringBitmap::new();
218        
219        // Determine target path for Iceberg delete matching
220        let target_path = if let Some(p) = &self.config.parquet_path {
221            p.clone()
222        } else {
223             format!("{}/{}.parquet", self.config.base_path, self.config.segment_id)
224        };
225
226        for delete_file in &self.config.delete_files {
227             // Handle Position deletes (RoaringBitmap files or Iceberg Parquet)
228             if let crate::core::manifest::DeleteContent::Position = &delete_file.content {
229                 let path_str = delete_file.file_path.as_str();
230                 
231                 // Relativize path
232                 let resolved_path = if path_str.starts_with("file://") {
233                     let root_local = self.root_uri.strip_prefix("file://").unwrap_or(&self.root_uri).trim_end_matches('/');
234                     let path_clean = path_str.strip_prefix("file://").unwrap_or(path_str);
235                     
236                     if !root_local.is_empty() && path_clean.starts_with(root_local) {
237                         path_clean[root_local.len()..].trim_start_matches('/').to_string()
238                     } else {
239                         path_clean.trim_start_matches('/').to_string()
240                     }
241                 } else {
242                     path_str.to_string()
243                 };
244                 
245                 // Check if it is an Iceberg Parquet Delete File
246                 if resolved_path.ends_with(".parquet") || resolved_path.ends_with(".avro") { 
247                     let reader = crate::core::iceberg::PositionDeleteReader::new(self.store.clone());
248                     match reader.read_deletes(&resolved_path, &target_path).await {
249                         Ok(positions) => {
250                             for pos in positions {
251                                 // Safely cast to u32, ignoring if out of bounds (current Roaring limitation)
252                                 if pos >= 0 && pos <= u32::MAX as i64 {
253                                     deleted_bitmap.insert(pos as u32);
254                                 }
255                             }
256                         },
257                         Err(e) => {
258                             tracing::warn!("Failed to read Iceberg delete file {}: {}", path_str, e);
259                         }
260                     }
261                 } else {
262                     // Native .del (RoaringBitmap)
263                     let path = Path::from(path_str);
264                     if let Ok(ret) = self.store.get(&path).await {
265                         if let Ok(bytes) = ret.bytes().await {
266                             if let Ok(bm) = RoaringBitmap::deserialize_from(&bytes[..]) {
267                                 deleted_bitmap |= bm;
268                             }
269                         }
270                     }
271                 }
272             }
273             // Handle V3 Deletion Vectors (Puffin files)
274             else if let crate::core::manifest::DeleteContent::DeletionVector { 
275                 puffin_file_path, 
276                 content_offset, 
277                 content_size_in_bytes 
278             } = &delete_file.content {
279                 // Read the deletion vector blob from the Puffin file
280                 let path = Path::from(puffin_file_path.as_str());
281                 match self.store.get_range(&path, (*content_offset as u64)..((*content_offset + *content_size_in_bytes) as u64)).await {
282                     Ok(bytes) => {
283                         match crate::core::puffin::read_deletion_vector_from_bytes(&bytes) {
284                             Ok(dv_bitmap) => {
285                                 deleted_bitmap |= dv_bitmap;
286                             },
287                             Err(e) => {
288                                 tracing::warn!("Failed to deserialize deletion vector from {}: {}", puffin_file_path, e);
289                             }
290                         }
291                     },
292                     Err(e) => {
293                         tracing::warn!("Failed to read deletion vector from Puffin file {}: {}", puffin_file_path, e);
294                     }
295                 }
296             }
297        }
298        
299        Ok(deleted_bitmap)
300    }

357                                      }

362                          }
363                      }
364                      Err(e) => tracing::warn!("Failed to read equality delete file {}: {}", resolved_path, e),
365                 }
366             }
367        }

371    /// Check if a value might exist in a column using Parquet Bloom Filters
372    pub async fn check_bloom_filter(&self, column: &str, value: &serde_json::Value) -> Result<bool> {
373        let metadata: std::sync::Arc<parquet::file::metadata::ParquetMetaData> = self.get_parquet_metadata().await?;
374        let schema = metadata.file_metadata().schema_descr();
375        
376        let col_idx = schema.columns().iter().position(|c| c.name() == column)
377            .ok_or_else(|| anyhow::anyhow!("Column '{}' not found in Parquet schema", column))?;
378        
379        // Split Block Bloom Filters (SBBF) are used in Parquet.
380        use parquet::bloom_filter::Sbbf;
381        
382        let mut possible = false;
383        let pq_path_str = self.config.parquet_path.clone().unwrap_or_default();
384        if pq_path_str.is_empty() { return Ok(true); }
385        let pq_path = object_store::path::Path::from(pq_path_str.clone());
386
387        // Get file size to handle Range correctly
388        let file_size = if let Some(s) = self.config.file_size {
389            s
390        } else {
391            self.store.head(&pq_path).await?.size as u64
392        };
393
394        for i in 0..metadata.num_row_groups() {
395            let rg_meta = metadata.row_group(i);
396            let col_meta = rg_meta.column(col_idx);
397            
398            if let Some(offset) = col_meta.bloom_filter_offset() {
399                let cache_key = format!("{}/{}:{}", self.root_uri, pq_path_str, offset);
400                
401                let sbbf = if let Some(cached) = crate::core::cache::BLOOM_FILTER_CACHE.get(&cache_key).await {
402                    tracing::debug!("Bloom Cache HIT for key: {}", cache_key);
403                    cached
404                } else {
405                    tracing::debug!("Bloom Cache MISS for key: {}", cache_key);
406                    let start = offset as u64;
407                    let end = (start + 2 * 1024 * 1024).min(file_size);
408                    
409                    // Fetch the Bloom Filter blob from storage
410                    let data_res = self.store.get_range(&pq_path, start..end).await;
411                    let data = match data_res {
412                        Ok(d) => d,
413                        Err(_) => self.store.get_range(&pq_path, start..file_size).await?,
414                    };
415                    
416                    let filter_res = Sbbf::from_bytes(&data);
417                    let filter = match filter_res {
418                        Ok(f) => f,
419                        Err(e) => {
420                            let msg = e.to_string();
421                            if msg.contains("extra bytes") {
422                                // Extract expected length from error: "expected 1048594 total bytes, got ..."
423                                if let Some(expected_pos) = msg.find("expected ") {
424                                    let rest = &msg[expected_pos + 9..];
425                                    if let Some(space_pos) = rest.find(' ') {
426                                        if let Ok(expected_len) = rest[..space_pos].parse::<usize>() {
427                                            if expected_len <= data.len() {
428                                                tracing::debug!("Retrying Bloom Filter parse with truncated length: {}", expected_len);
429                                                Sbbf::from_bytes(&data[..expected_len])?
430                                            } else { return Err(e.into()); }
431                                        } else { return Err(e.into()); }
432                                    } else { return Err(e.into()); }
433                                } else { return Err(e.into()); }
434                            } else { return Err(e.into()); }
435                        }
436                    };
437                    
438                    let arc_filter = Arc::new(filter);
439                    crate::core::cache::BLOOM_FILTER_CACHE.insert(cache_key.clone(), arc_filter.clone()).await;
440                    tracing::debug!("Bloom Cache INSERTED for key: {}", cache_key);
441                    arc_filter
442                };
443
444                let desc = schema.column(col_idx);
445                    let physical_type = desc.physical_type();
446                    
447                    let matches = match physical_type {
448                        parquet::basic::Type::INT64 => {
449                            if let Some(i) = value.as_i64() { sbbf.check(&i) } else { true }
450                        }
451                        parquet::basic::Type::INT32 => {
452                            if let Some(i) = value.as_i64() { 
453                                let i32_val = i as i32;
454                                sbbf.check(&i32_val) 
455                            } else { true }
456                        }
457                        parquet::basic::Type::BYTE_ARRAY | parquet::basic::Type::FIXED_LEN_BYTE_ARRAY => {
458                             if let Some(s) = value.as_str() { 
459                                 // s is &str. &s is &&str. &str implements AsBytes.
460                                 sbbf.check(&s) 
461                             } else if let Some(vals) = value.as_array() {
462                                 let bytes: Vec<u8> = vals.iter().filter_map(|v| v.as_u64().map(|n| n as u8)).collect();
463                                 if bytes.len() == vals.len() {
464                                     sbbf.check(&bytes)
465                                 } else { true }
466                             } else { true }
467                        }
468                        parquet::basic::Type::FLOAT => {
469                            if let Some(f) = value.as_f64() { 
470                                let f32_val = f as f32;
471                                sbbf.check(&f32_val) 
472                            } else { true }
473                        }
474                        parquet::basic::Type::DOUBLE => {
475                            if let Some(f) = value.as_f64() { sbbf.check(&f) } else { true }
476                        }
477                        _ => true,
478                    };
479
480                    if matches {
481                        possible = true;
482                        break;
483                    }
484                }
485            }
486        Ok(possible)
487    }

489    /// Check if a value exists in a column using the best available index
490    pub async fn check_value_exists(&self, column: &str, value: &serde_json::Value) -> Result<bool> {
491        // 1. Bloom Filter (Fastest rejection)
492        if !self.check_bloom_filter(column, value).await? {
493            return Ok(false);
494        }
495        
496        // 2. Inverted Index / Bitmap (Precise lookup)
497        // Convert serde_json::Value to QueryFilter for the index search
498        let filter = crate::core::planner::QueryFilter {
499             column: column.to_string(),
500             min: Some(value.clone()),
501             min_inclusive: true,
502             max: Some(value.clone()),
503             max_inclusive: true,
504             values: None,
505             negated: false,
506        };
507        
508        if let Some(bitmap) = self.get_scalar_filter_bitmap(&filter).await? {
509            return Ok(!bitmap.is_empty());
510        }
511        
512        // 3. Fallback: Full Scan (Slowest)
513        // This is only called if Bloom Filter said "Possible" and no Inverted Index exists.
514        let batches = self.stream_all(None).await?;
515        let mut stream = batches;
516        while let Some(batch_res) = stream.next().await {
517            let batch = batch_res?;
518            let col = batch.column_by_name(column)
519                .ok_or_else(|| anyhow::anyhow!("Column '{}' not found in scan", column))?;
520            
521            // Check rows in batch
522            for i in 0..batch.num_rows() {
523                let val = crate::core::manifest::ManifestValue::from_array(col, i);
524                // Simple comparison (might need type handling optimization)
525                if format!("{}", val) == format!("{}", value).trim_matches('"') {
526                    return Ok(true);
527                }
528            }
529        }
530        
531        Ok(false)
532    }

534    /// Helper: Get RoaringBitmap of rows matching a scalar filter using indexes
535    pub async fn get_scalar_filter_bitmap(&self, filter: &crate::core::planner::QueryFilter) -> Result<Option<RoaringBitmap>> {
536        let filter_column = &filter.column;
537        
538        let inv_idx_info = self.config.index_files.iter()
539            .find(|f| (f.index_type == "inverted" || f.index_type == "bitmap" || f.index_type == "bm25") && f.column_name.as_deref() == Some(filter_column));
540            
541        let matching_bitmap = if let Some(idx_info) = inv_idx_info {
542            let inv_path_str = &idx_info.file_path;
543            let mut dir_path = self.config.parquet_path.clone().unwrap_or_default();
544            if let Some(pos) = dir_path.rfind('/') {
545                dir_path.truncate(pos);
546            } else {
547                dir_path = "".to_string();
548            }
549            
550            let full_inv_path_str = if dir_path.is_empty() || inv_path_str.contains('/') {
551                inv_path_str.clone()
552            } else {
553                format!("{}/{}", dir_path, inv_path_str)
554            };
555
556            // Use Inverted Index (Value -> RowIDs)
557            // 1. Check Object Cache (Decoded RecordBatches)
558            let cache_key = if let Some(offset) = idx_info.offset {
559                 format!("{}/{}:{}", self.root_uri, full_inv_path_str, offset)
560            } else {
561                 format!("{}/{}", self.root_uri, full_inv_path_str)
562            };
563
564            let batches = if let Some(cached) = crate::core::cache::INVERTED_INDEX_CACHE.get(&cache_key).await {
565                cached.as_ref().clone()
566            } else {
567                // Cache Miss - Load from Disk/Byte Cache

701                            }

703                         // Time64 range

766        } else {

774            } else {

778                         let bitmap = RoaringBitmap::deserialize_from(&index_bytes[..])?;

782                     Err(_) => {

784                         return Ok(None);

786                 }
787            }
788        };
789        
790        // Step 1b: Handle Negation

799                // Fallback to full scan by returning None
800                return Ok(None);
801            }
802        }
803
804        // Step 1c: Apply Deletes (Difference)
805        let deleted = self.load_merged_deletes().await?;
806        if !deleted.is_empty() {
807            final_bitmap -= deleted;
808        }
809        
810        Ok(Some(final_bitmap))
811    }

813    /// The "Serverless Selectivity" Filter Query
814    /// 1. Reads ONLY the tiny index file (Range Request or full GET if small).
815    /// 2. Determines matching Row IDs.
816    /// 3. Reads ONLY those rows from Parquet (Range Requests via RowSelection).
817    /// 4. Applies column projection to read only specified columns (skips embeddings etc).
818    pub async fn query_index_first(&self, filter: &crate::core::planner::QueryFilter, target_schema: Option<arrow::datatypes::SchemaRef>) -> Result<Vec<arrow::record_batch::RecordBatch>> {
819        let matching_bitmap = match self.get_scalar_filter_bitmap(filter).await? {
820            Some(bm) => bm,
821            None => return Err(anyhow::anyhow!("No index for column {}", filter.column)),
822        };
823
824
825
826        if matching_bitmap.is_empty() {
827             return Ok(vec![]);
828        }
829
830        // Step 2: Configure Parquet Range Request with Row Selection
831        let pq_path = self.resolve_object_path("parquet");
832        let pq_path_str = pq_path.to_string();
833        
834        let mut builder = if let Some((meta, size)) = crate::core::cache::PARQUET_META_CACHE.get(&format!("{}/{}", self.root_uri, pq_path_str)).await {
835             // Cache Hit
836             let object_meta = ObjectMeta {
837                 location: pq_path.clone(),
838                 last_modified: Utc::now(),
839                 size: size as u64,
840                 e_tag: None,
841                 version: None,
842             };
843             let reader = ParquetObjectReader::new(self.store.clone(), object_meta.location);
844             
845             let options = ArrowReaderOptions::default();
846             let arrow_meta = ArrowReaderMetadata::try_new(meta, options)?;
847             ParquetRecordBatchStreamBuilder::new_with_metadata(reader, arrow_meta)
848        } else {
849              let head_res = if let Some(s) = self.config.file_size {
850                  Ok((None, s as usize))
851              } else {
852                  self.store.head(&pq_path).await.map(|m| {
853                      let s = m.size as usize;
854                      (Some(m), s)
855                  })
856              };
857
858              let (m_opt, size) = match head_res {
859                  Ok(pair) => pair,
860                  Err(e) if e.to_string().contains("not found") || e.to_string().contains("404") => {
861                      return Ok(vec![]); // Segment missing
862                  }
863                  Err(e) => return Err(e.into()),
864              };
865
866              let reader = if let Some(m) = m_opt {
867                  ParquetObjectReader::new(self.store.clone(), m.location)
868              } else {
869                  ParquetObjectReader::new(self.store.clone(), pq_path.clone())
870              };
871
872              let b_res = ParquetRecordBatchStreamBuilder::new(reader).await;
873              let b = match b_res {
874                  Ok(b) => b,
875                  Err(e) if e.to_string().contains("not found") || e.to_string().contains("404") => {
876              
877                      return Ok(vec![]);
878                  }
879                  Err(e) => return Err(e.into()),
880              };
881
882              crate::core::cache::PARQUET_META_CACHE.insert(format!("{}/{}", self.root_uri, pq_path_str), (b.metadata().clone(), size)).await;
883              b
884        };
885        
886        builder = builder.with_batch_size(65536);
887        
888        // Apply column projection if specified (skip reading unused columns like embeddings)
889        // Apply column projection/evolution
890        let target_schema_ref = target_schema.clone();
891        if let Some(schema) = &target_schema_ref {
892            let parquet_schema = builder.metadata().file_metadata().schema_descr();
893            let file_arrow_schema = builder.schema();
894            let column_indices: Vec<usize> = schema.fields().iter()
895                .filter_map(|field| file_arrow_schema.index_of(field.name()).ok())
896                .collect();
897            
898            let projection = ProjectionMask::roots(parquet_schema, column_indices);
899            builder = builder.with_projection(projection);
900        }
901        
902        // Construct RowSelection from Bitmap
903        let selection = self.bitmap_to_row_selection(&matching_bitmap, builder.metadata().file_metadata().num_rows() as usize);
904        builder = builder.with_row_selection(selection);
905
906        let mut stream = builder.build()?;
907        
908        
909        let mut batches = Vec::new();
910        while let Some(batch_result) = stream.next().await {
911            let batch = batch_result?;
912            // Schema Evolution Mapping
913            if let Some(target) = &target_schema_ref {
914                 let mut new_columns = Vec::new();
915                 for field in target.fields() {
916                     if let Ok(col) = batch.column_by_name(field.name()).ok_or(()) {
917                         if col.data_type() != field.data_type() {
918
919                              let casted = arrow::compute::cast(col, field.data_type())?;
920                              new_columns.push(casted);
921                         } else {
922                              new_columns.push(col.clone());
923                         }
924                     } else {
925                         let null_arr = arrow::array::new_null_array(field.data_type(), batch.num_rows());
926                         new_columns.push(null_arr);
927                     }
928                 }
929                batches.push(arrow::record_batch::RecordBatch::try_new(target.clone(), new_columns)?);
930            } else {
931                batches.push(batch);
932            }
933        }
934        
935
936        Ok(batches)
937    }

1071                          // Check which rows match the delete values

1074                               Ok(delete_mask) => {
1075                                   // We want to KEEP rows that are NOT in the delete set
1076                                   if let Ok(not_delete) = arrow::compute::not(&delete_mask) {
1077                                        if let Ok(new_mask) = arrow::compute::and(&keep_mask, &not_delete) {
1078                                            keep_mask = new_mask;
1079                                        }
1080                                   }
1081                               },
1082                               Err(e) => tracing::warn!("Failed to apply equality delete filter on {}: {}", delete.column_name, e),
1083                          }
1084                      }
1085                  }
1086                 
1087                 batch = arrow::compute::filter_record_batch(&batch, &keep_mask)?;
1088             }
1089
1090             Ok(batch)
1091        });
1092        
1093        Ok(mapped_stream.boxed())

1097    /// Returns: Vec<(RecordBatch, Vec<f32>)> where Vec<f32> are distances for each row
1098    /// 
1099    pub async fn vector_search_flat(&self, column: &str, query: &crate::core::index::VectorValue, k: usize, allowed_bitmap: &Option<RoaringBitmap>, metric: VectorMetric) -> Result<Vec<(usize, f32)>> {
1100        let mut stream = self.stream_all(None as Option<arrow::datatypes::SchemaRef>).await?;
1101        let mut matches = Vec::new();
1102        let mut current_row_offset = 0usize;
1103        
1104        let q_vec = match query {
1105            crate::core::index::VectorValue::Float32(v) => v,
1106            _ => anyhow::bail!("Flat search only supports Float32 vectors currently"),
1107        };
1108
1109        while let Some(batch_res) = stream.next().await {
1110            let batch = batch_res?;
1111            let rows = batch.num_rows();
1112            
1113            if let Some(col) = batch.column_by_name(column) {
1114                let vectors: Vec<Vec<f32>> = match col.data_type() {
1115                    arrow::datatypes::DataType::FixedSizeList(_, _) => {
1116                        let list = col.as_any().downcast_ref::<arrow::array::FixedSizeListArray>().unwrap();
1117                        (0..list.len()).map(|i| {
1118                            let item = list.value(i);
1119                            if let Some(floats) = item.as_any().downcast_ref::<arrow::array::Float32Array>() {
1120                                floats.values().to_vec()
1121                            } else if let Some(doubles) = item.as_any().downcast_ref::<arrow::array::Float64Array>() {
1122                                doubles.values().iter().map(|&d| d as f32).collect()
1123                            } else {
1124                                vec![0.0; item.len()]
1125                            }
1126                        }).collect()
1127                    },
1128                    arrow::datatypes::DataType::List(_) => {
1129                        let list = col.as_any().downcast_ref::<arrow::array::ListArray>().unwrap();
1130                        (0..list.len()).map(|i| {
1131                            let item = list.value(i);
1132                            if let Some(floats) = item.as_any().downcast_ref::<arrow::array::Float32Array>() {
1133                                floats.values().to_vec()
1134                            } else if let Some(doubles) = item.as_any().downcast_ref::<arrow::array::Float64Array>() {
1135                                doubles.values().iter().map(|&d| d as f32).collect()
1136                            } else {
1137                                vec![0.0; item.len()]
1138                            }
1139                        }).collect()
1140                    },
1141                    _ => vec![],
1142                };
1143
1144                for (i, v) in vectors.iter().enumerate() {
1145                    let row_id = current_row_offset + i;
1146                    
1147                    // Check filter (allowed_bitmap)
1148                    if let Some(bm) = allowed_bitmap {
1149                        if !bm.contains(row_id as u32) { continue; }
1150                    }
1151
1152                    let dist = match metric {
1153                        VectorMetric::L2 => {
1154                            v.iter().zip(q_vec.iter()).map(|(a,b)| (a-b)*(a-b)).sum::<f32>()
1155                        },
1156                        VectorMetric::Cosine => {
1157                            let dot: f32 = v.iter().zip(q_vec.iter()).map(|(a,b)| a*b).sum();
1158                            let mag_v: f32 = v.iter().map(|x| x*x).sum::<f32>().sqrt();
1159                            let mag_q: f32 = q_vec.iter().map(|x| x*x).sum::<f32>().sqrt();
1160                            1.0 - (dot / (mag_v * mag_q + 1e-10))
1161                        },
1162                        VectorMetric::InnerProduct => {
1163                            -v.iter().zip(q_vec.iter()).map(|(a,b)| a*b).sum::<f32>()
1164                        },
1165                        _ => 1e10, // Not implemented for flat search yet
1166                    };
1167                    matches.push((row_id, dist));
1168                }
1169            }
1170            current_row_offset += rows;
1171        }
1172
1173        // Sort by distance and take k
1174        matches.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
1175        if matches.len() > k {
1176            matches.truncate(k);
1177        }
1178        
1179        Ok(matches)
1180    }

1182    /// Vector search that returns results with distances for global ranking
1183    /// 
1184    /// Supports two index types:
1185    /// 1. HNSW-IVF (hybrid): Checks for `.centroids.parquet` - more memory efficient
1186    /// 2. Plain HNSW: Falls back to `.hnsw.graph` files
1187    pub async fn vector_search_index(&self, column: &str, query: &crate::core::index::VectorValue, k: usize, filter: Option<&FilterExpr>, metric: VectorMetric, ef_search: Option<usize>, target_schema: Option<arrow::datatypes::SchemaRef>) -> Result<Vec<(arrow::record_batch::RecordBatch, Vec<f32>)>> {
1188        // Resolve scalar filter to combined bitmap if present
1189        tracing::debug!("vector_search_index called with filter: {:?}", filter);
1190        let allowed_bitmap = if let Some(expr) = filter {
1191             let sub_filters = expr.extract_and_conditions();
1192             let mut combined_bitmap: Option<RoaringBitmap> = None;
1193             
1194             for sub_f in sub_filters {
1195                 let res = self.get_scalar_filter_bitmap(&sub_f).await;
1196                 if let Ok(Some(bm)) = res {
1197                     match combined_bitmap {
1198                         Some(ref mut existing) => { *existing &= bm; },
1199                         None => { combined_bitmap = Some(bm); }
1200                     }
1201                     
1202                     // Optimization: short-circuit if bitmap is empty
1203                     if let Some(ref bm) = combined_bitmap {
1204                         if bm.is_empty() {
1205                             return Ok(vec![]);
1206                         }
1207                     }
1208                 }
1209             }
1210             combined_bitmap
1211        } else {
1212             None
1213        };
1214        
1215        // Determine Index Path
1216        // If base_path is remote (s3://), we first check if we have the index locally in CWD (common for writers/benchmarks)
1217        // This bypasses the need for full download logic in this iteration.
1218        
1219        // Determine Best Available Vector Index
1220        let vector_indices: Vec<_> = self.config.index_files.iter()
1221            .filter(|f| f.index_type == "vector" && f.column_name.as_deref() == Some(column))
1222            .collect();
1223
1224        let vector_idx_info = if vector_indices.is_empty() {
1225            None
1226        } else {
1227            // Priority: hnsw_tq8 or hnsw_tq4 > hnsw_pq > hnsw_ivf > others
1228            let mut sorted = vector_indices.clone();
1229            sorted.sort_by_key(|f| {
1230                match f.blob_type.as_deref() {
1231                    Some("hnsw_tq8") | Some("hnsw_tq4") => 0,
1232                    Some("hnsw_pq") => 1,
1233                    Some("hnsw_ivf") => 2,
1234                    _ => 3,
1235                }
1236            });
1237            Some(sorted[0])
1238        };
1239
1240        if let crate::core::index::VectorValue::Keyword(ref q) = query {
1241            // BM25 Keyword Search path
1242            let results = self.keyword_search_index(column, q, k, filter).await?;
1243            if results.is_empty() { return Ok(vec![]); }
1244            
1245            let matches: Vec<(u32, f32)> = results.iter().map(|(id, s)| (*id as u32, *s)).collect();
1246            let batch = self.read_rows_by_id_with_schema(matches, target_schema).await?;
1247            
1248            let scores: Vec<f32> = results.into_iter().map(|(_, s)| s).collect();
1249            return Ok(vec![(batch, scores)]);
1250        }
1251
1252        let matches = if let Some(idx_info) = vector_idx_info {
1253             match self.search_hnsw_ivf(idx_info, query, k, &allowed_bitmap, metric, ef_search).await {
1254                 Ok(m) => m,
1255                 Err(e) => {
1256                     // Fallback to flat scan on index error
1257                     tracing::error!("Vector index listed in manifest failed, falling back to flat scan: {}", e);
1258                     self.vector_search_flat(column, query, k, &allowed_bitmap, metric).await?
1259                 }
1260             }
1261        } else {
1262             // No index entry found, try convention path first, then flat scan fallback
1263             let idx_path = self.resolve_object_path(column).to_string();
1264             let idx_info = crate::core::manifest::IndexFile {
1265                 file_path: idx_path,
1266                 index_type: "vector".to_string(),
1267                 column_name: Some(column.to_string()),
1268                 ..Default::default()
1269             };
1270             match self.search_hnsw_ivf(&idx_info, query, k, &allowed_bitmap, metric, ef_search).await {
1271                 Ok(m) => m,
1272                 Err(_) => self.vector_search_flat(column, query, k, &allowed_bitmap, metric).await?
1273             }
1274        };
1275
1276        self.fetch_rows_with_distances(matches, target_schema, filter).await
1277    }

1279    /// Execute Keyword Search (BM25) on this segment
1280    pub async fn keyword_search_index(
1281        &self,
1282        column: &str,
1283        query: &str,
1284        k: usize,
1285        _filter: Option<&FilterExpr>,
1286    ) -> Result<Vec<(usize, f32)>> {
1287        // 1. Find Inverted Index
1288        tracing::info!("keyword_search_index: Segment {} searching for inverted index on column '{}'", self.config.segment_id, column);
1289        tracing::info!("  Available index files: {:?}", self.config.index_files);
1290        
1291        let idx_info = self.config.index_files.iter()
1292            .find(|f| {
1293                let match_type = f.index_type == "inverted" || f.index_type == "bm25";
1294                let match_col = f.column_name.as_deref() == Some(column);
1295                match_type && match_col
1296            })
1297            .ok_or_else(|| anyhow::anyhow!("No keyword/inverted index found for column '{}' in segment {} (Available: {:?})", column, self.config.segment_id, self.config.index_files))?;
1298
1299        // 2. Load Inverted Index Batches
1300        // (Similar to get_scalar_filter_bitmap logic)
1301        let inv_path_str = &idx_info.file_path;
1302        let mut dir_path = self.config.parquet_path.clone().unwrap_or_default();
1303        if let Some(pos) = dir_path.rfind('/') {
1304            dir_path.truncate(pos);
1305        } else {
1306            dir_path = "".to_string();
1307        }
1308        
1309        let full_inv_path_str = if dir_path.is_empty() || inv_path_str.contains('/') {
1310            inv_path_str.clone()
1311        } else {
1312            format!("{}/{}", dir_path, inv_path_str)
1313        };
1314
1315        let cache_key = if let Some(offset) = idx_info.offset {
1316             format!("{}/{}:{}", self.root_uri, full_inv_path_str, offset)
1317        } else {
1318             format!("{}/{}", self.root_uri, full_inv_path_str)
1319        };
1320
1321        let batches = if let Some(cached) = crate::core::cache::INVERTED_INDEX_CACHE.get(&cache_key).await {
1322            cached.as_ref().clone()
1323        } else {
1324            // Cache Miss - Load from Disk
1325            let inv_path = Path::from(full_inv_path_str.as_str());
1326            let inv_bytes = match self.store.get(&inv_path).await {
1327                 Ok(res) => res.bytes().await?.to_vec(),
1328                 Err(e) => return Err(e.into()),
1329            };
1330
1331            let builder = parquet::arrow::arrow_reader::ParquetRecordBatchReaderBuilder::try_new(Bytes::from(inv_bytes))?;
1332            let reader = builder.build()?;
1333            let mut decoded = Vec::new();
1334            for batch_result in reader {
1335                decoded.push(batch_result?);
1336            }
1337            crate::core::cache::INVERTED_INDEX_CACHE.insert(cache_key.clone(), Arc::new(decoded.clone())).await;
1338            decoded
1339        };
1340
1341        // 3. Tokenize Query
1342        // Use the tokenizer defined in the index metadata, fallback to standard
1343        let tokenizer_name = "default";
1344        
1345        let tokenizer = crate::core::index::tokenizer::GLOBAL_TOKENIZER_REGISTRY.get(tokenizer_name)
1346            .unwrap_or_else(|| crate::core::index::tokenizer::GLOBAL_TOKENIZER_REGISTRY.get("standard").unwrap());
1347        
1348        let query_tokens = tokenizer.tokenize(query);
1349        if query_tokens.is_empty() {
1350             return Ok(vec![]);
1351        }
1352
1353        // 4. Scoring Map: RowID -> BM25 Score
1354        let mut scores: std::collections::HashMap<u32, f32> = std::collections::HashMap::new();
1355        let n_total = (self.config.record_count.unwrap_or(0) as f32).max(1.0); // Ensure at least 1.0 for IDF
1356        let k1 = 1.2;
1357
1358        for token in &query_tokens {
1359            // Find this token in the inverted index batches
1360            // Build document term frequencies for scoring
1361            for batch in batches.iter() {
1362                let key_array = batch.column(0).as_any().downcast_ref::<arrow::array::StringArray>()
1363                    .ok_or_else(|| anyhow::anyhow!("Expected StringArray in inverted index keys"))?;
1364                let row_ids_list = batch.column(1).as_any().downcast_ref::<arrow::array::ListArray>()
1365                    .ok_or_else(|| anyhow::anyhow!("Expected ListArray in inverted index row_ids"))?;
1366
1367                for i in 0..batch.num_rows() {
1368                    let key = key_array.value(i);
1369                    if key == token {
1370                        // Found the term!
1371                        let list = row_ids_list.value(i);
1372                        let row_ids = list.as_any().downcast_ref::<arrow::array::UInt32Array>().unwrap();
1373                        
1374                        // Count frequencies by document
1375                        let mut current_doc_counts: std::collections::HashMap<u32, u32> = std::collections::HashMap::new();
1376                        let mut last_id = 0;
1377                        for j in 0..row_ids.len() {
1378                            let rid = last_id + row_ids.value(j);
1379                            *current_doc_counts.entry(rid).or_default() += 1;
1380                            last_id = rid;
1381                        }
1382
1383                        // IDF for this token
1384                        let n_token = current_doc_counts.len() as f32;
1385                        let idf = ((n_total - n_token + 0.5) / (n_token + 0.5) + 1.0).ln();
1386
1387                        // Add to global scores
1388                        for (rid, tf) in current_doc_counts {
1389                            let tf = tf as f32;
1390                            let score_inc = idf * (tf * (k1 + 1.0)) / (tf + k1);
1391                            *scores.entry(rid).or_default() += score_inc;
1392                        }
1393                    }
1394                }
1395            }
1396        }
1397
1398        // 5. Sort by score and convert to (RowID, Distance)
1399        let mut results: Vec<(usize, f32)> = scores.into_iter()
1400            .map(|(rid, score)| (rid as usize, 1.0 / (1.0 + score))) // Convert score to distance-like metric
1401            .collect();
1402        
1403        results.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(std::cmp::Ordering::Equal));
1404        
1405        if results.len() > k {
1406            results.truncate(k);
1407        }
1408
1409        Ok(results)
1410    }

1444    /// Helper to fetch rows from Parquet for a list of matches
1445    async fn fetch_rows_with_distances(
1446        &self, 
1447        matches: Vec<(usize, f32)>, 
1448        target_schema: Option<arrow::datatypes::SchemaRef>,
1449        filter: Option<&FilterExpr>
1450    ) -> Result<Vec<(RecordBatch, Vec<f32>)>> {
1451        if matches.is_empty() {
1452            return Ok(vec![]);
1453        }
1454        
1455        // Build bitmap and track distances
1456        let mut bitmap = RoaringBitmap::new();
1457        let mut row_distances: std::collections::HashMap<u32, f32> = std::collections::HashMap::new();
1458        for (row_id, distance) in matches {
1459            bitmap.insert(row_id as u32);
1460            row_distances.insert(row_id as u32, distance);
1461        }
1462        
1463        // Apply Deletes
1464        let deleted = self.load_merged_deletes().await?;
1465        if !deleted.is_empty() {
1466             bitmap -= &deleted;  // Borrow instead of move
1467             // Remove distances for deleted rows
1468             for row_id in deleted.iter() {
1469                 row_distances.remove(&row_id);
1470             }
1471        }
1472
1473        if bitmap.is_empty() {
1474             return Ok(vec![]);
1475        }
1476        
1477        // Fetch Rows - use resolved path
1478        let pq_path = self.resolve_object_path("parquet");
1479        let pq_path_str = pq_path.to_string();
1480
1481        let mut builder = if let Some((meta, size)) = crate::core::cache::PARQUET_META_CACHE.get(&format!("{}/{}", self.root_uri, pq_path_str)).await {
1482             // Cache Hit
1483             let object_meta = ObjectMeta {
1484                 location: pq_path.clone(),
1485                 last_modified: Utc::now(),
1486                 size: size as u64,
1487                 e_tag: None,
1488                 version: None,
1489             };
1490             let reader = ParquetObjectReader::new(self.store.clone(), object_meta.location);
1491             
1492             let options = ArrowReaderOptions::default();
1493             let arrow_meta = ArrowReaderMetadata::try_new(meta, options)?;
1494             ParquetRecordBatchStreamBuilder::new_with_metadata(reader, arrow_meta)
1495        } else {
1496             // Miss
1497             // Ensure file exists/get meta (HEAD)
1498             let object_meta = self.store.head(&pq_path).await.context("Failed to get segment metadata")?;
1499             let size = object_meta.size;
1500             let reader = ParquetObjectReader::new(self.store.clone(), object_meta.location);
1501             
1502             let b = ParquetRecordBatchStreamBuilder::new(reader).await?;
1503             crate::core::cache::PARQUET_META_CACHE.insert(format!("{}/{}", self.root_uri, pq_path_str), (b.metadata().clone(), size as usize)).await;
1504             b
1505        };
1506        
1507        let selection = self.bitmap_to_row_selection(&bitmap, builder.metadata().file_metadata().num_rows() as usize);
1508        builder = builder.with_row_selection(selection);
1509        
1510        // Apply column projection if specified (skip reading unused columns like embeddings)
1511        let target_schema_ref = target_schema.clone();
1512        if let Some(schema) = &target_schema_ref {
1513            let parquet_schema = builder.metadata().file_metadata().schema_descr();
1514            let file_arrow_schema = builder.schema();
1515            let column_indices: Vec<usize> = schema.fields().iter()
1516                .filter_map(|field| file_arrow_schema.index_of(field.name()).ok())
1517                .collect();
1518            
1519            let projection = ProjectionMask::roots(parquet_schema, column_indices);
1520            builder = builder.with_projection(projection);
1521        }
1522        
1523        let mut stream = builder.build()?;
1524        
1525        // Collect bitmap row IDs into a Vec for O(1) indexing
1526        let row_ids: Vec<u32> = bitmap.iter().collect();
1527        
1528        let mut results = Vec::new();
1529        let mut current_offset = 0usize;
1530        
1531        while let Some(batch_result) = stream.next().await {
1532            let batch = batch_result?;
1533            let num_rows = batch.num_rows();
1534            
1535            // Handle case where RowSelection returns more rows than bitmap
1536            // (due to gap coalescing). Only process rows we have distances for.
1537            let rows_to_process = std::cmp::min(num_rows, row_ids.len() - current_offset);
1538            
1539            if rows_to_process == 0 {
1540                break;
1541            }
1542            
1543            // Extract distances for this batch
1544            let mut batch_distances = Vec::with_capacity(rows_to_process);
1545            for i in 0..rows_to_process {
1546                let row_id = row_ids[current_offset + i];
1547                let distance = row_distances.get(&row_id).copied().unwrap_or(f32::MAX);
1548                batch_distances.push(distance);
1549            }
1550            
1551            let mut final_batch = if rows_to_process < num_rows {
1552                batch.slice(0, rows_to_process)
1553            } else {
1554                batch
1555            };
1556
1557            // Add distance column to batch
1558            let distance_array = std::sync::Arc::new(arrow::array::Float32Array::from(batch_distances.clone()));
1559            let mut fields = final_batch.schema().fields().to_vec();
1560            fields.push(std::sync::Arc::new(arrow::datatypes::Field::new("distance", arrow::datatypes::DataType::Float32, true)));
1561            let mut columns = final_batch.columns().to_vec();
1562            columns.push(distance_array);
1563            final_batch = arrow::record_batch::RecordBatch::try_new(std::sync::Arc::new(arrow::datatypes::Schema::new(fields)), columns)?;
1564            
1565            // Schema Evolution Mapping
1566            if let Some(target) = &target_schema_ref {
1567                 let mut new_columns = Vec::new();
1568                 for field in target.fields() {
1569                     if let Ok(col) = final_batch.column_by_name(field.name()).ok_or(()) {
1570                         if col.data_type() != field.data_type() {
1571                              let casted = arrow::compute::cast(col, field.data_type())?;
1572                              new_columns.push(casted);
1573                         } else {
1574                              new_columns.push(col.clone());
1575                         }
1576                     } else {
1577                         let null_arr = arrow::array::new_null_array(field.data_type(), final_batch.num_rows());
1578                         new_columns.push(null_arr);
1579                     }
1580                 }
1581                final_batch = arrow::record_batch::RecordBatch::try_new(target.clone(), new_columns)?;
1582            }
1583            
1584            current_offset += rows_to_process;
1585
1586            // --- BUG FIX: Apply post-filtering if a filter was provided ---
1587            // This ensures correctness even if the index search was overly permissive (e.g. missing inverted indexes for some terms)
1588            if let Some(expr) = filter {
1589                let planner = crate::core::planner::QueryPlanner::new();
1590                match planner.evaluate_expr(&final_batch, expr) {
1591                    Ok(mask) => {
1592                        // Filter the batch
1593                        let prev_rows = final_batch.num_rows();
1594                        final_batch = arrow::compute::filter_record_batch(&final_batch, &mask)?;
1595                        
1596                        // Filter the distances to match the new batch
1597                        let mut filtered_distances = Vec::with_capacity(final_batch.num_rows());
1598                        for i in 0..prev_rows {
1599                            if mask.value(i) {
1600                                filtered_distances.push(batch_distances[i]);
1601                            }
1602                        }
1603                        batch_distances = filtered_distances;
1604                    },
1605                    Err(_) => {
1606                        // Silently skip if evaluation fails (to maintain backward compatibility with unknown exprs)
1607                    },
1608                }
1609            }
1610
1611            if final_batch.num_rows() > 0 {

1660    /// Convert RoaringBitmap to Parquet RowSelection
1661    fn bitmap_to_row_selection(&self, bitmap: &RoaringBitmap, total_rows: usize) -> RowSelection {
1662        if bitmap.is_empty() {
1663             return RowSelection::from(vec![RowSelector::skip(total_rows)]);
1664        }
1665
1666        let mut selectors = Vec::new();
1667        let mut last_idx = 0;
1668        
1669        // Strategy: Coalesce small gaps to avoid fragmented I/O
1670        // If a gap is smaller than this, we include the rows anyway (trusting the filter to drop them later)
1671        // BUG FIX: For correctness, we must be strict unless we filter post-read.
1672        // Set to 1 to ensure we only merge immediately adjacent rows.
1673        let gap_threshold = 1; 
1674
1675        let mut current_start: Option<usize> = None;
1676        let mut current_end: Option<usize> = None;
1677
1678        for idx in bitmap.iter() {
1679            let idx = idx as usize;
1680            
1681            match current_start {
1682                None => {
1683                    current_start = Some(idx);
1684                    current_end = Some(idx);
1685                }
1686                Some(start) => {
1687                    let end = current_end.unwrap();
1688                    if idx <= end + gap_threshold {
1689                        // Coalesce
1690                        current_end = Some(idx);
1691                    } else {
1692                        // Flush previous range
1693                        if start > last_idx {
1694                            selectors.push(RowSelector::skip(start - last_idx));
1695                        }
1696                        selectors.push(RowSelector::select(end - start + 1));
1697                        last_idx = end + 1;
1698
1699                        current_start = Some(idx);
1700                        current_end = Some(idx);
1701                    }
1702                }
1703            }
1704        }
1705
1706        // Flush last range
1707        if let Some(start) = current_start {
1708            let end = current_end.unwrap();
1709            if start > last_idx {
1710                selectors.push(RowSelector::skip(start - last_idx));
1711            }
1712            selectors.push(RowSelector::select(end - start + 1));
1713            last_idx = end + 1;
1714        }
1715        
1716        if last_idx < total_rows {
1717            selectors.push(RowSelector::skip(total_rows - last_idx));
1718        }
1719
1720        RowSelection::from(selectors)
1721    }
