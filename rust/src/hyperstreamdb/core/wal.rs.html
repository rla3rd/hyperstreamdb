15/// Write-Ahead Log for durability of in-memory writes.
16/// Uses Arrow IPC Stream format for append-only logging.
17/// Supports multiple processes by using unique log files in a shared directory.
18pub struct WriteAheadLog {
19    dir: PathBuf,
20    path: PathBuf,
21    writer: Option<StreamWriter<File>>,
22    schema: Option<SchemaRef>,
23    tx: Option<mpsc::Sender<LogOp>>,
24}
25
26enum LogOp {
27    Append(RecordBatch, oneshot::Sender<Result<()>>),
28}
29

32        f.debug_struct("WriteAheadLog")
33            .field("dir", &self.dir)
34            .field("path", &self.path)
35            .field("schema", &self.schema)
36            .finish()
37    }
38}

41    /// Open or create a WAL directory.
42    pub fn new(dir: impl Into<PathBuf>) -> Self {
43        let dir = dir.into();
44        // Create a unique filename for this instance to avoid clobbering by other processes
45        let id = uuid::Uuid::new_v4();
46        let path = dir.join(format!("log_{}.arrow", id));
47        
48        Self {
49            dir,
50            path,
51            writer: None,
52            schema: None,
53            tx: None,
54        }
55    }

57    /// Start a background worker for asynchronous writes.
58    pub fn spawn_worker(&mut self) -> Result<()> {
59        if self.tx.is_some() {
60            return Ok(());
61        }
62
63        let (tx, mut rx) = mpsc::channel::<LogOp>(1024);
64        self.tx = Some(tx);
65
66        // Move state into worker
67        let path = self.path.clone();
68        let mut writer_opt: Option<StreamWriter<File>> = self.writer.take();
69        
70        tokio::spawn(async move {
71            let mut pending_syncs = Vec::new();
72            let mut batch_count = 0;
73            
74            // Configurable sync batching: sync every N batches or every T milliseconds
75            let sync_batch_size = std::env::var("HYPERSTREAM_WAL_SYNC_BATCH_SIZE")
76            .unwrap_or_else(|_| "10".to_string())
77            .parse::<usize>().unwrap_or(10);
78            let sync_interval_ms: u64 = std::env::var("HYPERSTREAM_WAL_SYNC_INTERVAL_MS")
79                .unwrap_or_else(|_| "100".to_string())
80                .parse()
81                .unwrap_or(100);
82            
83            loop {
84                // Wait for a message or a wait timeout to sync
85                let timeout = tokio::time::sleep(std::time::Duration::from_millis(sync_interval_ms));
86                
87                tokio::select! {
88                    msg = rx.recv() => {
89                        match msg {
90                            Some(LogOp::Append(batch, reply_tx)) => {
91                                // Ensure writer
92                                if writer_opt.is_none() {
93                                    let file = match OpenOptions::new()
94                                        .create(true)
95                                        .append(true)
96                                        .open(&path) {
97                                            Ok(f) => f,
98                                            Err(e) => {
99                                                let _ = reply_tx.send(Err(anyhow::anyhow!("Failed to open WAL: {}", e)));
100                                                continue;
101                                            }
102                                        };
103                                    writer_opt = match StreamWriter::try_new(file, &batch.schema()) {
104                                        Ok(w) => Some(w),
105                                        Err(e) => {
106                                             let _ = reply_tx.send(Err(anyhow::anyhow!("Failed to create WAL writer: {}", e)));
107                                             continue;
108                                        }
109                                    };
110                                }
111
112                                if let Some(writer) = &mut writer_opt {
113                                    if let Err(e) = writer.write(&batch) {
114                                        let _ = reply_tx.send(Err(anyhow::anyhow!("WAL write failed: {}", e)));
115                                    } else {
116                                        pending_syncs.push(reply_tx);
117                                        batch_count += 1;
118                                        
119                                        // Sync if we've reached batch size threshold
120                                        if batch_count >= sync_batch_size {
121                                            // Use fdatasync for better performance (only syncs data, not metadata)
122                                            if let Err(e) = writer.get_ref().sync_data() {
123                                                eprintln!("WAL sync_data failed: {}", e);
124                                                // Fallback to sync_all if fdatasync not available
125                                                let _ = writer.get_ref().sync_all();
126                                            }
127                                            
128                                            // Reply to all pending syncs
129                                            for tx in pending_syncs.drain(..) {
130                                                let _ = tx.send(Ok(()));
131                                            }
132                                            batch_count = 0;
133                                        }
134                                    }
135                                }
136                            }
137                            None => break, // Channel closed
138                        }
139                    }
140                    _ = timeout => {
141                        // Periodic sync - sync any pending writes
142                        if !pending_syncs.is_empty() {
143                            if let Some(writer) = &mut writer_opt {
144                                // Use fdatasync for better performance
145                                if let Err(e) = writer.get_ref().sync_data() {
146                                    eprintln!("WAL sync_data failed: {}", e);
147                                    let _ = writer.get_ref().sync_all();
148                                }
149                            }
150                            for tx in pending_syncs.drain(..) {
151                                let _ = tx.send(Ok(()));
152                            }
153                            batch_count = 0;
154                        }
155                    }
156                }
157            }
158        });
159
160        Ok(())
161    }

163    /// Append a batch to the log asynchronously.
164    pub async fn append_async(&self, batch: RecordBatch) -> Result<()> {
165        if let Some(tx) = &self.tx {
166            let (reply_tx, reply_rx) = oneshot::channel();
167            tx.send(LogOp::Append(batch, reply_tx)).await
168                .map_err(|_| anyhow::anyhow!("WAL worker channel closed"))?;
169            
170            reply_rx.await
171                .map_err(|_| anyhow::anyhow!("WAL worker dropped request"))?
172        } else {
173            // Fallback to sync? Or error?
174            anyhow::bail!("WAL worker not started. Call spawn_worker() first.");
175        }
176    }

178    /// Replay all log files in the WAL directory and return all batches.
179    /// This should be called on startup.
180    pub fn replay(&self) -> Result<(Vec<RecordBatch>, Vec<String>)> {
181        if !self.dir.exists() {
182            return Ok((Vec::new(), Vec::new()));
183        }
184
185        let mut all_batches = Vec::new();
186
187        // 1. List all .arrow files in the directory
188        let entries = std::fs::read_dir(&self.dir)?;
189        let mut wal_files = Vec::new();
190
191        for entry in entries {
192            let entry = entry?;
193            let path = entry.path();
194            if path.is_file() && path.extension().and_then(|s| s.to_str()) == Some("arrow") {
195                wal_files.push(path);
196            }
197        }
198
199        // Sort for deterministic replay (optional but good)
200        wal_files.sort();
201
202        let mut replayed_paths = Vec::new();
203
204        for path in wal_files {
205            let file = File::open(&path)?;
206            if file.metadata()?.len() == 0 {
207                continue;
208            }
209            
210            let reader = BufReader::new(file);
211            let ipc_reader = StreamReader::try_new(reader, None);
212            
213            match ipc_reader {
214                Ok(reader) => {
215                    let mut count = 0;
216                    for batch in reader {
217                        all_batches.push(batch?);
218                        count += 1;
219                    }
220                    if count > 0 {
221                        println!("WAL: Replayed {} batches from {:?}", count, path);
222                        replayed_paths.push(path.to_str().unwrap().to_string());
223                    }
224                },
225                Err(e) => {
226                    println!("WAL Recovery Warning: Could not read log {:?}: {}", path, e);
227                }
228            }
229        }
230
231        Ok((all_batches, replayed_paths))
232    }

234    /// Initialize the writer with a schema.
235    /// Must be called before first append.
236    fn ensure_writer(&mut self, schema: SchemaRef) -> Result<()> {
237        if self.writer.is_some() {
238            return Ok(());
239        }
240
241        self.schema = Some(schema.clone());
242
243        let file = OpenOptions::new()
244            .create(true)
245            .append(true)
246            .open(&self.path)
247            .context("Failed to open WAL file")?;
248
249        // If file is new/empty, we need to write the schema header?
250        // Actually, IPC Stream format writes schema at the start.
251        // But if we are appending to an existing file, we can't just create a new StreamWriter
252        // because it writes a header every time.
253        
254        // Strategy:
255        // For simple WAL, we can just keep the file open.
256        // If we close and reopen, we must be careful.
257        // BUT: Arrow IPC Stream format allows concatenating messages? 
258        // Standard StreamWriter writes schema header.
259        
260        // Better approach for crash recovery:
261        // Always write to a NEW file for a new "session" or just overwrite if we flushed?
262        // Actually, if we are recovering, we read everything, put it in memory, 
263        // and can effectively TRUNCATE the log and start fresh for new writes since they are now in memory.
264        
265        // So:
266        // 1. replay() reads existing data.
267        // 2. truncate() clears the file (since data is now in memory).
268        // 3. append() starts a fresh stream.
269
270        let writer = StreamWriter::try_new(file, &schema)?;
271        self.writer = Some(writer);
272        Ok(())
273    }

275    pub fn append(&mut self, batch: &RecordBatch) -> Result<()> {
276        // Ensure writer exists
277        self.ensure_writer(batch.schema())?;
278        
279        if let Some(writer) = &mut self.writer {
280            writer.write(batch)?;
281            // Sync to disk for durability!
282            writer.get_ref().sync_all()?; 
283        }
284        Ok(())
285    }

287    /// Clear the log file owned by this instance.
288    /// Should be called after data is successfully persisted (flushed) to main storage.
289    pub fn truncate(&mut self) -> Result<()> {
290        self.writer = None; // Drop writer
291        if self.path.exists() {
292             std::fs::remove_file(&self.path)?;
293        }
294        Ok(())
295    }

301                std::fs::remove_file(path)?;
302            }

307    /// Check if WAL should be compacted based on file size
308    pub fn should_compact(&self) -> Result<bool> {
309        if !self.path.exists() {
310            return Ok(false);
311        }
312
313        let metadata = std::fs::metadata(&self.path)?;
314        let size_mb = metadata.len() / (1024 * 1024);
315        
316        let threshold_mb: u64 = std::env::var("HYPERSTREAM_WAL_COMPACT_MB")
317            .unwrap_or_else(|_| "1024".to_string())
318            .parse()
319            .unwrap_or(1024);
320            
321        Ok(size_mb > threshold_mb)
322    }

324    /// Compact WAL by consolidating all batches into one
325    /// This reduces recovery time and file size
326    pub fn compact(&mut self) -> Result<()> {
327        // 1. Replay all batches
328        let (batches, _) = self.replay()?;
329        if batches.is_empty() || batches.len() == 1 {
330            // Already compact or empty
331            return Ok(());
332        }
333
334        // 2. Concatenate batches
335        let schema = batches[0].schema();
336        let consolidated = arrow::compute::concat_batches(&schema, &batches)
337            .context("Failed to concatenate batches during WAL compaction")?;
338
339        // 3. Write to temp file
340        let temp_path = self.path.with_extension("arrow.tmp");
341        let temp_file = File::create(&temp_path)
342            .context("Failed to create temp WAL file")?;
343        let mut temp_writer = StreamWriter::try_new(temp_file, &schema)?;
344        temp_writer.write(&consolidated)?;
345        temp_writer.finish()?;
346        drop(temp_writer);
347
348        // 4. Atomic replace
349        std::fs::rename(&temp_path, &self.path)
350            .context("Failed to replace WAL file")?;
351
352        // 5. Reset writer (will be recreated on next append)
353        self.writer = None;
354        
355        println!("WAL: Compacted {} batches into 1 batch", batches.len());
356        Ok(())
357    }
