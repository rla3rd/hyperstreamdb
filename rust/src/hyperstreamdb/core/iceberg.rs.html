148    let avro_reader = Reader::new(reader)?;
149    let mut entries = Vec::new();
150

154            let mut status = 0;

156            let mut data_file = None;
157

159                match name.as_str() {
160                    "status" => if let AvroValue::Int(s) = val { status = s; },
161                    "snapshot_id" => if let AvroValue::Long(s) = val { snapshot_id = Some(s); },
162                    "data_file" => {
163                        if let AvroValue::Record(df_fields) = val {
164                            data_file = Some(parse_data_file(df_fields)?);
165                        }
166                    },
167                    _ => {}
168                }
169            }
170
171            if let Some(df) = data_file {
172                entries.push(IcebergManifestEntry {
173                    status,
174                    snapshot_id,
175                    data_file: df,
176                });
177            }
178        }
179    }
180    Ok(entries)

182fn parse_data_file(fields: Vec<(String, AvroValue)>) -> Result<IcebergDataFile> {
183    let mut content = 0;
184    let mut file_path = String::new();
185    let mut file_format = String::new();
186    let mut partition = Vec::new();
187    let mut record_count = 0;
188    let mut file_size_in_bytes = 0;
189    
190    let mut column_sizes = None;
191    let mut value_counts = None;
192    let mut null_value_counts = None;
193    let mut nan_value_counts = None;
194    let mut lower_bounds = None;
195    let mut upper_bounds = None;
196    let mut equality_ids = None;
197    
198    // V3 Deletion Vector fields
199    let mut referenced_data_file = None;
200    let mut content_offset = None;
201    let mut content_size_in_bytes = None;
202    let mut index_files = None;
203
204    for (name, val) in fields {
205        match name.as_str() {
206            "content" => if let AvroValue::Int(c) = val { content = c; },
207            "file_path" => if let AvroValue::String(s) = val { file_path = s; },
208            "file_format" => if let AvroValue::String(s) = val { file_format = s; },
209            "partition" => if let AvroValue::Record(p_fields) = val {
210                partition = p_fields.into_iter().map(|(_, v)| avro_to_json(v)).collect();
211            },
212            "record_count" => if let AvroValue::Long(c) = val { record_count = c; },
213            "file_size_in_bytes" => if let AvroValue::Long(s) = val { file_size_in_bytes = s; },
214            "column_sizes" => column_sizes = parse_map_int_long(val),
215            "value_counts" => value_counts = parse_map_int_long(val),
216            "null_value_counts" => null_value_counts = parse_map_int_long(val),
217            "nan_value_counts" => nan_value_counts = parse_map_int_long(val),
218            "lower_bounds" => lower_bounds = parse_map_int_bytes(val),
219            "upper_bounds" => upper_bounds = parse_map_int_bytes(val),
220            "equality_ids" => {
221                let inner = if let AvroValue::Union(_, b) = val { *b } else { val };
222                if let AvroValue::Array(items) = inner {
223                    let mut ids = Vec::new();
224                    for item in items {
225                        if let AvroValue::Int(id) = item {
226                            ids.push(id);
227                        }
228                    }
229                    equality_ids = Some(ids);
230                }
231            },
232            // V3 Deletion Vector fields
233            "referenced_data_file" => {
234                let inner = if let AvroValue::Union(_, b) = val { *b } else { val };
235                if let AvroValue::String(s) = inner {
236                    referenced_data_file = Some(s);
237                }
238            },
239            "content_offset" => {
240                let inner = if let AvroValue::Union(_, b) = val { *b } else { val };
241                if let AvroValue::Long(o) = inner {
242                    content_offset = Some(o);
243                }
244            },
245            "content_size_in_bytes" => {
246                let inner = if let AvroValue::Union(_, b) = val { *b } else { val };
247                if let AvroValue::Long(s) = inner {
248                    content_size_in_bytes = Some(s);
249                }
250            },
251            "index_files" => {
252                let inner = if let AvroValue::Union(_, b) = val { *b } else { val };
253                if let AvroValue::String(s) = inner {
254                    index_files = Some(s);
255                }
256            },
257            _ => {}
258        }
259    }
260
261    Ok(IcebergDataFile {
262        content,
263        file_path,
264        file_format,
265        partition,
266        record_count,
267        file_size_in_bytes,
268        column_sizes,
269        value_counts,
270        null_value_counts,
271        nan_value_counts,
272        lower_bounds,
273        upper_bounds,
274        equality_ids,
275        referenced_data_file,
276        content_offset,
277        content_size_in_bytes,
278        index_files,
279    })
280}

282fn parse_map_int_long(val: AvroValue) -> Option<std::collections::HashMap<i32, i64>> {
283    if let AvroValue::Array(items) = val {
284        let mut map = std::collections::HashMap::new();
285        for item in items {
286            if let AvroValue::Record(fields) = item {
287                let mut key = 0;
288                let mut value = 0;
289                for (k, v) in fields {
290                    match k.as_str() {
291                        "key" => if let AvroValue::Int(i) = v { key = i; },
292                        "value" => if let AvroValue::Long(l) = v { value = l; },
293                        _ => {}
294                    }
295                }
296                map.insert(key, value);
297            }
298        }
299        return Some(map);
300    }
301    None
302}

330        AvroValue::Int(i) => serde_json::json!(i),

341pub fn decode_iceberg_value(type_json: &serde_json::Value, bytes: &[u8]) -> crate::core::manifest::ManifestValue {
342    use crate::core::manifest::ManifestValue;
343    
344    let type_str = if let Some(s) = type_json.as_str() {
345        s
346    } else if let Some(obj) = type_json.as_object() {
347        obj.get("type").and_then(|t| t.as_str()).unwrap_or("unknown")
348    } else {
349        "unknown"
350    };
351
352    match type_str {
353        "boolean" => {
354            if !bytes.is_empty() {
355                ManifestValue::Boolean(bytes[0] != 0)
356            } else {
357                ManifestValue::Null
358            }
359        }
360        "int" | "date" => {
361            if bytes.len() >= 4 {
362                ManifestValue::Int32(i32::from_le_bytes(bytes[0..4].try_into().unwrap()))
363            } else {
364                ManifestValue::Null
365            }
366        }
367        "long" | "timestamp" | "timestamptz" => {
368            if bytes.len() >= 8 {
369                ManifestValue::Int64(i64::from_le_bytes(bytes[0..8].try_into().unwrap()))
370            } else {
371                ManifestValue::Null
372            }
373        }
374        "float" => {
375            if bytes.len() >= 4 {
376                ManifestValue::Float32(f32::from_le_bytes(bytes[0..4].try_into().unwrap()))
377            } else {
378                ManifestValue::Null
379            }
380        }
381        "double" => {
382            if bytes.len() >= 8 {
383                ManifestValue::Float64(f64::from_le_bytes(bytes[0..8].try_into().unwrap()))
384            } else {
385                ManifestValue::Null
386            }
387        }
388        "string" | "uuid" => {
389            if let Ok(s) = std::str::from_utf8(bytes) {
390                ManifestValue::String(s.to_string())
391            } else {
392                ManifestValue::String(base64::engine::general_purpose::STANDARD.encode(bytes))
393            }
394        }
395        "binary" | "fixed" => {
396            ManifestValue::String(base64::engine::general_purpose::STANDARD.encode(bytes))
397        }
398        _ => {
399            // Best effort fallback
400            parse_avro_value_bytes(bytes)
401        }
402    }
403}

405pub fn parse_avro_value_bytes(bytes: &[u8]) -> crate::core::manifest::ManifestValue {
406    // Iceberg stores min/max bounds as the serialized binary of the type.
407    // Without strict type info passed down, we might struggle.
408    // For now, let's attempt to guess or store as stringified bytes?
409    // Actually, `ManifestValue` is a wrapper around serde_json::Value.
410    // If it's a string, it's just bytes. If it's an int, it's little endian bytes (usually).
411    // Let's implement a best-effort simple one or just wrap bytes.
412    // TODO: Pass type context for better parsing.
413    
414    if bytes.len() == 4 {
415        // Could be int or float
416        let val = i32::from_le_bytes(bytes.try_into().unwrap());
417        crate::core::manifest::ManifestValue::Int64(val as i64)
418    } else if bytes.len() == 8 {
419        // Could be long or double
420        let val = i64::from_le_bytes(bytes.try_into().unwrap());
421        crate::core::manifest::ManifestValue::Int64(val)
422    } else {
423        // String or Binary
424        if let Ok(s) = std::str::from_utf8(bytes) {
425            crate::core::manifest::ManifestValue::String(s.to_string())
426        } else {
427             // Fallback
428             crate::core::manifest::ManifestValue::String(base64::engine::general_purpose::STANDARD.encode(bytes))
429        }
430    }
431}

433/// Convert Iceberg JSON schema to Arrow Schema
434pub fn iceberg_json_to_arrow_schema(schema_json: &serde_json::Value) -> Result<arrow::datatypes::SchemaRef> {
435    use arrow::datatypes::{Schema, Field};
436    use std::sync::Arc;
437
438    let fields_json = schema_json.get("fields")
439        .and_then(|f| f.as_array())
440        .ok_or_else(|| anyhow::anyhow!("Invalid Iceberg schema: missing fields"))?;
441
442    let mut fields = Vec::new();
443
444    for field in fields_json {
445        let name = field.get("name").and_then(|n| n.as_str()).unwrap_or("unknown");
446        let type_json = field.get("type").ok_or_else(|| anyhow::anyhow!("Field missing type"))?;
447        let required = field.get("required").and_then(|r| r.as_bool()).unwrap_or(false);
448        let id = field.get("id").and_then(|i| i.as_i64()).unwrap_or(0);
449
450        let dt = convert_iceberg_type_to_arrow(type_json)?;
451        let mut arrow_field = Field::new(name, dt, !required);
452        
453        // Store Iceberg ID in metadata
454        if id > 0 {
455            arrow_field.set_metadata(std::collections::HashMap::from([
456                ("iceberg.id".to_string(), id.to_string())
457            ]));
458        }
459        
460        fields.push(arrow_field);
461    }
462
463    Ok(Arc::new(Schema::new(fields)))
464}

466fn convert_iceberg_type_to_arrow(type_json: &serde_json::Value) -> Result<arrow::datatypes::DataType> {
467    use arrow::datatypes::{DataType, TimeUnit};
468
469    if let Some(type_str) = type_json.as_str() {
470        match type_str {
471            "boolean" => Ok(DataType::Boolean),
472            "int" => Ok(DataType::Int32),
473            "long" => Ok(DataType::Int64),
474            "float" => Ok(DataType::Float32),
475            "double" => Ok(DataType::Float64),
476            "string" => Ok(DataType::Utf8),
477            "binary" | "fixed" => Ok(DataType::Binary), // Simplified
478            "date" => Ok(DataType::Date32),
479            "timestamp" => Ok(DataType::Timestamp(TimeUnit::Microsecond, None)),
480            "timestamptz" => Ok(DataType::Timestamp(TimeUnit::Microsecond, Some("UTC".into()))),
481            "uuid" => Ok(DataType::Utf8), // Arrow doesn't have native UUID
482            _ => {
483                // Check for complex types like list, map, struct which might be encoded as strings in some contexts?
484                // Usually they are objects.
485                 Err(anyhow::anyhow!("Unsupported primitive type: {}", type_str))
486            }
487        }
488    } else if let Some(obj) = type_json.as_object() {
489        // Complex types: list, map, struct
490        let type_name = obj.get("type").and_then(|t| t.as_str()).unwrap_or("");
491        match type_name {
492            "struct" => {
493                 let fields = obj.get("fields")
494                    .and_then(|f| f.as_array())
495                    .map(|v| v.as_slice())
496                    .unwrap_or(&[]);
497                 let mut arrow_fields = Vec::new();
498                 for f in fields {
499                     let name = f.get("name").and_then(|n| n.as_str()).unwrap_or("unknown");
500                     let field_type = f.get("type").unwrap();
501                     let required = f.get("required").and_then(|r| r.as_bool()).unwrap_or(false);
502                     arrow_fields.push(arrow::datatypes::Field::new(name, convert_iceberg_type_to_arrow(field_type)?, !required));
503                 }
504                 Ok(DataType::Struct(arrow_fields.into()))
505            },
506            "list" => {
507                let element_type = obj.get("element").unwrap_or(obj.get("element-type").unwrap()); // 'element' or 'element-type'
508                let required = obj.get("element-required").and_then(|r| r.as_bool()).unwrap_or(true); // Default true?
509                let dt = convert_iceberg_type_to_arrow(element_type)?;
510                Ok(DataType::List(std::sync::Arc::new(arrow::datatypes::Field::new("item", dt, !required))))
511            },
512            "decimal" => {
513                let precision = obj.get("precision").and_then(|p| p.as_u64()).unwrap_or(38) as u8;
514                let scale = obj.get("scale").and_then(|s| s.as_u64()).unwrap_or(10) as i8;
515                Ok(DataType::Decimal128(precision, scale))
516            },
517            "map" => {
518                let key_type = obj.get("key").unwrap_or(obj.get("key-type").unwrap());
519                let value_type = obj.get("value").unwrap_or(obj.get("value-type").unwrap());
520                let value_required = obj.get("value-required").and_then(|r| r.as_bool()).unwrap_or(true);
521                
522                let kt = convert_iceberg_type_to_arrow(key_type)?;
523                let vt = convert_iceberg_type_to_arrow(value_type)?;
524                
525                Ok(DataType::Map(
526                    std::sync::Arc::new(Field::new("entries", DataType::Struct(vec![
527                        Field::new("key", kt, false), // Keys are usually non-nullable in Iceberg
528                        Field::new("value", vt, !value_required),
529                    ].into()), false)),
530                    false // Not sorted by default
531                ))
532            },
533            _ => Err(anyhow::anyhow!("Unsupported complex type: {:?}", type_json))
534        }
535    } else {
536        Err(anyhow::anyhow!("Invalid types definition: {:?}", type_json))
537    }
538}

540pub fn iceberg_partition_spec_to_hyperstream(
541    spec_json: &serde_json::Value,
542) -> Result<crate::core::manifest::PartitionSpec> {
543    use crate::core::manifest::{PartitionSpec, PartitionField};
544    
545    let spec_id = spec_json.get("spec-id").and_then(|id| id.as_i64()).unwrap_or(0) as i32;
546    let fields_json = spec_json.get("fields")
547        .and_then(|f| f.as_array())
548        .ok_or_else(|| anyhow::anyhow!("Invalid Iceberg partition spec: missing fields"))?;
549
550    let mut fields = Vec::new();
551    for field in fields_json {
552        let name = field.get("name").and_then(|n| n.as_str()).unwrap_or("unknown").to_string();
553        let source_id = field.get("source-id").and_then(|id| id.as_i64()).unwrap_or(0) as i32;
554        let transform = field.get("transform").and_then(|t| t.as_str()).unwrap_or("identity").to_string();
555        let field_id = field.get("field-id").and_then(|id| id.as_i64()).map(|id| id as i32);
556        
557        
558        fields.push(PartitionField::new_single(
559            source_id,
560            field_id,
561            name,
562            transform,
563        ));
564    }
565
566    Ok(PartitionSpec {
567        spec_id,
568        fields,
569    })
570}

583    use crate::core::manifest::{ManifestEntry, DeleteFile, DeleteContent, ColumnStats};
584    use std::collections::HashMap;
585
586    let df = &iceberg_entry.data_file;
587    
588    // Process partition values
589    let mut partition_values = HashMap::new();
590    if !df.partition.is_empty() && partition_spec.fields.len() == df.partition.len() {
591        for (i, p_field) in partition_spec.fields.iter().enumerate() {
592            partition_values.insert(p_field.name.clone(), df.partition[i].clone());
593        }
594    }
595
596    if df.content == 0 {
597        // Data File
598        let mut column_stats = HashMap::new();
599        let field_map: HashMap<i32, &crate::core::manifest::SchemaField> = schema.fields.iter()
600            .map(|f| (f.id, f))
601            .collect();
602
603        if let (Some(lowers), Some(uppers), Some(nulls)) = (&df.lower_bounds, &df.upper_bounds, &df.null_value_counts) {
604            for (id, lower_bytes) in lowers {
605                if let Some(field) = field_map.get(id) {
606                    let min_val = decode_iceberg_value(&serde_json::json!(field.type_str), lower_bytes); 
607                    let max_val = uppers.get(id).map(|upper_bytes| decode_iceberg_value(&serde_json::json!(field.type_str), upper_bytes));
608                    let null_count = *nulls.get(id).unwrap_or(&0);
609                    column_stats.insert(field.name.clone(), ColumnStats {
610                        min: Some(min_val),
611                        max: max_val,
612                        null_count,
613                        ..Default::default()
614                    });
615                }
616            }
617        }
618
619        let index_files = if let Some(idx_json) = &df.index_files {
620            serde_json::from_str(idx_json).unwrap_or_default()
621        } else {
622            Vec::new()
623        };
624        
625        Ok(IcebergManifestObject::Data(ManifestEntry {
626            file_path: df.file_path.clone(),
627            file_size_bytes: df.file_size_in_bytes,
628            record_count: df.record_count,
629            column_stats,
630            partition_values,
631            index_files,
632            ..Default::default()
633        }))
634    } else {
635        // Delete File
636        let content = if df.content == 1 {
637            DeleteContent::Position
638        } else if df.content == 2 {
639            DeleteContent::Equality {
640                equality_ids: df.equality_ids.clone().unwrap_or_default()
641            }
642        } else if df.content == 3 {
643            // V3 Deletion Vector (content=3)
644            // Check if we have the required DV fields
645            if let (Some(_ref_file), Some(offset), Some(size)) = (
646                df.referenced_data_file.clone(),
647                df.content_offset,
648                df.content_size_in_bytes
649            ) {
650                DeleteContent::DeletionVector {
651                    puffin_file_path: df.file_path.clone(),
652                    content_offset: offset,
653                    content_size_in_bytes: size,
654                }
655            } else {
656                // Fallback to position delete if DV fields are missing
657                DeleteContent::Position
658            }
659        } else {
660            // Unknown content type, default to position
661            DeleteContent::Position
662        };
663
664        Ok(IcebergManifestObject::Delete(DeleteFile {
665            file_path: df.file_path.clone(),
666            content,
667            file_size_bytes: df.file_size_in_bytes,
668            record_count: df.record_count,
669            partition_values,
670        }))
671    }
672}

684        let is_avro = path.ends_with(".avro");
685        let path_obj = object_store::path::Path::from(path);
686        let res = self.store.get(&path_obj).await?;
687        let bytes = res.bytes().await?;
688        
689        if is_avro {

696    fn read_deletes_avro(&self, bytes: bytes::Bytes, target_data_file_path: &str) -> Result<HashSet<i64>> {
697        let reader = apache_avro::Reader::new(&bytes[..])?;
698        let mut deleted_positions = HashSet::new();
699
700        for record in reader {
701            let value = record?;
702            if let apache_avro::types::Value::Record(fields) = value {
703                let mut file_path = None;
704                let mut pos = None;
705
706                for (name, val) in fields {
707                    match name.as_str() {
708                        "file_path" => if let apache_avro::types::Value::String(s) = val { file_path = Some(s); },
709                        "pos" => if let apache_avro::types::Value::Long(p) = val { pos = Some(p); },
710                        _ => {}
711                    }
712                }
713
714                if let (Some(fp), Some(p)) = (file_path, pos) {
715                    let fp_clean = fp.replace("file://", "");
716                    let target_clean = target_data_file_path.replace("file://", "");
717                    
718                    if fp_clean == target_clean || target_clean.ends_with(&fp_clean) || fp_clean.ends_with(&target_clean) {
719                        deleted_positions.insert(p);
720                    }
721                }
722            }
723        }
724        Ok(deleted_positions)
725    }
726
727    fn read_deletes_parquet(&self, bytes: bytes::Bytes, target_data_file_path: &str) -> Result<HashSet<i64>> {
728        use arrow::array::{StringArray, Int64Array};
729        use parquet::arrow::arrow_reader::ParquetRecordBatchReaderBuilder;
730        let cursor = bytes;
731        let builder = ParquetRecordBatchReaderBuilder::try_new(cursor)?;
732        let reader = builder.build()?;
733        
734        let mut deleted_positions = HashSet::new();
735        
736        for batch_res in reader {
737            let batch = batch_res?;
738            
739            if let (Ok(file_path_col), Ok(pos_col)) = (batch.column_by_name("file_path").ok_or(()), batch.column_by_name("pos").ok_or(())) {
740                let file_paths = file_path_col.as_any().downcast_ref::<StringArray>()
741                    .ok_or_else(|| anyhow::anyhow!("file_path column is not string"))?;
742                let positions = pos_col.as_any().downcast_ref::<Int64Array>()
743                    .ok_or_else(|| anyhow::anyhow!("pos column is not int64"))?;
744                
745                for i in 0..batch.num_rows() {
746                    let fp = file_paths.value(i);
747                    let fp_clean = fp.replace("file://", "");
748                    let target_clean = target_data_file_path.replace("file://", "");
749                    
750                    if fp_clean == target_clean || target_clean.ends_with(&fp_clean) || fp_clean.ends_with(&target_clean) {
751                        deleted_positions.insert(positions.value(i));
752                    }
753                }
754            }
755        }
756        Ok(deleted_positions)
757    }

791        &self,
792        bytes: bytes::Bytes,
793        equality_ids: &[i32],
794        schema: &crate::core::manifest::Schema
795    ) -> Result<Vec<RecordBatch>> {
796        use std::collections::HashMap;
797        use arrow::array::*;
798        use arrow::datatypes::{Field, Schema as ArrowSchema};
799        use std::sync::Arc;
800
801        let reader = apache_avro::Reader::new(&bytes[..])?;
802        
803        let mut column_names = Vec::new();
804        let field_map: HashMap<i32, &crate::core::manifest::SchemaField> = schema.fields.iter()
805            .map(|f| (f.id, f))
806            .collect();
807
808        for &id in equality_ids {
809            if let Some(field) = field_map.get(&id) {
810                column_names.push(field.name.clone());
811            } else {
812                return Err(anyhow::anyhow!("Equality Delete ID {} not found in schema", id));
813            }
814        }
815
816        // Build columns
817        let mut columns_data: Vec<Vec<apache_avro::types::Value>> = vec![Vec::new(); equality_ids.len()];
818        let mut row_count = 0;
819
820        for record in reader {
821            let value = record?;
822            if let apache_avro::types::Value::Record(fields) = value {
823                let field_map_record: HashMap<String, apache_avro::types::Value> = fields.into_iter().collect();
824                
825                for (idx, name) in column_names.iter().enumerate() {
826                    if let Some(val) = field_map_record.get(name) {
827                        columns_data[idx].push(val.clone());
828                    } else {
829                         columns_data[idx].push(apache_avro::types::Value::Null);
830                    }
831                }
832                row_count += 1;
833            }
834        }
835
836        if row_count == 0 {
837            return Ok(vec![]);
838        }
839
840        // Convert to Arrow RecordBatch
841        let mut arrow_columns = Vec::new();
842        let mut arrow_fields = Vec::new();
843
844        for (idx, name) in column_names.iter().enumerate() {
845            let field = field_map.get(&equality_ids[idx]).unwrap();
846            let arrow_field = Field::new(name, self.map_type_to_arrow(&field.type_str), true);
847            arrow_fields.push(arrow_field);
848            
849            let array = self.avro_to_arrow_array(&columns_data[idx], &field.type_str)?;
850            arrow_columns.push(array);
851        }
852
853        let arrow_schema = Arc::new(ArrowSchema::new(arrow_fields));
854        let batch = RecordBatch::try_new(arrow_schema, arrow_columns)?;
855        Ok(vec![batch])
856    }

945    fn read_equality_deletes_parquet(
946        &self,
947        bytes: bytes::Bytes,
948        equality_ids: &[i32],
949        schema: &crate::core::manifest::Schema
950    ) -> Result<Vec<RecordBatch>> {
951        use std::collections::HashMap;
952        use parquet::arrow::arrow_reader::ParquetRecordBatchReaderBuilder;
953        
954
955        let cursor = bytes;
956
957        // 1. Resolve column names from IDs
958        let mut column_names = Vec::new();
959        let field_map: HashMap<i32, &crate::core::manifest::SchemaField> = schema.fields.iter()
960            .map(|f| (f.id, f))
961            .collect();
962
963        for &id in equality_ids {
964            if let Some(field) = field_map.get(&id) {
965                column_names.push(field.name.clone());
966            } else {
967                // If ID not found in schema, we can't read it. Ideally shouldn't happen for valid manifests.
968                return Err(anyhow::anyhow!("Equality Delete ID {} not found in schema", id));
969            }
970        }
971
972        // 2. Open Parquet Reader
973        let builder = ParquetRecordBatchReaderBuilder::try_new(cursor)?;
974        let arrow_schema = builder.schema();
975        
976        // 3. Find projection mask indices
977        let mut indices = Vec::new();
978        for col_name in &column_names {
979            if let Ok(idx) = arrow_schema.index_of(col_name) {
980                indices.push(idx);
981            } else {
982                // If column missing in delete file, it implies no deletes for that ID? 
983                // Or maybe partial schema match. For now, strict requirement.
984                return Err(anyhow::anyhow!("Column {} not found in equality delete file", col_name));
985            }
986        }
987        
988        // 4. Project and Read
989        let projection = parquet::arrow::ProjectionMask::roots(builder.parquet_schema(), indices);
990        let reader = builder.with_projection(projection).build()?;
991
992        let mut batches = Vec::new();
993        for batch_res in reader {
994            batches.push(batch_res?);
995        }
996
997        Ok(batches)
998    }

1325    pub fn parse(s: &str) -> Self {
1326        match s {
1327            "identity" => Self::Identity,
1328            "year" => Self::Year,
1329            "month" => Self::Month,
1330            "day" => Self::Day,
1331            "hour" => Self::Hour,
1332            "void" => Self::Void,
1333            _ if s.starts_with("bucket[") => {

1335                 Self::Bucket(n)
1336            },
1337            _ if s.starts_with("truncate[") => {
1338                 let n = s.trim_start_matches("truncate[").trim_end_matches(']').parse().unwrap_or(0);

1341            _ => Self::Void
1342        }

1348    }

1350    pub fn apply_multi(&self, arrays: &[&dyn arrow::array::Array], row_i: usize) -> serde_json::Value {
1351        if arrays.is_empty() { return serde_json::Value::Null; }
1352        if arrays.iter().any(|a| a.is_null(row_i)) { return serde_json::Value::Null; }
1353        
1354        use chrono::Datelike;
1355
1356        match self {
1357            Self::Identity => {
1358                let array = arrays[0];
1359                match array.data_type() {
1360                    arrow::datatypes::DataType::Int32 => {
1361                        let a = array.as_any().downcast_ref::<arrow::array::Int32Array>().unwrap();
1362                        serde_json::json!(a.value(row_i))
1363                    },
1364                    arrow::datatypes::DataType::Int64 => {
1365                        let a = array.as_any().downcast_ref::<arrow::array::Int64Array>().unwrap();
1366                        serde_json::json!(a.value(row_i))
1367                    },
1368                    arrow::datatypes::DataType::Utf8 => {
1369                        let a = array.as_any().downcast_ref::<arrow::array::StringArray>().unwrap();
1370                        serde_json::json!(a.value(row_i))
1371                    },
1372                    arrow::datatypes::DataType::Date32 => {
1373                        let a = array.as_any().downcast_ref::<arrow::array::Date32Array>().unwrap();
1374                        serde_json::json!(a.value(row_i))
1375                    },
1376                    arrow::datatypes::DataType::Timestamp(_, _) => {
1377                         let a = array.as_any().downcast_ref::<arrow::array::TimestampMicrosecondArray>().unwrap();
1378                         serde_json::json!(a.value(row_i))
1379                    },
1380                    _ => serde_json::Value::Null
1381                }
1382            },
1383            Self::Bucket(n) => {
1384                let mut hash_val: u32 = 0;
1385                for array in arrays {
1386                    let field_hash = match array.data_type() {
1387                        arrow::datatypes::DataType::Int32 => {
1388                            let a = array.as_any().downcast_ref::<arrow::array::Int32Array>().unwrap();
1389                            let val = a.value(row_i);
1390                            let bytes = (val as i64).to_le_bytes();
1391                            murmur3_32_x86(&bytes, hash_val)
1392                        },
1393                        arrow::datatypes::DataType::Int64 => {
1394                            let a = array.as_any().downcast_ref::<arrow::array::Int64Array>().unwrap();
1395                            let val = a.value(row_i);
1396                            let bytes = val.to_le_bytes();
1397                            murmur3_32_x86(&bytes, hash_val)
1398                        },
1399                        arrow::datatypes::DataType::Utf8 => {
1400                            let a = array.as_any().downcast_ref::<arrow::array::StringArray>().unwrap();
1401                            let s = a.value(row_i);
1402                            murmur3_32_x86(s.as_bytes(), hash_val)
1403                        },
1404                        arrow::datatypes::DataType::Date32 => {
1405                            let a = array.as_any().downcast_ref::<arrow::array::Date32Array>().unwrap();
1406                            let val = a.value(row_i);
1407                            let bytes = (val as i64).to_le_bytes();
1408                            murmur3_32_x86(&bytes, hash_val)
1409                        },
1410                        _ => 0,
1411                    };
1412                    hash_val = field_hash;
1413                }
1414                // Iceberg Bucketing: (hash & Integer.MAX_VALUE) % N
1415                serde_json::json!((hash_val & 0x7FFFFFFF) % *n)
1416            },
1417            Self::Truncate(w) => {
1418                let array = arrays[0];
1419                match array.data_type() {
1420                    arrow::datatypes::DataType::Utf8 => {
1421                        let a = array.as_any().downcast_ref::<arrow::array::StringArray>().unwrap();
1422                        let s = a.value(row_i);
1423                        let limit = (*w as usize).min(s.len());
1424                        serde_json::json!(&s[..limit])
1425                    },
1426                    arrow::datatypes::DataType::Int32 => {
1427                        let a = array.as_any().downcast_ref::<arrow::array::Int32Array>().unwrap();
1428                        let v = a.value(row_i);
1429                        serde_json::json!(v - (v % (*w as i32)))
1430                    },
1431                    arrow::datatypes::DataType::Int64 => {
1432                        let a = array.as_any().downcast_ref::<arrow::array::Int64Array>().unwrap();
1433                        let v = a.value(row_i);
1434                        serde_json::json!(v - (v % (*w as i64)))
1435                    },
1436                    _ => serde_json::Value::Null,
1437                }
1438            },
1439            Self::Year => {
1440                let array = arrays[0];
1441                // Years from 1970
1442                if let arrow::datatypes::DataType::Date32 = array.data_type() {
1443                    let a = array.as_any().downcast_ref::<arrow::array::Date32Array>().unwrap();
1444                    let days = a.value(row_i);
1445                    // 1970-01-01 is epoch.
1446                    let opt_date = chrono::NaiveDate::from_num_days_from_ce_opt(days + 719163);
1447                    if let Some(d) = opt_date {
1448                        serde_json::json!(d.year() - 1970)
1449                    } else { serde_json::Value::Null }
1450                } else if let arrow::datatypes::DataType::Timestamp(arrow::datatypes::TimeUnit::Microsecond, _) = array.data_type() {
1451                     let a = array.as_any().downcast_ref::<arrow::array::TimestampMicrosecondArray>().unwrap();
1452                     let micros = a.value(row_i);
1453                     let seconds = micros / 1_000_000;
1454                     let opt_dt = chrono::DateTime::from_timestamp(seconds, 0);
1455                     if let Some(dt) = opt_dt {
1456                         serde_json::json!(dt.year() - 1970)
1457                     } else { serde_json::Value::Null }
1458                } else { serde_json::Value::Null }
1459            },
1460            Self::Month => {
1461                let array = arrays[0];
1462                // Months from 1970-01-01
1463                if let arrow::datatypes::DataType::Date32 = array.data_type() {
1464                    let a = array.as_any().downcast_ref::<arrow::array::Date32Array>().unwrap();
1465                    let days = a.value(row_i);
1466                    let opt_date = chrono::NaiveDate::from_num_days_from_ce_opt(days + 719163);
1467                    if let Some(d) = opt_date {
1468                        serde_json::json!((d.year() - 1970) * 12 + (d.month() as i32) - 1)
1469                    } else { serde_json::Value::Null }
1470                } else if let arrow::datatypes::DataType::Timestamp(arrow::datatypes::TimeUnit::Microsecond, _) = array.data_type() {
1471                     let a = array.as_any().downcast_ref::<arrow::array::TimestampMicrosecondArray>().unwrap();
1472                     let micros = a.value(row_i);
1473                     let seconds = micros / 1_000_000;
1474                     let opt_dt = chrono::DateTime::from_timestamp(seconds, 0);
1475                     if let Some(dt) = opt_dt {
1476                         serde_json::json!((dt.year() - 1970) * 12 + (dt.month() as i32) - 1)
1477                     } else { serde_json::Value::Null }
1478                } else { serde_json::Value::Null }
1479            },
1480            Self::Day => {
1481                let array = arrays[0];
1482                // Days from 1970-01-01
1483                if let arrow::datatypes::DataType::Date32 = array.data_type() {
1484                    let a = array.as_any().downcast_ref::<arrow::array::Date32Array>().unwrap();
1485                     serde_json::json!(a.value(row_i))
1486                } else if let arrow::datatypes::DataType::Timestamp(arrow::datatypes::TimeUnit::Microsecond, _) = array.data_type() {
1487                     let a = array.as_any().downcast_ref::<arrow::array::TimestampMicrosecondArray>().unwrap();
1488                     let micros = a.value(row_i);
1489                     // Spec: input timestamp (micros) -> days from epoch
1490                     serde_json::json!(micros / (1_000_000 * 60 * 60 * 24))
1491                } else { serde_json::Value::Null }
1492            },
1493            Self::Hour => {
1494                let array = arrays[0];
1495                // Hours from 1970-01-01 00:00:00
1496                if let arrow::datatypes::DataType::Timestamp(arrow::datatypes::TimeUnit::Microsecond, _) = array.data_type() {
1497                     let a = array.as_any().downcast_ref::<arrow::array::TimestampMicrosecondArray>().unwrap();
1498                     let micros = a.value(row_i);
1499                     serde_json::json!(micros / (1_000_000 * 60 * 60))
1500                } else { serde_json::Value::Null }
1501            },
1502            Self::Void => serde_json::Value::Null,
1503        }
1504    }

1513pub fn json_to_avro_value(v: &serde_json::Value) -> apache_avro::types::Value {
1514    match v {
1515        serde_json::Value::Null => apache_avro::types::Value::Null,
1516        serde_json::Value::Bool(b) => apache_avro::types::Value::Boolean(*b),
1517        serde_json::Value::Number(n) => {
1518            if let Some(i) = n.as_i64() {
1519                if i >= (i32::MIN as i64) && i <= (i32::MAX as i64) {
1520                    apache_avro::types::Value::Int(i as i32)
1521                } else {
1522                    apache_avro::types::Value::Long(i)
1523                }
1524            } else if let Some(f) = n.as_f64() {
1525                apache_avro::types::Value::Double(f)
1526            } else {
1527                apache_avro::types::Value::Null
1528            }
1529        },
1530        serde_json::Value::String(s) => apache_avro::types::Value::String(s.clone()),
1531        _ => apache_avro::types::Value::Null
1532    }
