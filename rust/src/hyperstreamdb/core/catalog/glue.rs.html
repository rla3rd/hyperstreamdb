12/// AWS Glue Catalog client
13#[derive(Clone)]
14pub struct GlueCatalogClient {
15    client: GlueClient,
16    catalog_id: Option<String>,  // AWS account ID (optional)
17}
18
19impl GlueCatalogClient {
20    /// Create a new Glue Catalog client
21    /// Uses default AWS credentials from environment
22    pub async fn new(catalog_id: Option<String>) -> Result<Self> {
23        let config = aws_config::load_defaults(BehaviorVersion::latest()).await;
24        let client = GlueClient::new(&config);
25        Ok(Self { client, catalog_id })
26    }
27    
28    /// Create with explicit AWS config
29    pub fn from_config(config: &aws_config::SdkConfig, catalog_id: Option<String>) -> Self {
30        let client = GlueClient::new(config);
31        Self { client, catalog_id }
32    }
33}

35#[async_trait]
36impl Catalog for GlueCatalogClient {
37    async fn create_table(
38        &self,
39        namespace: &str,  // Glue database name
40        table_name: &str,
41        schema: SchemaRef,
42        location: Option<&str>,
43    ) -> Result<()> {
44        // 1. Initialize table manifest if location is provided
45        if let Some(loc) = location {
46             crate::Table::create_async(loc.to_string(), schema.clone()).await?;
47        }
48
49        // 2. Extract columns from Arrow schema for Glue
50        let columns: Vec<Column> = schema.fields().iter().map(|f| {
51            Column::builder()
52                .name(f.name())
53                .r#type(f.data_type().to_string())
54                .build()
55                .unwrap() // Glue Column build is usually infallible for simple cases
56        }).collect();
57        
58        // Create storage descriptor
59        let storage_descriptor = StorageDescriptor::builder()
60            .set_location(location.map(|s| s.to_string()))
61            .set_columns(Some(columns))
62            .build();
63        
64        // Create table input
65        let mut table_input = TableInput::builder()
66            .name(table_name)
67            .storage_descriptor(storage_descriptor)
68            .table_type("EXTERNAL_TABLE");
69        
70        // Add Iceberg-specific parameters
71        let mut parameters = HashMap::new();
72        parameters.insert("table_type".to_string(), "ICEBERG".to_string());
73        if let Some(loc) = location {
74            parameters.insert("metadata_location".to_string(), loc.to_string());
75        }
76        table_input = table_input.set_parameters(Some(parameters));
77        
78        // Create table
79        let mut req = self.client.create_table()
80            .database_name(namespace)
81            .table_input(table_input.build().map_err(|e| anyhow!("Failed to build table input: {}", e))?);
82        
83        if let Some(catalog_id) = &self.catalog_id {
84            req = req.catalog_id(catalog_id);
85        }
86        
87        req.send().await
88            .map_err(|e: aws_sdk_glue::error::SdkError<aws_sdk_glue::operation::create_table::CreateTableError>| anyhow::anyhow!("Failed to create Glue table: {}", e))?;
89        
90        Ok(())
91    }
92    
93    async fn load_table(&self, namespace: &str, table_name: &str) -> Result<TableMetadata> {
94        let mut req = self.client.get_table()
95            .database_name(namespace)
96            .name(table_name);
97        
98        if let Some(catalog_id) = &self.catalog_id {
99            req = req.catalog_id(catalog_id);
100        }
101        
102        let resp = req.send().await
103            .map_err(|e: aws_sdk_glue::error::SdkError<aws_sdk_glue::operation::get_table::GetTableError>| anyhow::anyhow!("Failed to get Glue table: {}", e))?;
104        
105        let table = resp.table()
106            .ok_or_else(|| anyhow!("Table not found in Glue response"))?;
107        
108        // Extract metadata_location from parameters
109        let params = table.parameters()
110            .ok_or_else(|| anyhow!("No parameters in Glue table"))?;
111        
112        let location = params.get("metadata_location")
113            .ok_or_else(|| anyhow!("No metadata_location in Glue table parameters"))?
114            .clone();
115        
116        Ok(TableMetadata::minimal(location))
117    }

119    async fn create_branch(&self, _branch_name: &str, _source_ref: Option<&str>) -> Result<()> {
120        // Glue doesn't support branches
121        Err(anyhow!("AWS Glue does not support branching. Use Nessie for Git-like branching."))
122    }

124    async fn table_exists(&self, namespace: &str, table_name: &str) -> Result<bool> {
125        match self.load_table(namespace, table_name).await {
126            Ok(_) => Ok(true),
127            Err(_) => Ok(false),
128        }
129    }

131    async fn commit_table(&self, namespace: &str, table_name: &str, updates: Vec<serde_json::Value>) -> Result<()> {
132        // For Glue, we need to update the table's metadata_location parameter
133        // First, get the current table to preserve existing settings
134        let mut get_req = self.client.get_table()
135            .database_name(namespace)
136            .name(table_name);
137        
138        if let Some(catalog_id) = &self.catalog_id {
139            get_req = get_req.catalog_id(catalog_id);
140        }
141        
142        let resp = get_req.send().await
143            .map_err(|e| anyhow::anyhow!("Failed to get Glue table for commit: {}", e))?;
144        
145        let existing_table = resp.table()
146            .ok_or_else(|| anyhow!("Table not found in Glue response"))?;
147        
148        // Extract the new metadata location from updates (look for set-snapshot-ref or similar)
149        let mut new_metadata_location: Option<String> = None;
150        for update in &updates {
151            if let Some(action) = update.get("action").and_then(|v| v.as_str()) {
152                if action == "set-current-snapshot" || action == "add-snapshot" {
153                    // The snapshot update implies we need to update metadata location
154                    // In a full implementation, we would compute the new path from snapshot
155                    if let Some(snapshot) = update.get("snapshot") {
156                        if let Some(manifest_list) = snapshot.get("manifest-list").and_then(|v| v.as_str()) {
157                            // Derive metadata location from manifest list path
158                            // e.g., s3://bucket/table/_manifest/snap-1.avro -> s3://bucket/table/metadata/vN.metadata.json
159                            if let Some(table_root) = manifest_list.rsplit_once("/_manifest/") {
160                                new_metadata_location = Some(format!("{}/metadata/v{}.metadata.json", 
161                                    table_root.0,
162                                    snapshot.get("snapshot-id").and_then(|v| v.as_i64()).unwrap_or(1)
163                                ));
164                            }
165                        }
166                    }
167                }
168            }
169        }
170        
171        // Update parameters with new metadata location
172        let mut parameters: HashMap<String, String> = existing_table.parameters().cloned()
173            .unwrap_or_default();
174        
175        if let Some(new_loc) = new_metadata_location {
176            parameters.insert("metadata_location".to_string(), new_loc);
177        }
178        
179        // Preserve existing storage descriptor
180        let storage_descriptor = existing_table.storage_descriptor()
181            .cloned()
182            .unwrap_or_else(|| StorageDescriptor::builder().build());
183        
184        // Build updated table input
185        let table_input = TableInput::builder()
186            .name(table_name)
187            .storage_descriptor(storage_descriptor)
188            .table_type(existing_table.table_type().unwrap_or("EXTERNAL_TABLE"))
189            .set_parameters(Some(parameters))
190            .build()
191            .map_err(|e| anyhow!("Failed to build table input: {}", e))?;
192        
193        // Update table
194        let mut update_req = self.client.update_table()
195            .database_name(namespace)
196            .table_input(table_input);
197        
198        if let Some(catalog_id) = &self.catalog_id {
199            update_req = update_req.catalog_id(catalog_id);
200        }
201        
202        update_req.send().await
203            .map_err(|e| anyhow::anyhow!("Failed to update Glue table: {}", e))?;
204        
205        Ok(())
206    }
