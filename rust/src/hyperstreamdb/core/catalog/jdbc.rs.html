9/// JDBC Implementation of Iceberg Catalog
10pub struct JdbcCatalogClient {
11    pool: AnyPool,
12    warehouse: String,
13    catalog_name: String,
14}
15
16impl JdbcCatalogClient {
17    pub async fn new(uri: String, warehouse: Option<String>, catalog_name: String) -> Result<Self> {
18        // sqlx-any uses the scheme to determine the driver
19        let pool = AnyPoolOptions::new()
20            .max_connections(5)
21            .connect(&uri)
22            .await
23            .context(format!("Failed to connect to JDBC catalog database at {}", uri))?;
24        
25        let client = Self {
26            pool,
27            warehouse: warehouse.unwrap_or_else(|| "/tmp/hyperstream_warehouse".to_string()),
28            catalog_name,
29        };
30        
31        client.setup().await?;
32        
33        Ok(client)
34    }
35
36    async fn setup(&self) -> Result<()> {
37        // Create tables if they don't exist (using standard SQL)
38        sqlx::query(
39            "CREATE TABLE IF NOT EXISTS iceberg_tables (
40                catalog_name VARCHAR(255) NOT NULL,
41                table_namespace VARCHAR(255) NOT NULL,
42                table_name VARCHAR(255) NOT NULL,
43                metadata_location VARCHAR(255),
44                previous_metadata_location VARCHAR(255),
45                PRIMARY KEY (catalog_name, table_namespace, table_name)
46            )"
47        ).execute(&self.pool).await?;
48
49        sqlx::query(
50            "CREATE TABLE IF NOT EXISTS namespaces (
51                catalog_name VARCHAR(255) NOT NULL,
52                namespace VARCHAR(255) NOT NULL,
53                PRIMARY KEY (catalog_name, namespace)
54            )"
55        ).execute(&self.pool).await?;
56
57        Ok(())
58    }
59}
60
61#[async_trait]
62impl Catalog for JdbcCatalogClient {
63    async fn create_table(
64        &self,
65        namespace: &str,
66        table_name: &str,
67        schema: SchemaRef,
68        location: Option<&str>,
69    ) -> Result<()> {
70        // 1. Determine location if not provided
71        let table_location = location.map(|s| s.to_string()).unwrap_or_else(|| {
72             format!("{}/{}/{}", self.warehouse, namespace, table_name)
73        });
74        
75        // 2. Initialize table manifest in storage
76        crate::Table::create_async(table_location.clone(), schema.clone()).await?;
77        
78        // Initial metadata location
79        let metadata_location = format!("{}/manifest.json", table_location); 
80
81        // 3. Register in DB
82        sqlx::query(
83            "INSERT INTO iceberg_tables (catalog_name, table_namespace, table_name, metadata_location) 
84             VALUES (?, ?, ?, ?)"
85        )
86        .bind(&self.catalog_name)
87        .bind(namespace)
88        .bind(table_name)
89        .bind(&metadata_location)
90        .execute(&self.pool)
91        .await?;
92
93        Ok(())
94    }
95
96    async fn load_table(&self, namespace: &str, table_name: &str) -> Result<TableMetadata> {
97        let row = sqlx::query(
98            "SELECT metadata_location FROM iceberg_tables 
99             WHERE catalog_name = ? AND table_namespace = ? AND table_name = ?"
100        )
101        .bind(&self.catalog_name)
102        .bind(namespace)
103        .bind(table_name)
104        .fetch_one(&self.pool)
105        .await
106        .map_err(|e| anyhow!("Table {}.{} not found in JDBC catalog: {}", namespace, table_name, e))?;
107
108        let metadata_location: String = row.get(0);
109        
110        // Extract base location from metadata location (manifest.json)
111        let location = if metadata_location.contains("/manifest.json") {
112             metadata_location.replace("/manifest.json", "")
113        } else {
114             metadata_location
115        };
116
117        Ok(TableMetadata::minimal(location))
118    }
119
120    async fn create_branch(&self, _branch_name: &str, _source_ref: Option<&str>) -> Result<()> {
121        Err(anyhow!("JDBC Catalog does not support branching."))
122    }
123
124    async fn table_exists(&self, namespace: &str, table_name: &str) -> Result<bool> {
125         let row = sqlx::query(
126            "SELECT COUNT(*) FROM iceberg_tables 
127             WHERE catalog_name = ? AND table_namespace = ? AND table_name = ?"
128        )
129        .bind(&self.catalog_name)
130        .bind(namespace)
131        .bind(table_name)
132        .fetch_one(&self.pool)
133        .await?;
134        
135        // Handle potential differences in count return type (i32/i64)
136        let count: i64 = match row.try_get::<i64, _>(0) {
137            Ok(c) => c,
138            Err(_) => row.get::<i32, _>(0) as i64,
139        };
140        Ok(count > 0)
141    }
142
143    async fn commit_table(&self, namespace: &str, table_name: &str, updates: Vec<serde_json::Value>) -> Result<()> {
144        // Extract new metadata location from updates
145        let mut new_metadata_location: Option<String> = None;
146        for update in &updates {
147            if let Some(action) = update.get("action").and_then(|v| v.as_str()) {
148                if action == "add-snapshot" {
149                    if let Some(snapshot) = update.get("snapshot") {
150                        if let Some(manifest_list) = snapshot.get("manifest-list").and_then(|v| v.as_str()) {
151                            if let Some(table_root) = manifest_list.rsplit_once("/_manifest/") {
152                                let snap_id = snapshot.get("snapshot-id").and_then(|v| v.as_i64()).unwrap_or(1);
153                                new_metadata_location = Some(format!("{}/metadata/v{}.metadata.json", table_root.0, snap_id));
154                            }
155                        }
156                    }
157                }
158            }
159        }
160        
161        let new_loc = new_metadata_location.ok_or_else(|| anyhow!("No new metadata location in updates"))?;
162        
163        // Atomic UPDATE with previous_metadata_location for history
164        sqlx::query(
165            "UPDATE iceberg_tables 
166             SET previous_metadata_location = metadata_location, 
167                 metadata_location = ? 
168             WHERE catalog_name = ? AND table_namespace = ? AND table_name = ?"
169        )
170        .bind(&new_loc)
171        .bind(&self.catalog_name)
172        .bind(namespace)
173        .bind(table_name)
174        .execute(&self.pool)
175        .await?;
176        
177        Ok(())
178    }
179}
