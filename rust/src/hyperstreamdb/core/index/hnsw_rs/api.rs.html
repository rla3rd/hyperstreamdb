16pub trait AnnT {
17    /// type of data vectors 
18    type Val;
19    ///
20    fn insert_data(&mut self, data: &Vec<Self::Val>, id: usize);
21    ///
22    fn search_neighbours(&self,  data :&Vec<Self::Val> , knbn : usize, ef_s: usize) -> Vec<Neighbour>;
23    ///
24    fn parallel_insert_data(&mut self, data: &Vec<(&Vec<Self::Val>, usize)> );
25    ///
26    fn parallel_search_neighbours(&self, data: &Vec<Vec<Self::Val> >, knbn : usize, ef_s: usize) -> Vec<Vec<Neighbour>> ;
27    ///
28    /// dumps a data and graph in 2 files. 
29    /// Datas are dumped in file filename.hnsw.data and graph in filename.hnsw.graph
30    fn file_dump(&self, filename: &String) -> Result<i32, String>;
31 }

34impl <T,D> AnnT for Hnsw<T,D>  where T:Serialize+DeserializeOwned+Clone+Send+Sync, D: Distance<T>+Send+Sync {
35    type Val= T;
36    ///
37    fn insert_data(&mut self, data: &Vec<Self::Val>, id: usize) {
38       self.insert((data, id));
39    }
40    ///
41    fn search_neighbours(&self, data : &Vec<T>, knbn : usize, ef_s: usize) -> Vec<Neighbour> {
42        self.search(data, knbn, ef_s, None)
43    }
44    fn parallel_insert_data(&mut self, data: &Vec<(&Vec<Self::Val>, usize)> ) {
45        self.parallel_insert(data);
46    }
47
48   fn parallel_search_neighbours(&self, data: &Vec<Vec<Self::Val>>, knbn : usize, ef_s: usize) -> Vec<Vec<Neighbour>> {
49       self.parallel_search(data, knbn, ef_s)
50   }
51   /// The main entry point to do a dump.  
52   /// It will generate two files one for the graph part of the data. The other for the real data points of the structure.
53   fn file_dump(&self, filename: &String) -> Result<i32, String> {
54        log::debug!("\n in file_dump : {:?}", filename);
55        //
56        let mut graphname = filename.clone();
57        graphname.push_str(".hnsw.graph");
58        let graphpath = PathBuf::from(graphname);
59        let fileres = OpenOptions::new().write(true).create(true).truncate(true).open(&graphpath);
60        if fileres.is_err() {
61            log::error!("api::file_dump could not open file {:?}", graphpath.as_os_str());
62            println!("api::file_dump: could not open file {:?}", graphpath.as_os_str());
63            return Err("api::file_dump could not open file".to_string());            
64        }
65        let graphfile = fileres.unwrap();
66        //
67        let mut dataname = filename.clone();
68        dataname.push_str(".hnsw.data");
69        let datapath = PathBuf::from(dataname);
70        let fileres = OpenOptions::new().write(true).create(true).truncate(true).open(&datapath);
71        if fileres.is_err() {
72            println!("api::file_dumpcould not open file {:?}", datapath.as_os_str());
73            return Err("api::file_dump could not open file".to_string());            
74        }
75        let datafile = fileres.unwrap();
76        let mut graphbufw = BufWriter::with_capacity(50_000_000 , graphfile);
77        let mut databufw = BufWriter::with_capacity(50_000_000 , datafile);
78        let res = self.dump(DumpMode::Full, &mut graphbufw, &mut databufw);
79        graphbufw.flush().unwrap();
80        databufw.flush().unwrap();
81        log::debug!("\n end of dump");
82        return res;
83   }   // end of dump
84} // end of impl block AnnT for Hnsw<T,D>
