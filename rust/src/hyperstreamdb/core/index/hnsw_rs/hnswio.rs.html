1//! This file provides io dump/ reload of computed graph.
2//!
3//! A dump is constituted of 2 files. 
4//! One file stores just the graph (or topology) with id of points.  
5//! The other file stores the ids and vector in point.
6//! The graph file is suffixed by "hnsw.graph" the other is suffixed by "hnsw.data"
7//! 
8//! An example of dump and reload of structure Hnsw is given in the tests (see test_dump_reload)
9/// 
10/// 
11// datafile
12// MAGICDATAP : u32
13// dimension : u32
14// The for each point the triplet: (MAGICDATAP, origin_id , array of values.) ( u32, u64, ....)
15//
16// A point is dumped in graph file as given by its external id (type DataId i.e : a usize, possibly a hash value) 
17// and layer (u8) and rank_in_layer:i32.
18// In the data file the point dump consist in the triplet: (MAGICDATAP, origin_id , array of values.)
19//

47// magic head of data file and before each data vector

65/// structure describing main parameters for hnsnw data and written at the beginning of a dump file.
66/// 
67/// Name of distance and type of data must be encoded in the dump file for a coherent reload.
68#[repr(C)]
69pub struct Description {
70    ///  value is 1 for Full 0 for Light
71    pub dumpmode : u8,
72    /// max number of connections in layers != 0
73    pub max_nb_connection : u8,
74    /// number of observed layers
75    pub nb_layer : u8,
76    /// search parameter
77    pub ef: usize,
78    /// total number of points
79    pub nb_point: usize,
80    /// data dimension
81    pub dimension : usize,
82    /// name of distance
83    pub distname : String,
84    /// T typename
85    pub t_name : String,
86}
87
88impl Description {
89    /// The dump of Description consists in :
90    /// . The value MAGICDESCR_1 as a u32 (4 u8)
91    /// . The type of dump as u8
92    /// . max_nb_connection as u8
93    /// . ef (search parameter used in construction) as usize
94    /// . nb_point (the number points dumped) as a usize
95    /// . the name of distance used. (nb byes as a usize then list of bytes)
96    /// 
97    fn dump<W:Write>(&self, argmode : DumpMode, out : &mut io::BufWriter<W>) -> Result<i32, String> {
98        log::info!("in dump of description");
99        out.write(&MAGICDESCR_2.to_ne_bytes()).unwrap();
100        let mode : u8 = match argmode {
101            DumpMode::Full => 1,
102            _              => 0,
103        };
104        // CAVEAT should check mode == self.mode
105        out.write(&mode.to_ne_bytes()).unwrap();
106        // dump of max_nb_connection as u8!!
107        out.write(&self.max_nb_connection.to_ne_bytes()).unwrap();
108        out.write(&self.nb_layer.to_ne_bytes()).unwrap();
109        if self.nb_layer != NB_LAYER_MAX {
110            println!("dump of Description, nb_layer != NB_MAX_LAYER");
111            return Err(String::from("dump of Description, nb_layer != NB_MAX_LAYER"));
112        }
113        //
114        log::info!("dumping ef {:?}", self.ef);
115        out.write(&self.ef.to_ne_bytes()).unwrap();
116        //
117        log::info!("dumping nb point {:?}", self.nb_point);
118        out.write(&self.nb_point.to_ne_bytes()).unwrap();
119        //
120        log::info!("dumping dimension of data {:?}", self.dimension);
121        out.write(&self.dimension.to_ne_bytes()).unwrap();
122
123        // dump of distance name
124        let namelen : usize = self.distname.len();
125        log::info!("distance name {:?} ", self.distname);
126        out.write(&namelen.to_ne_bytes()).unwrap();
127        out.write(self.distname.as_bytes()).unwrap();
128        // dump of T value typename
129        let namelen : usize = self.t_name.len();
130        log::info!("T name {:?} ", self.t_name);
131        out.write(&namelen.to_ne_bytes()).unwrap();
132        out.write(self.t_name.as_bytes()).unwrap();
133        //
134        return Ok(1);
135    } // end fo dump
136
137} // end of HnswIO impl for Descr

140/// This method is a preliminary to do a full reload from a dump.
141/// The method load_hnsw needs to know the typename , distance used, and construction parameters.
142/// So the reload is made in two steps.
143pub fn load_description(io_in: &mut dyn Read)  -> io::Result<Description> {
144    //
145    let mut descr = Description{ dumpmode: 0, max_nb_connection: 0, nb_layer: 0, 
146                                ef: 0, nb_point: 0, dimension : 0, 
147                                distname: String::from(""), t_name : String::from("")};
148    let mut it_slice = [0u8; std::mem::size_of::<u32>()];
149    io_in.read_exact(&mut it_slice)?;
150    let magic = u32::from_ne_bytes(it_slice);
151    log::debug!(" magic {:X} ", magic);
152    if magic !=  MAGICDESCR_1 && magic !=  MAGICDESCR_2 {
153        log::info!("bad magic");
154        return Err(io::Error::new(io::ErrorKind::Other, "bad magic at descr beginning"));
155    }
156    else if magic ==  MAGICDESCR_1 {
157        log::info!("old version of dump..., exiting");
158        println!("old version of dump");
159        return Err(io::Error::new(io::ErrorKind::Other, "old format of dump"));
160    }
161    let mut it_slice = [0u8; std::mem::size_of::<u8>()];
162    io_in.read_exact(&mut it_slice)?;
163    descr.dumpmode = u8::from_ne_bytes(it_slice);
164    log::info!(" dumpmode {:?} ", descr.dumpmode);
165    //
166    let mut it_slice = [0u8; std::mem::size_of::<u8>()];
167    io_in.read_exact(&mut it_slice)?;
168    descr.max_nb_connection = u8::from_ne_bytes(it_slice);
169    log::info!(" max_nb_connection {:?} ", descr.max_nb_connection);
170    //
171    let mut it_slice = [0u8; std::mem::size_of::<u8>()];    
172    io_in.read_exact(&mut it_slice)?;
173    descr.nb_layer = u8::from_ne_bytes(it_slice);
174    log::info!("nb_layer  {:?} ", descr.nb_layer);
175    // ef 
176    let mut it_slice = [0u8; std::mem::size_of::<usize>()];    
177    io_in.read_exact(&mut it_slice)?;
178    descr.ef = usize::from_ne_bytes(it_slice);
179    log::info!("ef  {:?} ", descr.ef);
180    // nb_point
181    let mut it_slice = [0u8; std::mem::size_of::<usize>()];
182    io_in.read_exact(&mut it_slice)?;
183    descr.nb_point = usize::from_ne_bytes(it_slice);
184    // read dimension
185    let mut it_slice = [0u8; std::mem::size_of::<usize>()];
186    io_in.read_exact(&mut it_slice)?;
187    descr.dimension = usize::from_ne_bytes(it_slice);
188    log::info!("nb_point {:?} dimension {:?} ", descr.nb_point, descr.dimension);    
189    // distance name
190    let mut it_slice = [0u8; std::mem::size_of::<usize>()];
191    io_in.read_exact(&mut it_slice)?;
192    let len : usize = usize::from_ne_bytes(it_slice);
193    log::debug!("length of distance name {:?} ", len);
194    if len > 256 {
195        log::info!(" length of distance name > 256");
196        println!(" length of distance name should not exceed 256");
197        return Err(io::Error::new(io::ErrorKind::Other, "bad lenght for distance name"));
198    }
199    let mut distv = Vec::<u8>::new();
200    distv.resize(len , 0);
201    io_in.read_exact(distv.as_mut_slice())?;
202    let distname = String::from_utf8(distv).unwrap();
203    log::debug!("distance name {:?} ", distname);
204    descr.distname = distname;
205    // reload of type name
206    let mut it_slice = [0u8; std::mem::size_of::<usize>()];
207    io_in.read_exact(&mut it_slice)?;
208    let len : usize = usize::from_ne_bytes(it_slice);
209    log::debug!("length of T  name {:?} ", len);
210    if len > 256 {
211        println!(" length of T name should not exceed 256");
212        return Err(io::Error::new(io::ErrorKind::Other, "bad lenght for T name"));
213    }
214    let mut tnamev = Vec::<u8>::new();
215    tnamev.resize(len , 0);
216    io_in.read_exact(tnamev.as_mut_slice())?;
217    let t_name = String::from_utf8(tnamev).unwrap();
218    log::debug!("T type name {:?} ", t_name);
219    descr.t_name = t_name;   
220    log::debug!(" end of description load \n");
221    //
222    Ok(descr)
223}
224//
225// dump and load of Point<T>
226// ==========================
227//
228
229    ///  Graph part of point dump
230    /// dump of a point consist in  
231    ///  1. The value MAGICPOINT
232    ///  2. its identity ( a usize  rank in original data , hash value or else , and PointId)
233    ///  3. for each layer dump of the number of neighbours followed by :
234    ///      for each neighbour dump of its identity (: usize) and then distance (): u32) to point dumped.
235    ///
236    /// identity of a point is in full mode the triplet origin_id (: usize), layer (: u8) rank_in_layer (: u32)
237    ///                           light mode only origin_id (: usize)
238    ///  For data dump
239    ///  1. The value MAGICDATAP (u32)
240    ///  2. origin_id as a u64
241    ///  3. The vector of data (the length is known from Description)
242    
243fn dump_point<'a, T:Serialize+Clone+Sized+Send+Sync, W:Write>(point : &Point<T> , mode : DumpMode, 
244                    graphout : &mut io::BufWriter<W>, dataout : &mut io::BufWriter<W>) -> Result<i32, String> {
245    //
246    graphout.write(&MAGICPOINT.to_ne_bytes()).unwrap();
247    // dump ext_id: usize , layer : u8 , rank in layer : i32
248    graphout.write(&point.get_origin_id().to_ne_bytes()).unwrap();
249    let p_id = point.get_point_id();
250    if mode == DumpMode::Full {
251        graphout.write(&p_id.0.to_ne_bytes()).unwrap();
252        graphout.write(&p_id.1.to_ne_bytes()).unwrap();
253    }
254    log::trace!(" point dump {:?} {:?}  ", p_id, point.get_origin_id());
255    // then dump neighborhood info : nb neighbours : u32 , then list of origin_id, layer, rank_in_layer
256    let neighborhood = point.get_neighborhood_id();
257    // in any case nb_layers are dumped with possibly 0 neighbours at a layer, but this does not occur by construction
258    for l in 0..neighborhood.len() {
259        let neighbours_at_l = &neighborhood[l];
260        // Caution : we dump number of neighbours as a usize, even if it cannot be so large!
261        let nbg_l : usize = neighbours_at_l.len();
262        log::trace!("\t dumping nbng : {} at l {}", nbg_l, l);
263        graphout.write(&nbg_l.to_ne_bytes()).unwrap();
264        for n in neighbours_at_l { // dump d_id : uszie , distance : f32, layer : u8, rank in layer : i32
265            graphout.write(&n.d_id.to_ne_bytes()).unwrap();
266            if mode == DumpMode::Full {
267                graphout.write(&n.p_id.0.to_ne_bytes()).unwrap();
268                graphout.write(&n.p_id.1.to_ne_bytes()).unwrap();
269            }
270            graphout.write(&n.distance.to_ne_bytes()).unwrap();
271//                log::debug!("        voisins  {:?}  {:?}  {:?}", n.p_id,  n.d_id , n.distance);
272        }
273    }
274    // now we dump data vector!
275    dataout.write(&MAGICDATAP.to_ne_bytes()).unwrap();
276    let origin_u64 = point.get_origin_id() as u64;
277    dataout.write(&origin_u64.to_ne_bytes()).unwrap();
278    //
279    let serialized : Vec<u8> = bincode::serialize(point.get_v()).unwrap();
280//    log::debug!("serializing len {:?}", serialized.len());
281    let len_64 = serialized.len() as u64;
282    dataout.write(&len_64.to_ne_bytes()).unwrap();
283    dataout.write_all(&serialized).unwrap();
284    //
285    return Ok(1);
286} // end of dump for Point<T>

290//
291//  Reload a point from a dump.
292// 
293//  The graph part is loaded from graph_in file
294// the data vector itself is loaded from data_in
295// 
296fn load_point<T:'static+DeserializeOwned+Clone+Sized+Send+Sync>(graph_in: &mut dyn Read, descr: &Description, 
297                                                data_in: &mut dyn Read) -> io::Result<(Arc<Point<T>>, Vec<Vec<Neighbour> >) > {
298    //
299    // read and check magic
300    let mut it_slice  = [0u8; std::mem::size_of::<u32>()];
301    graph_in.read_exact(&mut it_slice).unwrap();
302    let magic = u32::from_ne_bytes(it_slice);
303    if magic != MAGICPOINT {
304        log::debug!("got instead of MAGICPOINT {:x}", magic);
305        return Err(io::Error::new(io::ErrorKind::Other, "bad magic at point beginning"));
306    }
307    let mut it_slice = [0u8; std::mem::size_of::<DataId>()];
308    graph_in.read_exact(&mut it_slice).unwrap();
309    let origin_id = DataId::from_ne_bytes(it_slice);
310    // read point_id
311    let mut it_slice = [0u8; std::mem::size_of::<u8>()];
312    graph_in.read_exact(&mut it_slice)?;
313    let layer = u8::from_ne_bytes(it_slice); 
314    //
315    let mut it_slice = [0u8; std::mem::size_of::<i32>()];
316    graph_in.read_exact(&mut it_slice)?;
317    let rank_in_l = i32::from_ne_bytes(it_slice);
318    let p_id = PointId{0: layer, 1:rank_in_l};
319//    log::debug!(" point load {:?} {:?}  ", p_id, origin_id);
320    // Now  for each layer , read neighbours
321    let nb_layer = descr.nb_layer;
322    let mut neighborhood = Vec::<Vec<Neighbour> >::with_capacity(NB_LAYER_MAX as usize);
323    for _l in 0..nb_layer {
324        let mut neighbour : Neighbour = Default::default();
325        // read nb_neighbour as usize!!! CAUTION, then nb_neighbours times identity(depends on Full or Light) distance : f32 
326        let mut it_slice = [0u8; std::mem::size_of::<usize>()];
327        graph_in.read_exact(&mut it_slice)?;
328        let nb_neighbours = usize::from_ne_bytes(it_slice);
329        let mut neighborhood_l : Vec<Neighbour> = Vec::with_capacity(nb_neighbours as usize);
330        for _j in 0..nb_neighbours {
331            let mut it_slice = [0u8; std::mem::size_of::<DataId>()];
332            graph_in.read_exact(&mut it_slice)?; 
333            neighbour.d_id = DataId::from_ne_bytes(it_slice);          
334            if descr.dumpmode == 1 {
335                let mut it_slice = [0u8; std::mem::size_of::<u8>()];
336                graph_in.read_exact(&mut it_slice)?;
337                neighbour.p_id.0 = u8::from_ne_bytes(it_slice);
338                //
339                let mut it_slice = [0u8; std::mem::size_of::<i32>() ];
340                graph_in.read_exact(&mut it_slice)?;
341                neighbour.p_id.1 = i32::from_ne_bytes(it_slice);
342            }
343            let mut it_slice = [0u8; std::mem::size_of::<f32>()];
344            graph_in.read_exact(&mut it_slice)?;
345            neighbour.distance = f32::from_ne_bytes(it_slice);
346        //  log::debug!("        voisins  load {:?} {:?} {:?} ", neighbour.p_id, neighbour.d_id , neighbour.distance);
347            // now we have a new neighbour, we must really fill neighbourhood info, so it means going from Neighbour to PointWithOrder
348            neighborhood_l.push(neighbour);
349        }
350        neighborhood.push(neighborhood_l);
351    }
352    for _l in nb_layer..NB_LAYER_MAX {
353        neighborhood.push(Vec::<Neighbour>::new());
354    }
355    //
356    // construct a point from data_in
357    //
358    let mut it_slice = [0u8; std::mem::size_of::<u32>()];
359    data_in.read_exact(&mut it_slice)?;
360    let magic = u32::from_ne_bytes(it_slice);
361    assert_eq!(magic, MAGICDATAP, "magic not equal to MAGICDATAP in load_point, point_id : {:?} ", origin_id);
362    // read origin id
363    let mut it_slice = [0u8; std::mem::size_of::<u64>()];
364    data_in.read_exact(&mut it_slice)?;
365    let origin_id_data = u64::from_ne_bytes(it_slice) as usize;
366    assert_eq!(origin_id, origin_id_data, "origin_id incoherent between graph and data");
367    // now read data. we use size_t that is in description, to take care of the casewhere we reload
368    let mut it_slice = [0u8; std::mem::size_of::<u64>()];
369    data_in.read_exact(&mut it_slice)?;
370    let serialized_len = u64::from_ne_bytes(it_slice);
371//    log::debug!("serialized len to reload {:?}", serialized_len);
372    let mut v_serialized = Vec::<u8>::new();
373    // TODO avoid initialization
374    v_serialized.resize(serialized_len as usize, 0);
375    data_in.read_exact(&mut v_serialized)?;
376    let v : Vec<T>;
377    if std::any::TypeId::of::<T>() != std::any::TypeId::of::<NoData>() {
378        v = bincode::deserialize(&v_serialized).unwrap();
379    }
380    else {
381        v = Vec::<T>::new();
382    }
383    let point = Point::<T>::new(&v, origin_id as usize, p_id);
384    log::trace!("load_point  origin {:?} allocated size {:?}, dim {:?}", origin_id, point.get_v().len(), descr.dimension);
385    //
386    return Ok((Arc::new(point), neighborhood));
387}  // end of load_point

391//
392// dump and load of PointIndexation<T>
393// ===================================
394//
395//
396// nb_layer : 8
397// a magick at each Layer : u32
398// . number of points in layer (usize), 
399// . list of point of layer
400// dump entry point
401// 
402impl <T:Serialize+DeserializeOwned+Clone+Send+Sync> HnswIO for PointIndexation<T> {
403    fn dump<W:Write>(&self, mode : DumpMode, graphout : &mut io::BufWriter<W>, dataout : &mut io::BufWriter<W>) -> Result<i32, String> {
404        // dump max_layer
405        let layers = self.points_by_layer.read();
406        let nb_layer = layers.len() as u8;
407        graphout.write(&nb_layer.to_ne_bytes()).unwrap();
408        // dump layers from lower (most populatated to higher level)
409        for i in 0..layers.len() {
410            let nb_point = layers[i].len();
411            log::debug!("dumping layer {:?}, nb_point {:?}", i, nb_point);
412            graphout.write(&MAGICLAYER.to_ne_bytes()).unwrap();
413            graphout.write(&nb_point.to_ne_bytes()).unwrap();
414            for j in 0..layers[i].len() {
415                assert_eq!(layers[i][j].get_point_id() , PointId{0: i as u8,1:j as i32 });
416                dump_point(&layers[i][j], mode, graphout, dataout)?;
417            }
418        }
419        // dump id of entry point
420        let ep_read = self.entry_point.read();
421        assert!(ep_read.is_some());
422        let ep = ep_read.as_ref().unwrap();
423        graphout.write(&ep.get_origin_id().to_ne_bytes()).unwrap();
424        let p_id = ep.get_point_id();
425        if mode == DumpMode::Full {
426            graphout.write(&p_id.0.to_ne_bytes()).unwrap();
427            graphout.write(&p_id.1.to_ne_bytes()).unwrap();
428        }
429        log::info!("dumped entry_point origin_d {:?}, p_id {:?} ", ep.get_origin_id(), p_id);
430        //
431        Ok(1)
432    } // end of dump for PointIndexation<T>
433} // end of impl HnswIO

436fn load_point_indexation<T:'static+Serialize+DeserializeOwned+Clone+Sized+Send+Sync>(graph_in: &mut dyn Read, 
437                descr : &Description, 
438                data_in:  &mut dyn Read) -> io::Result<PointIndexation<T> > {
439    //
440    log::debug!(" in load_point_indexation");
441    //
442    // now we check that except for the case NoData, the typename are the sames.
443    if std::any::TypeId::of::<T>() != std::any::TypeId::of::<NoData>() && std::any::type_name::<T>() != descr.t_name {
444        log::error!("typename loaded in  description {:?} do not correspond to instanciation type {:?}", 
445                            descr.t_name,std::any::type_name::<T>() );
446        panic!("incohrent size of T in description");
447    }
448    //
449    let mut points_by_layer : Vec<Vec<Arc<Point<T>> > >= Vec::with_capacity(NB_LAYER_MAX as usize);
450    let mut neighbourhood_map : HashMap<PointId, Vec<Vec<Neighbour>> > =  HashMap::new();
451    // load max layer
452    let mut it_slice = [0u8; ::std::mem::size_of::<u8>() ];
453    graph_in.read_exact(&mut it_slice)?;
454    let nb_layer = u8::from_ne_bytes(it_slice);
455    log::debug!("nb layer {:?}", nb_layer);
456    if nb_layer > NB_LAYER_MAX {
457        return Err(io::Error::new(io::ErrorKind::Other, "inconsistent number of layErrers"));
458    }
459    //
460    let mut nb_points_loaded : usize = 0;
461    //
462    for l in 0..nb_layer as usize {
463        // read and check magic
464        log::debug!("loading layer {:?}", l);
465        let mut it_slice = [0u8; ::std::mem::size_of::<u32>() ];
466        graph_in.read_exact(&mut it_slice)?;
467        let magic = u32::from_ne_bytes(it_slice);
468        if magic != MAGICLAYER {
469            return Err(io::Error::new(io::ErrorKind::Other, "bad magic at layer beginning"));
470        }
471        let mut it_slice = [0u8; ::std::mem::size_of::<usize>() ];
472        graph_in.read_exact(&mut it_slice)?;
473        let nbpoints = usize::from_ne_bytes(it_slice);
474        log::debug!(" layer {:?} , nb points {:?}", l ,  nbpoints);
475        let mut vlayer : Vec<Arc<Point<T>>> = Vec::with_capacity(nbpoints);
476        for r in 0..nbpoints {
477            // load graph and data part of point. Points are dumped in the same order.
478            let load_point_res = load_point(graph_in, descr, data_in);
479            match load_point_res {
480                Err(other) => {
481                            log::error!("in load_point_indexation, loading of point {} failed", r);
482                            return Err(other);
483                }
484                 _  =>  {},
485            }
486            let load_point_res = load_point_res.unwrap();
487            let point = load_point_res.0;
488            let p_id = point.get_point_id();
489            // some checks
490            assert_eq!(l, p_id.0 as usize);
491            if r != p_id.1 as usize {
492                log::debug!("\n\n origin= {:?},  p_id = {:?}", point.get_origin_id(), p_id);
493                log::debug!("storing at l {:?}, r {:?}",  l, r);
494            }
495            assert_eq!(r , p_id.1 as usize);
496            // store neoghbour info of this point
497            neighbourhood_map.insert(p_id, load_point_res.1);
498            vlayer.push(point);
499        }
500        points_by_layer.push(vlayer);
501        nb_points_loaded += nbpoints;
502    }
503    // at this step all points are loaded , but without their neighbours fileds are not yet initialized
504    let mut nbp: usize = 0;
505    for (p_id , neighbours) in &neighbourhood_map {
506        let point = &points_by_layer[p_id.0 as usize][p_id.1 as usize];
507        for l in 0..neighbours.len() {
508            for n in &neighbours[l] {
509                let n_point = &points_by_layer[n.p_id.0 as usize][n.p_id.1 as usize];
510                // now n_point is the Arc<Point> corresponding to neighbour n of point, 
511                // construct a corresponding PointWithOrder
512                let n_pwo = PointWithOrder::<T>::new(n_point, n.distance);
513                point.neighbours.write()[l].push(Arc::new(n_pwo));
514            } // end of for n
515            //  must sort
516            point.neighbours.write()[l].sort_unstable();
517        } // end of for l
518        nbp += 1;
519        if nbp % 500_000 == 0{
520            log::debug!("reloading nb_points neighbourhood completed : {}", nbp);
521        }
522    } // end loop in neighbourhood_map
523    // 
524    // get id of entry_point
525    // load entry point
526    log::info!("\n end of layer loading, allocating PointIndexation, nb points loaded {:?}", nb_points_loaded);
527    //
528    let mut it_slice = [0u8; std::mem::size_of::<DataId>()];
529    graph_in.read_exact(&mut it_slice)?;
530    let origin_id = DataId::from_ne_bytes(it_slice);
531    //
532    let mut it_slice = [0u8; ::std::mem::size_of::<u8>()];
533    graph_in.read_exact(&mut it_slice)?;
534    let layer = u8::from_ne_bytes(it_slice);
535    //
536    let mut it_slice = [0u8; std::mem::size_of::<i32>() ];
537    graph_in.read_exact(&mut it_slice)?;
538    let rank_in_l = i32::from_ne_bytes(it_slice);
539    //
540    log::info!("found entry point, origin_id {:?} , layer {:?}, rank in layer {:?} ", origin_id, layer, rank_in_l);
541    let entry_point = Arc::clone(&points_by_layer[layer as usize][rank_in_l as usize]);
542    log::info!(" loaded entry point, origin_id {:} p_id {:?}", entry_point.get_origin_id(),entry_point.get_point_id());
543
544    //
545    let point_indexation = PointIndexation {
546        max_nb_connection : descr.max_nb_connection as usize,
547        max_layer : NB_LAYER_MAX as usize,
548        points_by_layer : Arc::new(RwLock::new(points_by_layer)),
549        layer_g : LayerGenerator::new(descr.max_nb_connection as usize , NB_LAYER_MAX as usize),
550        nb_point : Arc::new(RwLock::new(nb_points_loaded)),   // CAVEAT , we should increase , the whole thing is to be able to increment graph ?
551        entry_point : Arc::new(RwLock::new(Some(entry_point))),
552    };
553    //  
554    log::debug!("\n exiting load_pointIndexation");
555    Ok(point_indexation)
556} // end of load_pointIndexation

566impl <T:Serialize+DeserializeOwned+Clone+Sized+Send+Sync, D: Distance<T>+Send+Sync> HnswIO for Hnsw<T, D> {
567    /// The dump method for hnsw.  
568    /// - graphout is a BufWriter dedicated to the dump of the graph part of Hnsw
569    /// - dataout is a bufWriter dedicated to the dump of the data stored in the Hnsw structure.
570    fn dump<W:Write>(&self, mode : DumpMode, graphout : &mut io::BufWriter<W>, dataout : &mut io::BufWriter<W>) -> Result<i32, String> {
571        // dump description , then PointIndexation
572        let dumpmode : u8 = match mode {
573                DumpMode::Full => 1,
574                            _  => 0,
575        };  
576        let datadim : usize = self.layer_indexed_points.get_data_dimension();
577
578        let description = Description {
579               //  value is 1 for Full 0 for Light
580            dumpmode : dumpmode,
581            max_nb_connection : self.get_max_nb_connection(),
582            nb_layer : self.get_max_level() as u8,
583            ef: self.get_ef_construction(),
584            nb_point: self.get_nb_point(),
585            dimension : datadim,
586            distname : self.get_distance_name(),
587            t_name: type_name::<T>().to_string(),
588        };
589        log::debug!("dump  obtained typename {:?}", type_name::<T>());
590        description.dump(mode, graphout)?;
591        // We must dump a header for dataout.
592        dataout.write(&MAGICDATAP.to_ne_bytes()).unwrap();
593        dataout.write(&datadim.to_ne_bytes()).unwrap();
594        //
595        self.layer_indexed_points.dump(mode, graphout, dataout)?;
596        Ok(1)
597    }
598}   // end impl block for Hnsw

602/// The reload is made in two steps.
603/// First a call to load_description must be used to get basic information
604/// about structure to reload (Typename, distance type, construction parameters).  
605/// Cf fn load_description(io_in: &mut dyn Read) -> io::Result\<Description\>
606///
607pub fn load_hnsw<T:'static+Serialize+DeserializeOwned+Clone+Sized+Send+Sync, D:Distance<T>+Default+Send+Sync>(graph_in: &mut dyn Read, 
608                                            description: &Description, 
609                                            data_in : &mut dyn Read) -> io::Result<Hnsw<T,D> > {
610    //  In datafile , we must read MAGICDATAP and dimension and check
611    let mut it_slice = [0u8; std::mem::size_of::<u32>()];
612    data_in.read_exact(&mut it_slice)?;
613    let magic = u32::from_ne_bytes(it_slice);
614    assert_eq!(magic, MAGICDATAP, "magic not equal to MAGICDATAP in load_point");
615    //
616    let mut it_slice = [0u8; std::mem::size_of::<usize>()];
617    data_in.read_exact(&mut it_slice)?;
618    let dimension = usize::from_ne_bytes(it_slice);
619    assert_eq!(dimension, description.dimension, "data dimension incoherent {:?} {:?} ", 
620            dimension, description.dimension);
621    //
622    let _mode = description.dumpmode;
623    let distname = description.distname.clone();
624    // We must ensure that the distance stored matches the one asked for in loading hnsw
625    // for that we check for short names equality stripping 
626    log::debug!("distance in description = {:?}", distname);
627    let d_type_name = type_name::<D>().to_string();
628    let d_short_name = d_type_name.rsplit("::").next().unwrap_or(&d_type_name);
629    let dist_short_name = distname.rsplit("::").next().unwrap_or(&distname);
630
631    if (std::any::TypeId::of::<T>() != std::any::TypeId::of::<NoData>())  &&  (d_short_name != dist_short_name) {
632        // for all types except NoData , distance asked in reload declaration and distance in dump must be equal!
633        let mut errmsg = String::from("error in distances : dumped distance is : ");
634        errmsg.push_str(&distname);
635        errmsg.push_str(" asked distance in loading is : ");
636        errmsg.push_str(&d_type_name);
637        log::error!(" distance in type argument : {:?}", d_type_name);
638        log::error!("error , dump is for distance = {:?}", distname);
639        return Err(io::Error::new(io::ErrorKind::Other, errmsg));
640    }
641    let t_type = description.t_name.clone();
642    log::debug!("T type name in dump = {:?}", t_type);
643    let layer_point_indexation = load_point_indexation(graph_in, &description, data_in)?;
644    let data_dim = layer_point_indexation.get_data_dimension();
645    //
646    let hnsw : Hnsw::<T,D> =  Hnsw{  max_nb_connection : description.max_nb_connection as usize,
647                        ef_construction : description.ef, 
648                        extend_candidates : true, 
649                        keep_pruned: false,
650                        max_layer: description.nb_layer as usize, 
651                        layer_indexed_points: layer_point_indexation,
652                        data_dimension : data_dim,
653                        dist_f: D::default(),
654                        searching : false,
655                    } ;
656    //
657    log::debug!("load_hnsw completed");
658    //
659    Ok(hnsw)
660}  // end of load_hnsw

720    //
