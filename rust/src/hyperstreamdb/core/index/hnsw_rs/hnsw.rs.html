187        let nb_layer = ref_neighbours.len();

198            }
199            neighborhood.push(neighbours);
200        }

288/// a struct to randomly generate a level for an item according to an exponential law
289/// of parameter given by scale. 
290/// The distribution is constrained to be in [0..maxlevel[
291pub struct LayerGenerator {
292    rng: Arc<Mutex<rand::rngs::StdRng>>,
293    unif: Uniform<f64>,
294    scale: f64,
295    maxlevel: usize,
296}
297
298impl LayerGenerator {
299    pub fn new(max_nb_connection: usize, maxlevel:usize) -> Self {
300        let scale = 1./(max_nb_connection as f64).ln();
301        LayerGenerator{rng: Arc::new(Mutex::new(StdRng::from_entropy())), 
302            unif: Uniform::<f64>::new(0.,1.), scale:scale, maxlevel:maxlevel}
303    }
304    //
305    // l=0 most densely packed layer
306    // if S is scale we sample so that P(l=n) = exp(-n/S) - exp(- (n+1)/S)
307    // with S = 1./ln(max_nb_connection) P(l >= maxlevel) = exp(-maxlevel * ln(max_nb_connection))
308    // for nb_conn = 10, even with maxlevel = 10,  we get P(l >= maxlevel) = 1.E-13
309    // In Malkov(2016) S = 1./log(max_nb_connection)
310    //
311    /// generate a layer with given maxlevel. upper layers (higher index) are of decreasing probabilities.
312    /// thread safe method.
313    fn generate(&self) -> usize {
314        let mut protected_rng = self.rng.lock();
315        let xsi = protected_rng.sample(self.unif);
316        let level = -xsi.ln() * self.scale;
317        let mut ulevel = level.floor() as usize;
318        // we redispatch possibly sampled level  >= maxlevel to required range
319        if ulevel >= self.maxlevel {
320            // This occurs with very low probability. Cf commentary above.
321            ulevel = protected_rng.sample(Uniform::<usize>::new(0,self.maxlevel));
322        }
323        ulevel
324    }
325
326    /// just to try some variations on exponential level sampling. Unused.
327    pub fn set_scale_modification(&mut self, scale_modification : f64) {
328        self.scale = 1./( (1./self.scale) + scale_modification.ln());
329    }
330
331} // end impl for LayerGenerator

356// A point indexation may contain circular references. To deallocate these after a point indexation goes out of scope,
357// implement the Drop trait.

361impl <T:Clone+Send+Sync> Drop for PointIndexation<T> {
362
363    fn drop(&mut self) {
364        let cpu_start = ProcessTime::now();
365        let sys_now = SystemTime::now(); 
366        log::info!("entering PointIndexation drop");
367        // clear_neighborhood. There are no point in neighborhoods that are not referenced directly in layers.
368        // so we cannot loose reference to a point by cleaning neighborhood 
369        fn clear_neighborhoods<T: Clone+Send+Sync>(init: &Point<T>) {
370                let mut neighbours = init.neighbours.write();
371                let nb_layer = neighbours.len();
372                for l in 0..nb_layer {
373                    neighbours[l].clear();
374                }
375                neighbours.clear();
376        }    
377        // clear entry point
378        match self.entry_point.write().as_ref() {
379            Some(i) => {
380                clear_neighborhoods(i.as_ref());
381            }
382            _ => {}
383        }
384        //
385        let nb_level = self.get_max_level_observed();
386        for l in 0..=nb_level {
387            log::trace!("clearing layer {}", l);
388            let layer = &mut self.points_by_layer.write()[l as usize];
389            layer.into_par_iter().for_each(|p| clear_neighborhoods(p));
390        }
391        //
392        log::trace!("dropping layers ...");
393        for l in 0..=nb_level {
394            let layer = &mut self.points_by_layer.write()[l as usize];
395            let _ = layer;
396        }
397        log::debug!("clearing self.points_by_layer...");
398        drop(self.points_by_layer.write());
399        log::debug!("exiting PointIndexation drop");
400        log::info!(" drop sys time(s) {:?} cpu time {:?}", sys_now.elapsed().unwrap().as_secs(), cpu_start.elapsed().as_secs());
401    } // end my drop
402
403} // end implementation Drop

409    pub fn new(max_nb_connection: usize, max_layer:usize, max_elements:usize) -> Self {
410        let mut points_by_layer = Vec::with_capacity(max_layer);
411        for i in 0..max_layer { // recall that range are right extremeity excluded 
412            // compute fraction of points going into layer i and do expected memory reservation
413            let frac = (-(i as f64)/(max_layer as f64)).exp() - (-((i+1) as f64)/(max_layer as f64));
414            let expected_size = ((frac * max_elements as f64).round()) as usize;
415            points_by_layer.push(Vec::with_capacity(expected_size));
416        }
417        let layer_g = LayerGenerator::new(max_nb_connection, max_layer);
418        PointIndexation{ 
419            max_nb_connection,
420            max_layer,
421            points_by_layer: Arc::new(RwLock::new(points_by_layer)),
422            layer_g,
423            nb_point:Arc::new(RwLock::new(0)),
424            entry_point: Arc::new(RwLock::new(None)),

448    /// real insertion of point in point indexation
449    // generate a new Point/ArcPoint (with neigbourhood info empty) and store it in global table
450    // The function is called by Hnsw insert method
451    fn generate_new_point(& self, data : &[T], origin_id : usize) -> (Arc<Point<T>> , usize) {
452        // get a write lock at the beginning of the function
453        let level = self.layer_g.generate();
454        let new_point;
455        { // open a write lock on points_by_layer
456            let mut points_by_layer_ref = self.points_by_layer.write();
457            let mut p_id = PointId{0:level as u8, 1:-1};
458            p_id.1 = points_by_layer_ref[p_id.0 as usize].len() as i32;
459            // make a Point and then an Arc<Point>
460            let point = Point::new(data, origin_id, p_id);
461            new_point = Arc::new(point);
462            log::trace!("definitive pushing of point {:?}", p_id);
463            points_by_layer_ref[p_id.0 as usize].push(Arc::clone(&new_point));
464        } // close write lock on points_by_layer
465        //
466        let nb_point;
467        {
468            let mut lock_nb_point = self.nb_point.write();
469            *lock_nb_point += 1;
470            nb_point = *lock_nb_point;
471            if nb_point % 50000 == 0 {

473            }
474        }
475        log::trace!(" setting number of points {:?} ", *self.nb_point);
476        // Now possibly this is a point on a new layer that will have no neighbours in its layer
477        return (Arc::clone(&new_point), nb_point);
478    } // end of insert

583/// The iteration begins at level 0 (most populated level) and goes upward in levels.
584/// The iterator takes a ReadGuard on the PointIndexation structure
585pub struct IterPoint<'a,T:Clone+Send+Sync> {

587    pi_guard :  RwLockReadGuard<'a, Vec<Layer<T>> >,

590}

592impl <'a, T:Clone+Send+Sync> IterPoint<'a,T>{
593    pub fn new(point_indexation : &'a PointIndexation<T>) -> Self {
594        let pi_guard : RwLockReadGuard<Vec<Layer<T>> > = point_indexation.points_by_layer.read();
595        IterPoint{ point_indexation, pi_guard, layer:-1, slot_in_layer : -1 }
596    }

600/// iterator for layer 0 to upper layer.
601impl <'a,T:Clone+Send+Sync> Iterator for IterPoint<'a,T> {
602    type Item = Arc<Point<T>>;
603    //
604    fn next(&mut self) -> Option<Self::Item>{
605        if self.layer == -1 {
606            self.layer = 0;
607            self.slot_in_layer = 0;         
608        }
609        if (self.slot_in_layer  as usize) < self.pi_guard[self.layer as usize].len() {
610            let slot = self.slot_in_layer as usize;
611            self.slot_in_layer += 1;
612            return Some(self.pi_guard[self.layer as usize][slot].clone());
613        }
614        else {
615            self.slot_in_layer = 0;
616            self.layer += 1;
617            // must reach a non empty layer if possible
618            let entry_point_ref = self.point_indexation.entry_point.read();
619            let points_by_layer = self.point_indexation.points_by_layer.read();
620            let entry_point_level = entry_point_ref.as_ref().unwrap().p_id.0;
621            while (self.layer as u8) <= entry_point_level &&

629                self.slot_in_layer += 1;
630                return Some(points_by_layer[self.layer as usize][slot].clone());
631           }
632            else {
633                return None;
634            }

829    // The efficacity depends on greedy part depends on how near entry point is from point.
830    // ef is the number of points to return
831    // The method returns a BinaryHeap with positive distances. The caller must transforms it according its need
832    //** NOTE: the entry point is pushed into returned point at the beginning of the function, but in fact entry_point is in a layer
833    //** with higher (one more) index than the argument layer. If the greedy search matches a sufficiently large number of points
834    //** nearer to point searched (arg point) than entry_point it will finally get popped up from the heap of returned points
835    //** but otherwise it will stay in the binary heap and so we can have a point in neighbours that is in fact in a layer
836    //** above the one we search in. 
837    //** The guarantee is that the binary heap will return points in layer 
838    //** with a larger index, although we can expect that most often (at least in densely populated layers) the returned
839    //** points will be found in searched layer 
840    ///
841    /// Greedy algorithm n° 2 in Malkov paper.
842    /// search in a layer (layer) for the ef points nearest a point to be inserted in hnsw.
843    fn search_layer(& self, point: &[T], entry_point: Arc<Point<T>> , ef:usize, layer: u8, filter: Option<&roaring::RoaringBitmap>) -> BinaryHeap<Arc<PointWithOrder<T>> > {
844        //
845        trace!("entering search_layer with entry_point_id {:?} layer : {:?} ef {:?} ", entry_point.p_id, layer, ef);
846        //
847        // here we allocate a binary_heap on values not on reference beccause we want to return
848        // log2(skiplist_size) must be greater than 1.
849        let skiplist_size = ef.max(2);
850        // we will store positive distances in this one
851        let mut return_points = BinaryHeap::<Arc<PointWithOrder<T>> >::with_capacity(skiplist_size);
852        //
853        if self.layer_indexed_points.points_by_layer.read()[layer as usize].len() == 0 {
854            // at the beginning we can have nothing in layer
855            trace!("search layer {:?}, empty layer", layer);
856            return return_points;
857        }
858        if entry_point.p_id.1 < 0 {
859            trace!("search layer negative point id : {:?}", entry_point.p_id);
860            return return_points;
861        }
862        // initialize visited points
863        let dist_to_entry_point = self.dist_f.eval(point, & entry_point.v);
864        log::trace!("       distance to entry point: {:?} ", dist_to_entry_point);
865        // keep a list of id visited
866        let mut visited_point_id = HashMap::<PointId, Arc<Point<T>>>::new();
867        visited_point_id.insert(entry_point.p_id, Arc::clone(&entry_point));
868        //
869        let mut candidate_points = BinaryHeap::<Arc<PointWithOrder<T>> >::with_capacity(skiplist_size);
870        candidate_points.push(Arc::new(PointWithOrder::new(&entry_point, -dist_to_entry_point)));
871        // Only return entry point if it matches the filter!
872        let mut entry_is_valid = true;
873        if let Some(f) = filter {
874            if !f.contains(entry_point.origin_id as u32) {
875                entry_is_valid = false;
876            }
877        }
878        if entry_is_valid {
879            return_points.push(Arc::new(PointWithOrder::new(&entry_point, dist_to_entry_point)));
880        }
881        // at the beginning candidate_points contains point passed as arg in layer entry_point_id.0
882        while candidate_points.len() > 0 {
883            // get nearest point in candidate_points
884            let c = candidate_points.pop().unwrap();
885            // f farthest point to 
886            let f = return_points.peek();
887            if let Some(f_pt) = f {
888                assert!(f_pt.dist_to_ref >= 0.);
889                assert!(c.dist_to_ref <= 0.);
890                log::trace!("comparaing c : {:?} f : {:?}", -(c.dist_to_ref), f_pt.dist_to_ref);
891                // IF return_points is full (>= ef), AND closest candidate is further than furthest result
892                if return_points.len() >= ef && -(c.dist_to_ref) > f_pt.dist_to_ref {
893                    // this comparison requires that we are sure that distances compared are distances to the same point : 
894                    // This is the case we compare distance to point passed as arg.
895                    log::trace!("fast return from search_layer, nb points : {:?} \n \t c {:?} \n \t f {:?} dists: {:?}  {:?}", 
896                                    return_points.len(), c.point_ref.p_id, f_pt.point_ref.p_id, -(c.dist_to_ref), f_pt.dist_to_ref);
897                    return return_points;
898                }
899            }
900            // We already have the if-check logic replacing fast return above!
901            // now we scan neighborhood of c in layer and increment visited_point, candidate_points 
902            // and optimize candidate_points so that it contains points with lowest distances to point arg
903            // 
904            let neighbours_c_l = &c.point_ref.neighbours.read()[layer as usize];
905            let c_pid = c.point_ref.p_id;
906            log::trace!("       search_layer, {:?} has  nb neighbours  : {:?} ", c_pid, neighbours_c_l.len());
907            for e in neighbours_c_l {
908                // HERE WE sEE THAT neighbours should be stored as PointIdWithOrder !!
909                // CAVEAT what if several point_id with same distance to ref point?
910                if visited_point_id.contains_key(&e.point_ref.p_id) != true {
911                    visited_point_id.insert(e.point_ref.p_id, Arc::clone(&e.point_ref));
912                    log::trace!("             visited insertion {:?}", e.point_ref.p_id);
913                    let e_dist_to_p = self.dist_f.eval(point, & e.point_ref.v);
914                    
915                    // We must EXPLORE this node regardless of filter, but filter determines if it enters return_points
916                    let mut enters_return = false;
917                    
918                    if let Some(f) = filter {
919                        if !f.contains(e.point_ref.origin_id as u32) {
920                            // Does not match filter, CANNOT be a return point
921                            enters_return = false;
922                        } else {
923                            if return_points.len() < ef {
924                                enters_return = true;
925                            } else if let Some(f_pt) = return_points.peek() {
926                                if e_dist_to_p < f_pt.dist_to_ref {
927                                    enters_return = true;
928                                }
929                            }
930                        }
931                    } else {
932                        if return_points.len() < ef {
933                            enters_return = true;
934                        } else if let Some(f_pt) = return_points.peek() {
935                            if e_dist_to_p < f_pt.dist_to_ref {
936                                enters_return = true;
937                            }
938                        }
939                    }
940
941                    // For exploring, we use the graph traversal logic!
942                    // Is this node promising enough to explore?
943                    let is_promising = if return_points.len() < ef {
944                        true
945                    } else if let Some(f_pt) = return_points.peek() {
946                        e_dist_to_p < f_pt.dist_to_ref
947                    } else {
948                        true
949                    };
950
951                    if is_promising || enters_return {
952                        let e_prime = Arc::new(PointWithOrder::new(&e.point_ref, e_dist_to_p));
953                        // a neighbour of neighbour is better, we insert it into candidate with the distance to point
954                        log::trace!("                inserting new candidate {:?}", e_prime.point_ref.p_id);
955                        candidate_points.push(Arc::new(PointWithOrder::new(&e.point_ref, -e_dist_to_p)));
956                        
957                        if enters_return {
958                            return_points.push(Arc::clone(&e_prime));
959                            if return_points.len() > ef {
960                                return_points.pop();
961                            }
962                        }
963                    }
964                }
965            } // end of for on neighbours_c
966        } // end of while in candidates
967        //
968        trace!("return from search_layer, nb points : {:?}", return_points.len());
969        return_points
970    } // end of search_layer

1107      //  println!("reverse update neighbourhood for  new point {:?} ", new_point.p_id);
1108        log::trace!("reverse update neighbourhood for  new point {:?} ", new_point.p_id);

1111            for q in &new_point.neighbours.read()[l as usize] {

1162    // This is best explained in : Navarro. Searching in metric spaces by spatial approximation.
1163    /// simplest searh neighbours
1164    // The binary heaps here is with negative distance sorted.
1165    fn select_neighbours(&self, data: &[T], candidates: &mut BinaryHeap<Arc<PointWithOrder<T>> >, 
1166                        nb_neighbours_asked: usize, 
1167                        extend_candidates_asked: bool, 
1168                        layer:u8,
1169                        keep_pruned: bool,
1170                        neighbours_vec : &mut Vec<Arc<PointWithOrder<T>> >)
1171        { 
1172        //
1173        log::trace!("entering select_neighbours : nb candidates: {}", candidates.len());
1174        //
1175        neighbours_vec.clear();
1176        // we will extend if we do not have enough candidates and it is explicitly asked in arg
1177        let mut extend_candidates = false;
1178        if candidates.len() <= nb_neighbours_asked {
1179           if !extend_candidates_asked {
1180               // just transfer taking care of signs
1181               while !candidates.is_empty() {
1182                    let p = candidates.pop().unwrap();
1183                    assert!(-p.dist_to_ref >= 0.);
1184                    neighbours_vec.push(Arc::new(PointWithOrder::new(&p.point_ref, -p.dist_to_ref)));
1185                }
1186                return;
1187           }
1188            else {
1189                extend_candidates = true;
1190            }
1191        }
1192        //
1193        //
1194        //extend_candidates = true;
1195        //
1196        if extend_candidates {
1197            let mut candidates_set = HashMap::<PointId, Arc<Point<T>> >::new();
1198            for c in candidates.iter() {
1199                    candidates_set.insert(c.point_ref.p_id, Arc::clone(&c.point_ref));
1200            }
1201            let mut new_candidates_set = HashMap::<PointId, Arc<Point<T>> >::new();
1202            // get a list of all neighbours of candidates
1203            for (_p_id, p_point) in candidates_set.iter() {
1204                let n_p_layer = &p_point.neighbours.read()[layer as usize];
1205                for q in n_p_layer {
1206                    if !candidates_set.contains_key(& q.point_ref.p_id) && !new_candidates_set.contains_key(& q.point_ref.p_id) { 
1207                        new_candidates_set.insert(q.point_ref.p_id, Arc::clone(&q.point_ref));
1208                    }
1209                }
1210            } // end of for p
1211            log::trace!("select neighbours extend candidates from  : {:?} adding : {:?}", candidates.len(), new_candidates_set.len());
1212            for (_p_id, p_point) in new_candidates_set.iter() {
1213                let dist_topoint = self.dist_f.eval(data, &p_point.v);
1214                candidates.push(Arc::new(PointWithOrder::new(p_point, -dist_topoint)));
1215            }
1216        }  // end if extend_candidates
1217        //
1218        let mut discarded_points = BinaryHeap::<Arc<PointWithOrder<T> > >::new();
1219        while candidates.len() > 0 && neighbours_vec.len() < nb_neighbours_asked {
1220            // compare distances of e to data. we do not need to recompute dists!
1221            if let Some(e_p) = candidates.pop() {
1222                let mut e_to_insert = true;
1223                let e_point_v = &e_p.point_ref.v;
1224                assert!(e_p.dist_to_ref <= 0.);
1225                // is e_p the nearest to reference? data than to previous neighbours
1226                if neighbours_vec.len() > 0 {
1227                    e_to_insert = !neighbours_vec.iter().any(|d|
1228                        self.dist_f.eval(e_point_v, &(d.point_ref.v))
1229                            <= -e_p.dist_to_ref);
1230                }
1231                if e_to_insert {
1232                    log::trace!("inserting neighbours : {:?} ", e_p.point_ref.p_id);
1233                    neighbours_vec.push(Arc::new(PointWithOrder::new(&e_p.point_ref, -e_p.dist_to_ref)));
1234                } else {
1235                    log::trace!("discarded neighbours : {:?} ", e_p.point_ref.p_id);
1236                    // ep is taken from a binary heap, so it has a negative sign, we keep its sign
1237                    // to store it in another binary heap will possibly need to retain the best ones from the discarde binaryHeap
1238                    if keep_pruned {
1239                        discarded_points.push(Arc::new(PointWithOrder::new(&e_p.point_ref, e_p.dist_to_ref)));
1240                    }
1241                }
1242            }
1243        }
1244        // now this part of neighbours is the most interesting and is distance sorted.
1245    
1246        // not pruned are at the end of neighbours_vec which is not re-sorted , but discarded are sorted.
1247        if keep_pruned {
1248            while discarded_points.len() > 0 && neighbours_vec.len() < nb_neighbours_asked {
1249                let best_point = discarded_points.pop().unwrap();
1250                // do not forget to reverse sign
1251                assert!(best_point.dist_to_ref <= 0.);
1252                neighbours_vec.push(Arc::new(PointWithOrder::new(&best_point.point_ref, -best_point.dist_to_ref)));
1253            }
1254        };
1255        //
1256        if log::log_enabled!(log::Level::Trace) {
1257            log::trace!("exiting select_neighbours : nb candidates: {}", neighbours_vec.len());
1258            for n in neighbours_vec {
1259                println!("   neighbours {:?} ", n.point_ref.p_id);
1260            }
1261        }
1262        //
1263    } // end of select_neighbours
