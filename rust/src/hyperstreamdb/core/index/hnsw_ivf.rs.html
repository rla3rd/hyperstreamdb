5/// HNSW-IVF Hybrid Index Implementation
6/// 
7/// Combines IVF coarse quantization with HNSW fine search for optimal performance.
8/// This is the industry-standard approach used by FAISS, Milvus, and Weaviate.
9/// 
10/// Architecture:
11/// 1. Coarse quantization: Cluster vectors into N lists using k-means (IVF)
12/// 2. Fine search: Build small HNSW graphs within each cluster
13/// 3. Search: Find nearest clusters, then search HNSW graphs in those clusters
14///
15/// Attribution: Underlying HNSW graph logic relies on the vendored `hnsw_rs` library (MIT/Apache 2.0).
16/// Copyright Jean-Pierre Both and hnsw_rs contributors. Vendored and patched to support exact pre-filtering.

37#[derive(Clone, Copy, Debug, Default)]
38pub struct DistL1;
39impl Distance<f32> for DistL1 {
40    fn eval(&self, va: &[f32], vb: &[f32]) -> f32 {
41        crate::core::index::distance::l1_distance(va, vb)
42    }

69#[derive(Clone, Copy, Debug, Default)]
70pub struct DistSparseDot;
71impl Distance<crate::core::index::SparseVector> for DistSparseDot {
72    fn eval(&self, va: &[crate::core::index::SparseVector], vb: &[crate::core::index::SparseVector]) -> f32 {
73        // HNSW-rs uses &[T]. For sparse vectors, T is SparseVector, so we compare the first elements.
74        1.0 - crate::core::index::distance::sparse_dot_product(&va[0].indices, &va[0].values, &vb[0].indices, &vb[0].values)
75    }
76}

78/// Internal wrapper for different HNSW distance metrics
79pub enum HnswGraph {
80    L2(Hnsw<f32, DistL2>),
81    Cosine(Hnsw<f32, DistCosine>),
82    Dot(Hnsw<f32, DistDot>),
83    L1(Hnsw<f32, DistL1>),
84    Hamming(Hnsw<f32, DistHamming>),
85    Jaccard(Hnsw<f32, DistJaccard>),
86    BinaryHamming(Hnsw<u8, DistHammingPacked>),
87    TurboQuant8(Hnsw<u8, crate::core::index::distance::DistL2u8>),
88    TurboQuant4(Hnsw<u8, crate::core::index::distance::DistL2u4>),
89    SparseDot(Hnsw<crate::core::index::SparseVector, DistSparseDot>),
90}

92impl HnswGraph {
93    pub fn search(&self, query: &VectorValue, k: usize, ef: usize, filter: Option<&roaring::RoaringBitmap>) -> Vec<Neighbour> {
94        match (self, query) {
95            (HnswGraph::L2(h), VectorValue::Float32(q)) => h.search(q, k, ef, filter),
96            (HnswGraph::Cosine(h), VectorValue::Float32(q)) => h.search(q, k, ef, filter),
97            (HnswGraph::Dot(h), VectorValue::Float32(q)) => h.search(q, k, ef, filter),
98            (HnswGraph::L1(h), VectorValue::Float32(q)) => h.search(q, k, ef, filter),
99            (HnswGraph::Hamming(h), VectorValue::Float32(q)) => h.search(q, k, ef, filter),
100            (HnswGraph::Jaccard(h), VectorValue::Float32(q)) => h.search(q, k, ef, filter),
101            
102            (HnswGraph::BinaryHamming(h), VectorValue::Binary(q)) => h.search(q, k, ef, filter),
103            (HnswGraph::TurboQuant8(h), VectorValue::Binary(q)) => h.search(q, k, ef, filter),
104            (HnswGraph::TurboQuant4(h), VectorValue::Binary(q)) => h.search(q, k, ef, filter),
105            (HnswGraph::SparseDot(h), VectorValue::Sparse(q)) => h.search(std::slice::from_ref(q), k, ef, filter),
106            
107            // Casting paths
108            (HnswGraph::L2(h), VectorValue::Float16(q)) => h.search(q, k, ef, filter),
109            
110            _ => {
111                tracing::error!("HnswGraph / query type mismatch");
112                vec![]
113            }
114        }
115    }

117    pub fn insert(&self, data: (VectorValue, usize)) {
118        let (val, local_id) = data;
119        match (self, val) {
120            (HnswGraph::L2(h), VectorValue::Float32(v)) => h.insert_slice((&v, local_id)),
121            (HnswGraph::Cosine(h), VectorValue::Float32(v)) => h.insert_slice((&v, local_id)),
122            (HnswGraph::Dot(h), VectorValue::Float32(v)) => h.insert_slice((&v, local_id)),
123            (HnswGraph::L1(h), VectorValue::Float32(v)) => h.insert_slice((&v, local_id)),
124            (HnswGraph::Hamming(h), VectorValue::Float32(v)) => h.insert_slice((&v, local_id)),
125            (HnswGraph::Jaccard(h), VectorValue::Float32(v)) => h.insert_slice((&v, local_id)),
126            
127            (HnswGraph::BinaryHamming(h), VectorValue::Binary(v)) => h.insert_slice((&v, local_id)),
128            (HnswGraph::TurboQuant8(h), VectorValue::Binary(v)) => h.insert_slice((&v, local_id)),
129            (HnswGraph::TurboQuant4(h), VectorValue::Binary(v)) => h.insert_slice((&v, local_id)),
130            (HnswGraph::SparseDot(h), VectorValue::Sparse(v)) => h.insert_slice((&vec![v], local_id)),
131            
132            (HnswGraph::L2(h), VectorValue::Float16(v)) => h.insert_slice((&v, local_id)),
133            
134            _ => tracing::error!("HnswGraph / insert value type mismatch"),
135        }
136    }

138    pub fn parallel_insert(&self, data: Vec<(&[f32], usize)>) {
139        if data.is_empty() { return; }
140        
141        match self {
142            HnswGraph::L2(h) => h.parallel_insert_slice(&data),
143            HnswGraph::Cosine(h) => h.parallel_insert_slice(&data),
144            HnswGraph::Dot(h) => h.parallel_insert_slice(&data),
145            HnswGraph::L1(h) => h.parallel_insert_slice(&data),
146            _ => {
147                // Fallback to sequential for other types
148                for (v, id) in data {
149
150                    let val = VectorValue::Float32(v.to_vec());
151                    self.insert((val, id));
152                }
153            }
154        }
155    }

175/// HNSW-IVF Hybrid Index
176pub struct HnswIvfIndex {
177    /// Cluster centroids for coarse quantization
178    centroids: Vec<Vec<f32>>,
179    /// Distance metric used for search
180    metric: VectorMetric,
181    /// Per-cluster HNSW graphs: cluster_id -> (hnsw_index, row_id_mapping)
182    cluster_graphs: HashMap<usize, (HnswGraph, Vec<usize>)>,
183    /// Number of clusters (for metadata/scaling)
184    _n_lists: usize,
185    /// Vector dimensionality
186    dim: usize,
187    /// Optional quantizer for compression (TurboQuant or PQ)
188    pub quantizer: Option<QuantizerImpl>,
189    /// Hardware acceleration context (Open Source)
190    _compute_context: crate::core::index::gpu::ComputeContext,
191}
192
193impl HnswIvfIndex {
194    /// Build HNSW-IVF index from vectors
195    pub fn build(
196        vectors: Vec<Vec<f32>>,
197        metric: VectorMetric,
198        n_lists: Option<usize>,
199        hnsw_m: Option<usize>,
200        algo: &IndexAlgorithm,
201    ) -> Result<Self> {
202        let start = std::time::Instant::now();
203        if vectors.is_empty() {
204            anyhow::bail!("Cannot build HNSW-IVF index from empty vector set");
205        }
206
207        let dim = vectors[0].len();
208        let n_vectors = vectors.len();
209        
210        // Auto-detect optimal cluster count
211        let num_cpus = num_cpus::get();
212        let n_lists = n_lists.unwrap_or_else(|| {
213            if n_vectors < 1000 {
214                num_cpus.min(4)
215            } else if n_vectors < 10_000 {
216                num_cpus.max(16)
217            } else {
218                ((n_vectors as f64).sqrt() / 5.0) as usize
219            }
220        }).max(1).min(n_vectors / 10).max(1);
221        
222        let hnsw_m = hnsw_m.unwrap_or(16);
223        
224        tracing::info!("Building HNSW-IVF index: {} vectors, {} clusters, algo={:?}", 
225                 n_vectors, n_lists, algo);
226
227        // Step 1: Cluster vectors using k-means
228        let t0 = std::time::Instant::now();
229        let max_iters = 10;
230        let (centroids, labels) = simple_kmeans(&vectors, n_lists, max_iters)?;
231        
232        // Step 2: Train Quantizer if requested
233        let quantizer = match algo {
234            IndexAlgorithm::HnswTq8 { .. } => {
235                Some(QuantizerImpl::TurboQuant(TurboQuantEncoder::train(&vectors, 8)?))
236            },
237            IndexAlgorithm::HnswTq4 { .. } => {
238                Some(QuantizerImpl::TurboQuant(TurboQuantEncoder::train(&vectors, 4)?))
239            },
240            IndexAlgorithm::HnswPq { compression, .. } => {
241                let subspaces = dim / compression;
242                let config = PqConfig { m: subspaces, k: 256, dim };
243                Some(QuantizerImpl::Pq(PqEncoder::train(&vectors, config)?))
244            },
245            _ => None,
246        };
247
248        tracing::debug!("  - K-Means took: {:.2?} ({} iterations, {} clusters)", t0.elapsed(), max_iters, n_lists);
249
250        // Step 2: Group vectors by cluster in parallel
251        let t1 = std::time::Instant::now();
252        let cluster_vectors: HashMap<usize, Vec<(Vec<f32>, usize)>> = vectors
253            .into_par_iter()
254            .zip(labels.into_par_iter())
255            .enumerate()
256            .fold(
257                || HashMap::<usize, Vec<(Vec<f32>, usize)>>::new(),
258                |mut acc, (row_id, (vec, cluster_id))| {
259                    acc.entry(cluster_id as usize).or_default().push((vec, row_id));
260                    acc
261                }
262            )
263            .reduce(
264                || HashMap::new(),
265                |mut a, b| {
266                    for (k, v) in b {
267                        a.entry(k).or_default().extend(v);
268                    }
269                    a
270                }
271            );
272        tracing::debug!("  - Grouping vectors took: {:.2?}", t1.elapsed());
273
274        // Step 3: Build HNSW graph for each cluster in parallel
275        let t2 = std::time::Instant::now();
276        
277        let q_ref = quantizer.as_ref();
278
279        let cluster_graphs: HashMap<usize, (HnswGraph, Vec<usize>)> = cluster_vectors
280            .into_par_iter()
281            .map(|(cluster_id, vecs): (usize, Vec<(Vec<f32>, usize)>)| {
282                let max_layers = 16;
283                let ef_construction = (hnsw_m * 2).max(40);
284                
285                // Determine HNSW node type and distance metric
286                let hnsw = if let Some(q) = quantizer.as_ref() {
287                    if q.bits() == 4 {
288                        HnswGraph::TurboQuant4(Hnsw::new(hnsw_m, vecs.len(), max_layers, ef_construction, crate::core::index::distance::DistL2u4))
289                    } else {
290                        HnswGraph::TurboQuant8(Hnsw::new(hnsw_m, vecs.len(), max_layers, ef_construction, crate::core::index::distance::DistL2u8))
291                    }
292                } else {
293                    match metric {
294                        VectorMetric::L2 => HnswGraph::L2(Hnsw::new(hnsw_m, vecs.len(), max_layers, ef_construction, DistL2)),
295                        VectorMetric::Cosine => HnswGraph::Cosine(Hnsw::new(hnsw_m, vecs.len(), max_layers, ef_construction, DistCosine)),
296                        VectorMetric::InnerProduct => HnswGraph::Dot(Hnsw::new(hnsw_m, vecs.len(), max_layers, ef_construction, DistDot)),
297                        VectorMetric::L1 => HnswGraph::L1(Hnsw::new(hnsw_m, vecs.len(), max_layers, ef_construction, DistL1)),
298                        VectorMetric::Hamming => HnswGraph::Hamming(Hnsw::new(hnsw_m, vecs.len(), max_layers, ef_construction, DistHamming)),
299                        VectorMetric::Jaccard => HnswGraph::Jaccard(Hnsw::new(hnsw_m, vecs.len(), max_layers, ef_construction, DistJaccard)),
300                    }
301                };
302                
303                // Build locally sequential
304                for (local_id, (vec, _)) in vecs.iter().enumerate() {
305                    if let Some(ref q) = q_ref {
306                        let encoded = q.encode(vec);
307                        hnsw.insert((crate::core::index::VectorValue::Binary(encoded), local_id));
308                    } else {
309                        hnsw.insert((crate::core::index::VectorValue::Float32(vec.clone()), local_id));
310                    }
311                }
312                
313                let row_id_mapping: Vec<usize> = vecs.iter().map(|(_, row_id)| *row_id).collect();
314                (cluster_id, (hnsw, row_id_mapping))
315            })
316            .collect();
317            
318        tracing::debug!("  - Building HNSW graphs took: {:.2?} ({} clusters)", t2.elapsed(), cluster_graphs.len());
319
320        tracing::info!("Hnsw-IVF index built in {:.2?}: {} non-empty clusters", start.elapsed(), cluster_graphs.len());
321
322        Ok(HnswIvfIndex {
323            centroids,
324            metric,
325            cluster_graphs,
326            _n_lists: n_lists,
327            dim,
328            quantizer,
329            _compute_context: crate::core::index::gpu::ComputeContext::auto_detect(),
330        })
331    }

333    /// Search for k nearest neighbors
334    pub fn search(&self, query: &VectorValue, k: usize, n_probe: usize, filter: Option<&roaring::RoaringBitmap>) -> Result<Vec<(usize, f32)>> {
335        let query_f32 = match query {
336            VectorValue::Float32(v) => v,
337            VectorValue::Float16(v) => v,
338            VectorValue::Binary(_) => return Ok(vec![]), 
339            VectorValue::Sparse(_) => return Ok(vec![]), 
340            VectorValue::Keyword(_) => return Ok(vec![]), 
341        };
342
343        // Pre-quantize query if needed for faster graph traversal (SDC)
344        let effective_query = if let Some(QuantizerImpl::TurboQuant(ref q)) = self.quantizer {
345            VectorValue::Binary(q.encode(query_f32))
346        } else {
347            query.clone()
348        };
349
350        // Step 1: Find n_probe nearest clusters (coarse search)
351        // (Centroids are always f32 for stability)
352        let all_centroids_flat: Vec<f32> = self.centroids.iter().flatten().cloned().collect();
353
354        let distances = crate::core::index::gpu::compute_distance(query_f32, &all_centroids_flat, self.dim, self.metric)
355            .unwrap_or_else(|_| {
356                match self.metric {
357                    VectorMetric::L2 => crate::core::index::distance::l2_distance_batch(query_f32, &self.centroids),
358                    VectorMetric::Cosine => crate::core::index::distance::cosine_similarity_batch(query_f32, &self.centroids),
359                    VectorMetric::InnerProduct => crate::core::index::distance::dot_product_batch(query_f32, &self.centroids),
360                    VectorMetric::L1 => self.centroids.par_iter().map(|c| crate::core::index::distance::l1_distance(query_f32, c)).collect(),
361                    VectorMetric::Hamming => self.centroids.par_iter().map(|c| crate::core::index::distance::hamming_distance(query_f32, c)).collect(),
362                    VectorMetric::Jaccard => self.centroids.par_iter().map(|c| crate::core::index::distance::jaccard_distance(query_f32, c)).collect(),
363                }
364            });
365
366        let mut cluster_distances: Vec<(usize, f32)> = distances.into_iter().enumerate().collect();
367        cluster_distances.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(std::cmp::Ordering::Equal));
368        
369        let clusters_to_search: Vec<usize> = cluster_distances.iter().take(n_probe).map(|(i, _)| *i).collect();
370
371        // Step 2: Search HNSW graphs in selected clusters in parallel (fine search)
372        let mut candidates: Vec<(usize, f32)> = clusters_to_search
373            .into_par_iter()
374            .flat_map(|cluster_id| {
375                if let Some((hnsw, row_id_mapping)) = self.cluster_graphs.get(&cluster_id) {
376                    let cluster_k = if filter.is_some() { k * 4 } else { k * 2 };
377                    let ef_search = if filter.is_some() { 100 } else { 40 };
378                    
379                    let mut local_filter_opt: Option<roaring::RoaringBitmap> = None;
380                    if let Some(f) = filter {
381                        let mut bm = roaring::RoaringBitmap::new();
382                        for (local_id, &global_id) in row_id_mapping.iter().enumerate() {
383                            if f.contains(global_id as u32) {
384                                bm.insert(local_id as u32);
385                            }
386                        }
387                        local_filter_opt = Some(bm);
388                    }
389                    
390                    let results = hnsw.search(&effective_query, cluster_k, ef_search, local_filter_opt.as_ref());
391                    
392                    // Map local indices back to global row IDs and apply filter
393                    results.into_iter().filter_map(|neighbor| {
394                        let local_id = neighbor.d_id;
395                        if local_id < row_id_mapping.len() {
396                            let global_id = row_id_mapping[local_id];
397                            if let Some(f) = filter {
398                                if f.contains(global_id as u32) {
399                                    Some((global_id, neighbor.distance))
400                                } else {
401
402                                    None
403                                }
404                            } else {
405                                Some((global_id, neighbor.distance))
406                            }
407                        } else {
408                            None
409                        }
410                    }).collect::<Vec<_>>()
411                } else {
412                    vec![]
413                }
414            })
415            .collect();
416
417        // Step 3: Merge and return top-k
418        candidates.par_sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
419        candidates.dedup_by_key(|x| x.0);  // Remove duplicates
420        candidates.truncate(k);
421        Ok(candidates)
422    }

424    /// Save HNSW-IVF index to disk
425    pub fn save(&self, local_path: &str) -> Result<Vec<String>> {
426        use arrow::array::{UInt32Array, ListBuilder, Float32Builder};
427        use arrow::datatypes::{Schema, Field, DataType};
428        use arrow::record_batch::RecordBatch;
429        use parquet::arrow::ArrowWriter;
430        use parquet::file::properties::WriterProperties;
431        use std::fs::File;
432        use std::sync::Arc;
433
434        let mut saved_files = Vec::new();
435
436        // 1. Save centroids
437        let centroids_path = format!("{}.centroids.parquet", local_path);
438        let centroids_tmp = format!("{}.tmp", centroids_path);
439        {
440            let mut centroid_builder = ListBuilder::new(Float32Builder::new());
441            for centroid in &self.centroids {
442                for &val in centroid {
443                    centroid_builder.values().append_value(val);
444                }
445                centroid_builder.append(true);
446            }
447            let centroid_array = Arc::new(centroid_builder.finish());
448
449            let schema = Arc::new(Schema::new(vec![
450                Field::new("centroid", DataType::List(
451                    Arc::new(Field::new("item", DataType::Float32, true))
452                ), false),
453            ]));
454
455            let batch = RecordBatch::try_new(schema.clone(), vec![centroid_array])?;
456            
457            let file = File::create(&centroids_tmp)?;
458            let props = WriterProperties::builder()
459                .set_key_value_metadata(Some(vec![
460                    parquet::file::metadata::KeyValue {
461                        key: "vector_metric".to_string(),
462                        value: Some(match self.metric {
463                            VectorMetric::L2 => "l2".to_string(),
464                            VectorMetric::Cosine => "cosine".to_string(),
465                            VectorMetric::InnerProduct => "ip".to_string(),
466                            VectorMetric::L1 => "l1".to_string(),
467                            VectorMetric::Hamming => "hamming".to_string(),
468                            VectorMetric::Jaccard => "jaccard".to_string(),
469                        }),
470                    },
471                    parquet::file::metadata::KeyValue {
472                        key: "quantizer_config".to_string(),
473                        value: self.quantizer.as_ref().map(|q| serde_json::to_string(q).unwrap_or_default()),
474                    }
475                ]))
476                .build();
477            let mut writer = ArrowWriter::try_new(file, schema, Some(props))?;
478            writer.write(&batch)?;
479            writer.close()?;
480            
481            std::fs::rename(&centroids_tmp, &centroids_path)?;
482            saved_files.push(centroids_path.clone());
483        }
484
485        // 2. Save per-cluster HNSW graphs
486        for (cluster_id, (hnsw, row_id_mapping)) in &self.cluster_graphs {
487            let cluster_base = format!("{}.cluster_{}", local_path, cluster_id);
488            let cluster_tmp = format!("{}.tmp", cluster_base);
489            
490            hnsw.file_dump(&cluster_tmp).map_err(|e| anyhow::anyhow!("HNSW dump failed: {}", e))?;
491            
492            let graph_orig = format!("{}.hnsw.graph", cluster_tmp);
493            let data_orig = format!("{}.hnsw.data", cluster_tmp);
494            let graph_dest = format!("{}.hnsw.graph", cluster_base);
495            let data_dest = format!("{}.hnsw.data", cluster_base);
496            
497            if !std::path::Path::new(&graph_orig).exists() {
498                 anyhow::bail!("HNSW dump did not create expected file: {}", graph_orig);
499            }
500            if !std::path::Path::new(&data_orig).exists() {
501                 anyhow::bail!("HNSW dump did not create expected file: {}", data_orig);
502            }
503            
504            std::fs::rename(&graph_orig, &graph_dest)?;
505            std::fs::rename(&data_orig, &data_dest)?;
506
507            saved_files.push(graph_dest);
508            saved_files.push(data_dest);
509            
510            let mapping_path = format!("{}.cluster_{}.mapping.parquet", local_path, cluster_id);
511            let mapping_tmp = format!("{}.tmp", mapping_path);
512            
513            {
514                let row_ids: Vec<u32> = row_id_mapping.iter().map(|&x| x as u32).collect();
515                let row_id_array = Arc::new(UInt32Array::from(row_ids));
516
517                let schema = Arc::new(Schema::new(vec![
518                    Field::new("row_id", DataType::UInt32, false),
519                ]));
520
521                let batch = RecordBatch::try_new(schema.clone(), vec![row_id_array])?;
522                
523                let file = File::create(&mapping_tmp)?;
524                let props = WriterProperties::builder().build();
525                let mut writer = ArrowWriter::try_new(file, schema, Some(props))?;
526                writer.write(&batch)?;
527                writer.close()?;
528                
529                std::fs::rename(&mapping_tmp, &mapping_path)?;
530                saved_files.push(mapping_path);
531            }
532        }
533
534        Ok(saved_files)
535    }

537    pub async fn load_puffin_async(store: Arc<dyn ObjectStore>, path: &str) -> Result<Arc<Self>> {
538        let path_obj = Path::from(path);
539        let bytes = store.get(&path_obj).await?.bytes().await?;
540        let mut reader = PuffinReader::new(Cursor::new(bytes))?;
541        
542        let mut centroids = Vec::new();
543        let mut cluster_graphs = HashMap::new();
544        let mut dim = 0;
545        let mut metric = VectorMetric::L2;
546        let mut _quantizer: Option<crate::core::index::QuantizerImpl> = None;

548        // Temporary storage for building clusters

550        let mut graphs_graph: HashMap<usize, Vec<u8>> = HashMap::new();
551        let mut graphs_mapping: HashMap<usize, Vec<u8>> = HashMap::new();
552
553        let blobs = reader.footer().blobs.clone();
554        for (i, blob) in blobs.iter().enumerate() {
555            match blob.r#type.as_str() {
556                "hnsw-ivf-centroids" => {
557                    let data = reader.read_blob(i)?;
558                    // Decode centroids from parquet or raw? For now let's assume it's the centroids parquet bytes
559                    // BUT: We could just store them as raw F32 blobs for speed.
560                    // Let's stick to the current Parquet centroids for now to reuse logic.
561                    use parquet::arrow::arrow_reader::ParquetRecordBatchReaderBuilder;
562                    let builder = ParquetRecordBatchReaderBuilder::try_new(bytes::Bytes::from(data))?;
563                    let reader = builder.build()?;
564                    let batches: Vec<RecordBatch> = reader.map(|r| r.map_err(anyhow::Error::from)).collect::<Result<Vec<_>>>()?;
565                    let batch = arrow::compute::concat_batches(&batches[0].schema(), &batches)?;
566                    let centroid_list = batch.column(0).as_list::<i32>();
567                    for j in 0..centroid_list.len() {
568                        let values = centroid_list.value(j);
569                        let float_array = values.as_primitive::<arrow::datatypes::Float32Type>();
570                        centroids.push(float_array.values().to_vec());
571                    }
572                    if !centroids.is_empty() {
573                        dim = centroids[0].len();
574                    }
575                    
576                    if let Some(m_str) = blob.properties.get("vector-metric") {
577                        metric = match m_str.as_str() {
578                            "l2" => VectorMetric::L2,
579                            "cosine" => VectorMetric::Cosine,
580                            "ip" => VectorMetric::InnerProduct,
581                            _ => VectorMetric::L2,
582                        };
583                    }
584                },
585                "hnsw-cluster-graph" => {
586                    if let Some(cid_str) = blob.properties.get("cluster-id") {
587                        if let Ok(cid) = cid_str.parse::<usize>() {
588                            graphs_graph.insert(cid, reader.read_blob(i)?);
589                        }
590                    }
591                },
592                "hnsw-cluster-data" => {
593                    if let Some(cid_str) = blob.properties.get("cluster-id") {
594                        if let Ok(cid) = cid_str.parse::<usize>() {
595                            graphs_data.insert(cid, reader.read_blob(i)?);
596                        }
597                    }
598                },
599                "hnsw-cluster-mapping" => {
600                    if let Some(cid_str) = blob.properties.get("cluster-id") {
601                        if let Ok(cid) = cid_str.parse::<usize>() {
602                            graphs_mapping.insert(cid, reader.read_blob(i)?);
603                        }
604                    }
605                },
606                _ => {}
607            }
608        }
609        
610        // Assemble clusters
611        for (cid, graph_bytes) in graphs_graph {
612            let data_bytes = graphs_data.remove(&cid).context("Missing cluster data")?;
613            let mapping_bytes = graphs_mapping.remove(&cid).context("Missing cluster mapping")?;
614            
615            let mut graph_reader = std::io::BufReader::new(Cursor::new(graph_bytes));
616            let mut data_reader = std::io::BufReader::new(Cursor::new(data_bytes));
617            
618            let description = crate::core::index::hnsw_rs::hnswio::load_description(&mut graph_reader)
619                .map_err(|e| anyhow::anyhow!("Failed to load HNSW description: {}", e))?;
620            
621            let hnsw = match metric {
622                VectorMetric::L2 => HnswGraph::L2(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistL2, &mut data_reader)
623                    .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
624                VectorMetric::Cosine => HnswGraph::Cosine(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistCosine, &mut data_reader)
625                    .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
626                VectorMetric::InnerProduct => HnswGraph::Dot(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistDot, &mut data_reader)
627                    .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
628                VectorMetric::L1 => HnswGraph::L1(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistL1, &mut data_reader)
629                    .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
630                VectorMetric::Hamming => HnswGraph::Hamming(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistHamming, &mut data_reader)
631                    .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
632                VectorMetric::Jaccard => HnswGraph::Jaccard(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistJaccard, &mut data_reader)
633                    .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
634            };
635            
636            use parquet::arrow::arrow_reader::ParquetRecordBatchReaderBuilder;
637            let map_builder = ParquetRecordBatchReaderBuilder::try_new(bytes::Bytes::from(mapping_bytes))?;
638            let map_reader = map_builder.build()?;
639            let batches: Vec<RecordBatch> = map_reader.map(|r| r.map_err(anyhow::Error::from)).collect::<Result<Vec<_>>>()?;
640            let batch = arrow::compute::concat_batches(&batches[0].schema(), &batches)?;
641            let row_id_array = batch.column(0).as_primitive::<arrow::datatypes::UInt32Type>();
642            let row_id_mapping: Vec<usize> = row_id_array.values().iter().map(|&x| x as usize).collect();
643            
644            cluster_graphs.insert(cid, (hnsw, row_id_mapping));
645        }
646        
647        let index = HnswIvfIndex {
648            centroids,
649            metric,
650            cluster_graphs,
651            _n_lists: 0, // Not strictly used for search
652            dim,
653            quantizer: None,
654            _compute_context: crate::core::index::gpu::ComputeContext::auto_detect(),
655        };
656        
657        let index_arc = Arc::new(index);
658        // HNSW_IVF_CACHE.insert(path.to_string(), index_arc.clone()).await;
659        
660        Ok(index_arc)
661    }

866        };
867        
868        let index_arc = Arc::new(index);
869        HNSW_IVF_CACHE.insert(cache_key_str, index_arc.clone()).await;
870        
871        Ok(index_arc)
872    }

968                    HnswGraph::TurboQuant8(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, crate::core::index::distance::DistL2u8, &mut data_reader)
969                        .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?)
970                }
971            } else {
972                match metric {
973                    VectorMetric::L2 => HnswGraph::L2(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistL2, &mut data_reader)
974                        .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
975                    VectorMetric::Cosine => HnswGraph::Cosine(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistCosine, &mut data_reader)
976                        .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
977                    VectorMetric::InnerProduct => HnswGraph::Dot(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistDot, &mut data_reader)
978                        .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
979                    VectorMetric::L1 => HnswGraph::L1(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistL1, &mut data_reader)
980                        .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
981                    VectorMetric::Hamming => HnswGraph::Hamming(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistHamming, &mut data_reader)
982                        .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
983                    VectorMetric::Jaccard => HnswGraph::Jaccard(crate::core::index::hnsw_rs::hnswio::load_hnsw_with_dist(&mut graph_reader, &description, DistJaccard, &mut data_reader)
984                        .map_err(|e| anyhow::anyhow!("HNSW load failed: {}", e))?),
985                }
986            };
987            
988            let file = File::open(&mapping_path)?;
989            let builder = ParquetRecordBatchReaderBuilder::try_new(file)?;
990            let reader = builder.build()?;
991            
992            let batches: Vec<RecordBatch> = reader.collect::<std::result::Result<_, _>>()?;
993            if batches.is_empty() {
994                 anyhow::bail!("No data in mapping file");
995            }
996            let batch = arrow::compute::concat_batches(&batches[0].schema(), &batches)?;
997            
998            let row_id_array = batch.column(0).as_primitive::<arrow::datatypes::UInt32Type>();
999            let row_id_mapping: Vec<usize> = row_id_array.values().iter().map(|&x| x as usize).collect();
1000            
1001            cluster_graphs.insert(cluster_id, (hnsw, row_id_mapping));
1002        }
1003
1004        println!("Loaded {} cluster graphs", cluster_graphs.len());
1005
1006        Ok(HnswIvfIndex {
1007            centroids,
1008            metric,
1009            cluster_graphs,
1010            _n_lists: n_lists,
1011            dim,
1012            quantizer,
1013            _compute_context: crate::core::index::gpu::ComputeContext::auto_detect(),
1014        })
1015    }

1017    pub fn size_in_bytes(&self) -> usize {
1018        let mut size = 0;
1019        for c in &self.centroids {
1020            size += c.len() * 4;
1021        }
1022        for (_hnsw, mapping) in self.cluster_graphs.values() {
1023            size += mapping.len() * 8;
1024            let data_count = mapping.len();
1025            size += data_count * self.dim * 4;
1026            size += data_count * 16 * 4; 
1027        }
1028        size
1029    }

1037    #[test]
1038    fn test_hnsw_ivf_pre_filtering() {
1039        let mut vectors = Vec::new();
1040        for i in 0..100 {
1041            let mut vec = vec![0.0; 4];
1042            vec[0] = (i as f32) / 100.0;
1043            vectors.push(vec);
1044        }
1045
1046        let algo = crate::core::manifest::IndexAlgorithm::hnsw();
1047        let index = HnswIvfIndex::build(vectors.clone(), VectorMetric::L2, Some(2), Some(16), &algo).unwrap();
1048        let query = VectorValue::Float32(vectors[50].clone());
1049
1050        // Unfiltered search (should find ID 50 easily since it's an exact match)
1051        let results_unfiltered = index.search(&query, 5, 40, None).unwrap();
1052        assert!(!results_unfiltered.is_empty(), "Should return results");
1053        
1054        let mut found_50 = false;
1055        for (id, _) in &results_unfiltered {
1056            if *id == 50 { found_50 = true; break; }
1057        }
1058        assert!(found_50, "Unfiltered search should easily find exact match ID 50");
1059
1060        // Filtered search (exclude ID 50, only allow 51, 52, 53)
1061        let mut filter = RoaringBitmap::new();
1062        filter.insert(51);
1063        filter.insert(52);
1064        filter.insert(53);
1065
1066        let results_filtered = index.search(&query, 3, 40, Some(&filter)).unwrap();
1067        assert!(!results_filtered.is_empty(), "Filtered search should return results");
1068
1069        let mut found_50_filtered = false;
1070        let mut found_51_filtered = false;
1071        for (id, _) in &results_filtered {
1072            if *id == 50 { found_50_filtered = true; }
1073            if *id == 51 { found_51_filtered = true; }
1074        }
1075
1076        assert!(!found_50_filtered, "Pre-filtering failed to exclude exact match ID 50");
1077        assert!(found_51_filtered, "Pre-filtering failed to find the next best allowed match ID 51");
1078        
1079        for (id, _) in &results_filtered {
1080            assert!(filter.contains(*id as u32), "Result ID {} was not in the filter!", id);
1081        }
1082    }
