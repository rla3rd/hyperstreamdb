130/// Generates an Excel-style column label (A, B, C... AA, AB...) for a given index.
131pub fn excel_column_label(mut index: usize) -> String {
132    let mut label = String::new();
133    loop {
134        let remainder = index % 26;
135        label.push((b'A' + remainder as u8) as char);
136        if index < 26 {
137            break;
138        }
139        index = (index / 26) - 1;
140    }
141    label.chars().rev().collect()
142}

180impl Drop for Table {
181    fn drop(&mut self) {
182        if let Ok(tasks) = self.background_tasks.try_lock() {
183            let pending = tasks.iter().filter(|t| !t.is_finished()).count();
184            if pending > 0 {
185                tracing::warn!(
186                    "Table instance for '{}' dropped with {} pending background tasks. These tasks are now detached.",
187                    self.uri,
188                    pending
189                );
190            }
191        }
192    }

232    /// Asynchronously set the primary key. This commits a new manifest version
233    /// and ensures the columns are marked as NOT NULL (required: true).
234    pub async fn set_primary_key_async(&self, columns: Vec<String>) -> Result<()> {
235        let manifest = self.manifest().await?;
236        let latest_schema = manifest.schemas.last().ok_or_else(|| anyhow::anyhow!("No schema found"))?;
237        
238        let mut field_ids = Vec::new();
239        for col in &columns {
240            let id = latest_schema.fields.iter()
241                .find(|f| f.name == *col)
242                .map(|f| f.id)
243                .ok_or_else(|| anyhow::anyhow!("Column '{}' not found in schema", col))?;
244            field_ids.push(id);
245        }
246
247        let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
248        manifest_manager.update_identifier_fields(field_ids).await?;
249
250        // Update in-memory state
251        let mut pk = self.primary_key.write().unwrap();
252        *pk = columns;
253        
254        // Update in-memory schema ref (it's slightly stale now, but will refresh on next use)
255        // or we could force a reload.
256        let (new_manifest, _) = manifest_manager.load_latest().await?;
257        if let Some(s) = new_manifest.schemas.last() {
258            let mut schema_lock = self.schema.write().unwrap();
259            *schema_lock = Arc::new(s.to_arrow());
260        }
261
262        Ok(())
263    }

289    /// Update indexing specifications for multiple columns at once.
290    /// This is an atomic operation that commits a new manifest version.
291    pub async fn set_index_columns(&self, column_indexes: HashMap<String, Vec<IndexAlgorithm>>) -> Result<()> {
292        let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
293        manifest_manager.update_index_specs(column_indexes.clone()).await?;
294
295        // Update in-memory state
296        {
297            let mut index_cols = self.indexing.index_columns.write().unwrap();
298            let mut index_configs = self.indexing.index_configs.write().unwrap();
299            
300            for (col, algs) in &column_indexes {
301                if algs.is_empty() {
302                    // Drop index
303                    index_cols.retain(|c| c != col);
304                    index_configs.remove(col);
305                } else {
306                    if !index_cols.contains(col) {
307                        index_cols.push(col.clone());
308                    }
309                    
310                    // Extract tokenizer from algorithms if present
311                    let tokenizer = algs.iter().find_map(|alg| {
312                        match alg {
313                            IndexAlgorithm::Bm25 { tokenizer, .. } => {
314                                if tokenizer.is_empty() || tokenizer == "default" {
315                                    Some("default".to_string())
316                                } else {
317                                    Some(tokenizer.clone())
318                                }
319                            },
320                            _ => None,
321                        }
322                    });
323
324                    // Update config
325                    let config = index_configs.entry(col.clone()).or_insert_with(|| ColumnIndexConfig {
326                        enabled: true,
327                        algorithms: algs.clone(),
328                        ..Default::default()
329                    });
330                    config.algorithms = algs.clone();
331                    if let Some(tok) = tokenizer {
332                        config.tokenizer = Some(tok);
333                    }
334                }
335            }
336            index_cols.sort();
337        }
338
339        // Trigger backfill for updated columns
340        let cols_to_backfill: Vec<String> = column_indexes.keys().cloned().collect();
341        let table_clone = self.clone();
342        let handle = tokio::spawn(async move {
343            if let Err(e) = table_clone.backfill_indexes_async(cols_to_backfill).await {
344                tracing::error!("Failed to backfill indexes: {}", e);
345            }
346        });
347        
348        self.background_tasks.lock().await.push(handle);
349
350        Ok(())
351    }

353    /// Add an indexing strategy to a column.
354    /// This is an atomic operation that commits a new manifest version.
355    pub async fn add_index(&self, column: String, algorithm: IndexAlgorithm) -> Result<()> {
356        let manifest = self.manifest().await?;
357        let latest_schema = manifest.schemas.last().ok_or_else(|| anyhow::anyhow!("No schema found"))?;
358        
359        let field = latest_schema.fields.iter()
360            .find(|f| f.name == column)
361            .ok_or_else(|| anyhow::anyhow!("Column '{}' not found in schema", column))?;
362
363        let mut next_indexes = field.indexes.clone();
364        
365        // Deduplicate by type for unique algorithms (Vector families, BM25, Bloom)
366        match &algorithm {
367            IndexAlgorithm::Hnsw { .. } | IndexAlgorithm::HnswPq { .. } | 
368            IndexAlgorithm::HnswTq4 { .. } | IndexAlgorithm::HnswTq8 { .. } => {
369                // Vector index family - replace existing ones
370                next_indexes.retain(|idx| !matches!(idx, 
371                    IndexAlgorithm::Hnsw { .. } | IndexAlgorithm::HnswPq { .. } | 
372                    IndexAlgorithm::HnswTq4 { .. } | IndexAlgorithm::HnswTq8 { .. }));
373            }
374            IndexAlgorithm::Bm25 { .. } => {
375                next_indexes.retain(|idx| !matches!(idx, IndexAlgorithm::Bm25 { .. }));
376            }
377            IndexAlgorithm::Bloom { .. } => {
378                next_indexes.retain(|idx| !matches!(idx, IndexAlgorithm::Bloom { .. }));
379            }
380            _ => {
381                if !next_indexes.contains(&algorithm) {
382                    next_indexes.push(algorithm.clone());
383                    return self.set_index_columns(HashMap::from([(column, next_indexes)])).await;
384                }
385            }
386        }
387        
388        next_indexes.push(algorithm);
389        
390        let mut updates = HashMap::new();
391        updates.insert(column, next_indexes);
392        
393        self.set_index_columns(updates).await
394    }

404    /// Add a column to the primary key. 
405    /// This is an atomic operation that commits a new manifest version.
406    /// Validation: Ensures no duplicate keys exist for the new definition.
407    pub async fn add_primary_key(&self, column: String) -> Result<()> {
408        let manifest = self.manifest().await?;
409        let latest_schema = manifest.schemas.last().ok_or_else(|| anyhow::anyhow!("No schema found"))?;
410        
411        // Find field ID for column
412        let field_id = latest_schema.fields.iter()
413            .find(|f| f.name == column)
414            .map(|f| f.id)
415            .ok_or_else(|| anyhow::anyhow!("Column '{}' not found in schema", column))?;
416
417        let mut next_ids = latest_schema.identifier_field_ids.clone();
418        if next_ids.contains(&field_id) {
419            return Ok(()); // Already in PK
420        }
421        next_ids.push(field_id);
422        
423        // Validate uniqueness before committing
424        self._validate_pk_uniqueness(&next_ids, &latest_schema).await?;
425        
426        // Atomic commit to manifest
427        let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
428        manifest_manager.update_identifier_fields(next_ids).await?;
429        
430        // Update in-memory state
431        let mut pk = self.primary_key.write().unwrap();
432        if !pk.contains(&column) {
433            pk.push(column);
434        }
435        Ok(())
436    }

462    /// Internal helper to validate that a set of field IDs form a unique key across existing data.
463    async fn _validate_pk_uniqueness(&self, field_ids: &[i32], schema: &crate::core::manifest::Schema) -> Result<()> {
464        let col_names: Vec<String> = field_ids.iter()
465            .map(|id| schema.fields.iter().find(|f| f.id == *id).map(|f| f.name.clone()).unwrap())
466            .collect();
467            
468        // 1. Acceleration: Single-column PK check using indexes
469        if col_names.len() == 1 {
470            let col_name = &col_names[0];
471            
472            // To validate uniqueness for the WHOLE table (during add_primary_key), 
473            // we currently still need to check all values.
474            // But if we are validating INCOMING data, we check existing segments.
475            
476            // For now, we perform an optimized read of just the PK column.
477            let batches = self.read_with_columns(None, None, col_names.clone())
478                .map_err(|e| anyhow::anyhow!("Validation read failed: {}", e))?;
479            
480            let mut seen = std::collections::HashSet::new();
481            for batch in batches {
482                let col = batch.column_by_name(col_name)
483                    .ok_or_else(|| anyhow::anyhow!("Column '{}' not found in validation batch", col_name))?;
484                for i in 0..batch.num_rows() {
485                    let val = crate::core::manifest::ManifestValue::from_array(col, i).to_string();
486                    if !seen.insert(val) {
487                        return Err(anyhow::anyhow!("Primary key violation detected for column {}: Duplicate value found.", col_name));
488                    }
489                }
490            }
491            return Ok(());
492        }
493
494        // Fallback: Multi-column PK scan
495        let batches = self.read_with_columns(None, None, col_names.clone())
496            .map_err(|e| anyhow::anyhow!("Validation read failed: {}", e))?;
497        
498        let mut seen = std::collections::HashSet::new();
499        for batch in batches {
500            let sort_fields = batch.schema().fields().iter()
501                .map(|f| arrow::row::SortField::new(f.data_type().clone()))
502                .collect::<Vec<_>>();
503            let converter = arrow::row::RowConverter::new(sort_fields)
504                .map_err(|e| anyhow::anyhow!("RowConverter error: {}", e))?;
505            
506            let rows = converter.convert_columns(batch.columns())
507                .map_err(|e| anyhow::anyhow!("Row conversion error: {}", e))?;
508                
509            for row in rows.iter() {
510                if !seen.insert(row.as_ref().to_vec()) {
511                    return Err(anyhow::anyhow!("Primary key violation detected for columns {:?}: Duplicate row values found.", col_names));
512                }
513            }
514        }
515        
516        Ok(())
517    }

519    /// Check if a single primary key value exists in the committed storage.
520    /// Uses index-first searching (Bloom Filter -> Inverted Index -> Data Scan).
521    async fn _check_pk_in_storage_async(&self, column: &str, value: &serde_json::Value) -> Result<bool> {
522        let manifest = self.manifest().await?;
523        let manager = ManifestManager::new(self.store.clone(), "", &self.uri);
524        let all_entries = manager.load_all_entries(&manifest).await?;
525        
526        use futures::stream::{self, StreamExt};
527        
528        // Parallelize segment checks with a concurrency limit
529        let concurrency = 8; // Adjust based on hardware
530        let result_stream = stream::iter(all_entries.into_iter().rev())
531            .map(|entry| {
532                let store = self.store.clone();
533                let uri = self.uri.clone();
534                let config = self.get_config();
535                let schema = manifest.schemas.last().cloned().unwrap_or_default();
536                let column = column.to_string();
537                let value = value.clone();
538                let entry_path = entry.file_path.clone();
539                let entry_size = entry.file_size_bytes as u64;
540
541                async move {
542                    let mut reader = HybridReader::new(config, store, &uri)
543                        .with_iceberg_schema(schema);
544                    
545                    reader.config.parquet_path = Some(entry_path);
546                    reader.config.file_size = Some(entry_size);
547                    
548                    reader.check_value_exists(&column, &value).await
549                }
550            })
551            .buffer_unordered(concurrency)
552            .filter_map(|res| async {
553                match res {
554                    Ok(true) => Some(Ok(true)),
555                    Ok(false) => None,
556                    Err(e) => Some(Err(e)),
557                }
558            });
559            
560        let mut pinned_stream = Box::pin(result_stream);
561        let result = pinned_stream.next().await;
562
563        match result {
564            Some(res) => res, // Found a match or error
565            None => Ok(false), // No matches found in any segment
566        }
567    }

569    /// Internal helper: Detect existing physical indexes and migrate them to logical Schema specifications.
570    /// This provides zero-touch backward compatibility for tables created before the IndexSpec refactor.
571    async fn infer_index_metadata_from_physical_async(&self) -> Result<()> {
572        let manifest = self.manifest().await?;
573        let latest_schema = manifest.schemas.last().ok_or_else(|| anyhow::anyhow!("No schema found"))?;
574        
575        // Check if any field already has logical index metadata
576        if latest_schema.fields.iter().any(|f| !f.indexes.is_empty()) {
577            return Ok(());
578        }
579
580        // Scan latest manifest entries for physical index files
581        let mut inferred_specs: HashMap<String, Vec<IndexAlgorithm>> = HashMap::new();
582        
583        for entry in &manifest.entries {
584            for index_file in &entry.index_files {
585                if let Some(col_name) = &index_file.column_name {
586                    let algorithms = inferred_specs.entry(col_name.clone()).or_insert_with(Vec::new);
587                    
588                    let alg = match index_file.index_type.as_str() {
589                        "vector" | "hnsw" => Some(IndexAlgorithm::Hnsw {
590                             metric: "l2".to_string(), // Default for v0.2.x
591                             complexity: 16,
592                             quality: 128,
593                             build_device: None,
594                             search_device: None,
595                        }),
596                        "scalar" | "inverted" => Some(IndexAlgorithm::Bitmap),
597                        "bloom" => Some(IndexAlgorithm::Bloom { fpr: 0.05 }),
598                        _ => None,
599                    };
600
601                    if let Some(a) = alg {
602                        if !algorithms.contains(&a) {
603                            algorithms.push(a);
604                        }
605                    }
606                }
607            }
608        }
609
610        if !inferred_specs.is_empty() {
611            tracing::info!("Implicit Inference: Detected legacy physical indexes for columns {:?}. Updating manifest...", inferred_specs.keys());
612            self.set_index_columns(inferred_specs).await?;
613        }
614
615        Ok(())
616    }

710    /// Register an existing Iceberg table for Layered Indexing
711    pub async fn register_external(uri: String, iceberg_metadata_uri: &str) -> Result<Self> {
712        create_object_store(&uri)?;
713        
714        // Determine if iceberg_metadata_uri is a directory (table location) or a file (metadata file)
715        // If it's a directory, we need to find the metadata file inside it
716        let (meta_store_uri, filename) = {
717            let meta_url = url::Url::parse(iceberg_metadata_uri).context("Invalid iceberg_metadata_uri")?;
718            let meta_path_str = meta_url.path();
719            let meta_path = std::path::Path::new(meta_path_str);
720            
721            // Check if the path looks like a metadata file (ends with .json and contains "metadata")
722            let path_str = meta_path_str.to_string();
723            if path_str.ends_with(".metadata.json") || path_str.ends_with(".json") && path_str.contains("metadata") {
724                // It's a metadata file path
725                let parent_dir = meta_path.parent().context("No parent directory for metadata file")?;
726                let fname = meta_path.file_name().context("No filename for metadata file")?.to_str().unwrap().to_string();
727                
728                let store_uri = if iceberg_metadata_uri.starts_with("file://") {
729                    format!("file://{}", parent_dir.display())
730                } else {
731                    let mut base_url = meta_url.clone();
732                    base_url.set_path(parent_dir.to_str().unwrap());
733                    base_url.to_string()
734                };
735                (store_uri, fname)
736            } else {
737                // It's a table directory, look for metadata file inside metadata/ subdirectory
738                // Try to find the latest metadata file
739                let table_dir = meta_path_str;
740                let metadata_dir = if iceberg_metadata_uri.starts_with("file://") {
741                    format!("file://{}/metadata", table_dir)
742                } else {
743                    format!("{}/metadata", iceberg_metadata_uri.trim_end_matches('/'))
744                };
745                
746                tracing::debug!("Table location appears to be a directory. Looking for metadata in: {}", metadata_dir);
747                
748                // Use v1.metadata.json as default
749                (metadata_dir, "v1.metadata.json".to_string())
750            }
751        };
752
753        let iceberg_meta_store = create_object_store(&meta_store_uri)?;
754        
755        tracing::info!("Linking external Iceberg table: {}", iceberg_metadata_uri);
756        
757        // 1. Load Iceberg Metadata
758        let path = object_store::path::Path::from(filename.as_str());
759        let ret = iceberg_meta_store.get(&path).await?;
760        let bytes = ret.bytes().await?;
761        let iceberg_meta: crate::core::iceberg::IcebergTableMetadata = serde_json::from_slice(&bytes)?;
762        
763        // 2. Map Iceberg Schema to HyperStreamDB Schema
764        // For simplicity, we use the current schema
765        let current_schema_json = iceberg_meta.schemas.iter()
766            .find(|s| s.get("schema-id").and_then(|v| v.as_i64()) == Some(iceberg_meta.current_schema_id as i64))
767            .unwrap_or_else(|| {
768                tracing::warn!("Could not find schema with ID {}, falling back to first schema", iceberg_meta.current_schema_id);
769                &iceberg_meta.schemas[0]
770            });
771        let hdb_schema: crate::core::manifest::Schema = serde_json::from_value(current_schema_json.clone())?;
772        let schema_ref: SchemaRef = Arc::new(hdb_schema.to_arrow());
773
774        // 3. Initialize HyperStream Table
775        // This creates the directory and the initial manifest v0
776        let mut table = Self::create_async(uri.clone(), schema_ref.clone()).await?;
777        
778        // Root data_store carefully: if file, root at / to support absolute paths in manifests.
779        // If s3, root at bucket level.
780        let data_store_uri = if iceberg_meta.location.starts_with("file://") {
781            "file:///".to_string()
782        } else if iceberg_meta.location.starts_with("s3://") {
783             let url = url::Url::parse(&iceberg_meta.location)?;
784             format!("s3://{}/", url.host_str().unwrap_or(""))
785        } else {
786             iceberg_meta.location.clone()
787        };
788        table.data_store = Some(create_object_store(&data_store_uri)?);
789        
790        // 4. Import Snapshot (Incremental Indexing Trigger)
791        if let Some(snapshot_id) = iceberg_meta.current_snapshot_id {
792            table.import_iceberg_snapshot(snapshot_id, &iceberg_meta, iceberg_meta_store).await?;
793        }
794
795        Ok(table)
796    }

798    /// Detect if a URI is an Iceberg REST Catalog endpoint
799    fn detect_iceberg_rest(uri: &str) -> Option<(String, Option<String>, String, String)> {
800        if !uri.starts_with("http") { return None; }
801        
802        let ns_marker = "/namespaces/";
803        let t_marker = "/tables/";
804        
805        let ns_idx = uri.find(ns_marker)?;
806        let t_idx = uri.find(t_marker)?;
807        
808        if ns_idx >= t_idx { return None; }
809        
810        // Find /v1/ which precedes namespaces
811        let v1_idx = uri.find("/v1")?;
812        if v1_idx >= ns_idx { return None; }
813        
814        let base_url = uri[..v1_idx].to_string();
815        
816        // Extract optional prefix (e.g., /v1/warehouse/namespaces/...)
817        let prefix_start = v1_idx + 3;
818        let prefix_end = ns_idx;
819        let prefix = if prefix_end > prefix_start {
820            let p = &uri[prefix_start..prefix_end];
821            let trimmed = p.trim_matches('/');
822            if trimmed.is_empty() { None } else { Some(trimmed.to_string()) }
823        } else {
824            None
825        };
826        
827        let namespace = uri[ns_idx + ns_marker.len() .. t_idx].to_string();
828        let table_name = uri[t_idx + t_marker.len()..].to_string();
829        
830        Some((base_url, prefix, namespace, table_name))
831    }

833    async fn new_from_rest(base_url: String, prefix: Option<String>, namespace: String, table_name: String, rest_uri: &str) -> Result<Self> {
834        use crate::core::catalog::rest::RestCatalogClient;
835        use crate::core::catalog::Catalog;
836        
837        let client = RestCatalogClient::new(base_url, prefix);
838        let metadata = client.load_table(&namespace, &table_name).await?;
839        
840        // Derive local native URI (Cache location for layered index)
841        let cache_dir = std::env::var("HYPERSTREAM_CACHE_DIR")
842            .map(std::path::PathBuf::from)
843            .unwrap_or_else(|_| std::env::temp_dir().join("hyperstream_cache"));
844        
845        // Ensure directory exists
846        if !cache_dir.exists() {
847             let _ = std::fs::create_dir_all(&cache_dir);
848        }
849        
850        let safe_name = rest_uri.replace("://", "_").replace("/", "_").replace(":", "_");
851        let native_uri_path = cache_dir.join(safe_name);
852        let native_uri = format!("file://{}", native_uri_path.display());
853        
854        let store = create_object_store(&native_uri)?;
855        
856        // Check if already registered
857        let manager = ManifestManager::new(store.clone(), "", &native_uri);
858        let (_, version) = manager.load_latest().await.unwrap_or((Manifest::default(), 0));
859        
860        if version > 0 {
861            let mut table = TableBuilder::new(native_uri).with_index_all(false).build_async().await?;
862            table.catalog_state.catalog = Some(Arc::new(client));
863            table.catalog_state.namespace = Some(namespace);
864            table.catalog_state.table_name = Some(table_name);
865            Ok(table)
866        } else {
867            // Try to open the warehouse location as a HyperStreamDB native table first
868            // (REST-created tables are stored as HyperStreamDB tables in the warehouse)
869            tracing::debug!("Checking if warehouse location is a HyperStreamDB table: {}", metadata.location);
870            
871            match create_object_store(&metadata.location) {
872                Ok(warehouse_store) => {
873                    let warehouse_manager = ManifestManager::new(
874                        warehouse_store,
875                        "",
876                        &metadata.location
877                    );
878                    
879                    if let Ok((_, warehouse_version)) = warehouse_manager.load_latest().await {
880                        if warehouse_version > 0 {
881                            // It's a HyperStreamDB table in the warehouse, open it directly
882                            tracing::info!("✅ Opening existing HyperStreamDB table from REST catalog: {}", metadata.location);
883                            let mut table = TableBuilder::new(metadata.location).with_index_all(false).build_async().await?;
884                            table.catalog_state.catalog = Some(Arc::new(client));
885                            table.catalog_state.namespace = Some(namespace);
886                            table.catalog_state.table_name = Some(table_name);
887                            return Ok(table);
888                        } else {
889                            // Version=0, but this could be an empty newly-created table
890                            // Create a new native table at the warehouse location using the metadata schema
891                            tracing::info!("✅ Creating new HyperStreamDB table at warehouse location: {}", metadata.location);
892                            let current_schema = metadata.schemas.iter()
893                                .find(|s| s.schema_id == metadata.current_schema_id)
894                                .or_else(|| metadata.schemas.last())
895                                .ok_or_else(|| anyhow::anyhow!("No schema found in table metadata"))?;
896                            let schema_ref = Arc::new(current_schema.to_arrow());
897                            let table = Self::create_async(metadata.location.clone(), schema_ref).await?;
898                            // Don't set catalog fields - this is a newly-created empty table and committing back
899                            // to the catalog would fail. On next load, we'll detect the existing manifest and open normally.
900                            return Ok(table);
901                        }
902                    } else {
903                        tracing::debug!("ℹ️  Warehouse location has no manifest, trying external Iceberg import or creating new table");
904                    }
905                },
906                Err(e) => {
907                    tracing::debug!("ℹ️  Could not access warehouse location: {}. Trying external Iceberg import.", e);
908                }
909            }
910            
911            // If not a HyperStreamDB table, try to import as external Iceberg table
912            tracing::info!("Auto-registering Iceberg table from REST catalog: {}", rest_uri);
913            let mut table = Self::register_external(native_uri, &metadata.location).await?;
914
915            table.catalog_state.catalog = Some(Arc::new(client));
916            table.catalog_state.namespace = Some(namespace);
917            table.catalog_state.table_name = Some(table_name);
918            Ok(table)
919        }
920    }

922    /// Start a background observer to watch an external Iceberg table for changes
923    pub async fn spawn_iceberg_observer(&self, iceberg_metadata_uri: String, interval: std::time::Duration) -> Result<()> {
924        let table = self.clone();
925        
926        let handle = tokio::spawn(async move {
927            let mut last_processed_snapshot = None;
928            
929            // Try to set initial state from current manifest
930            if let Ok((_, _)) = table.get_snapshot_segments_with_version().await {
931                 // For now, we don't store the external snapshot id in our manifest explicitly,
932                 // but we could. For simplicity, we just start fresh or process what's new.
933            }
934
935            loop {
936                match table.check_and_import_new_snapshot(&iceberg_metadata_uri, &mut last_processed_snapshot).await {
937                    Ok(true) => tracing::debug!("Snapshot Observer: New snapshot processed."),
938                    Ok(false) => {}, // No new snapshot
939                    Err(e) => tracing::error!("Snapshot Observer Error: {}", e),
940                }
941                tokio::time::sleep(interval).await;
942            }
943        });
944        
945        self.background_tasks.lock().await.push(handle);
946        Ok(())
947    }

950        &self, 
951        iceberg_metadata_uri: &str,
952        last_snapshot_id: &mut Option<i64>
953    ) -> Result<bool> {
954        let meta_url = url::Url::parse(iceberg_metadata_uri).context("Invalid iceberg_metadata_uri")?;
955        let meta_path_str = meta_url.path();
956        let meta_path_obj = std::path::Path::new(meta_path_str);

959        
960        let meta_store_uri = if iceberg_metadata_uri.starts_with("file://") {
961            format!("file://{}", parent_dir.display())
962        } else {
963            let mut base_url = meta_url.clone();
964            base_url.set_path(parent_dir.to_str().unwrap());
965            base_url.to_string()
966        };

968        let iceberg_meta_store = create_object_store(&meta_store_uri)?;
969        let path = object_store::path::Path::from(filename);
970        let ret = iceberg_meta_store.get(&path).await?;
971        let bytes = ret.bytes().await?;
972        let iceberg_meta: crate::core::iceberg::IcebergTableMetadata = serde_json::from_slice(&bytes)?;
973        
974        if let Some(current_id) = iceberg_meta.current_snapshot_id {
975            if Some(current_id) != *last_snapshot_id {
976                self.import_iceberg_snapshot(current_id, &iceberg_meta, iceberg_meta_store).await?;
977                *last_snapshot_id = Some(current_id);
978                return Ok(true);
979            }
980        }
981        
982        Ok(false)
983    }

985    async fn import_iceberg_snapshot(
986        &self, 
987        snapshot_id: i64, 
988        meta: &crate::core::iceberg::IcebergTableMetadata,
989        iceberg_store: Arc<dyn ObjectStore>
990    ) -> Result<()> {
991        let snapshot = meta.snapshots.iter().find(|s| s.snapshot_id == snapshot_id)
992            .context("Snapshot not found")?;
993            
994        tracing::info!("Importing Iceberg Snapshot {}...", snapshot_id);
995        
996        // Load Manifest List
997        let ml_path_str = snapshot.manifest_list.trim_start_matches("file://");
998        let ml_path = object_store::path::Path::from(ml_path_str);
999        // If ml_path_str is absolute and store is relative, we might need more effort.
1000        // But for local files, object_store handles absolute paths if rooted correctly.
1001        // However, a better way is to make it relative to the store.
1002        
1003        let ret = iceberg_store.get(&ml_path).await?;
1004        let bytes = ret.bytes().await?;
1005        let manifest_list = crate::core::iceberg::read_manifest_list(&bytes[..])?;
1006        
1007        let mut all_entries = Vec::new();
1008        
1009        for ml_entry in manifest_list {
1010            // Load Manifest
1011            let m_path_str = ml_entry.manifest_path.trim_start_matches("file://");
1012            let m_path = object_store::path::Path::from(m_path_str);
1013            let ret = iceberg_store.get(&m_path).await?;
1014            let bytes = ret.bytes().await?;
1015            let manifest = crate::core::iceberg::read_manifest(&bytes[..])?;
1016            
1017            let iceberg_schema = crate::core::manifest::Schema {
1018                schema_id: meta.current_schema_id,
1019                fields: meta.schemas.iter()
1020                    .find(|s| s["schema-id"].as_i64().map(|id| id as i32) == Some(meta.current_schema_id))
1021                    .map(|s| {
1022                        s["fields"].as_array().unwrap_or(&Vec::new()).iter().map(|f| {
1023                            crate::core::manifest::SchemaField {
1024                                id: f["id"].as_i64().unwrap_or(0) as i32,
1025                                name: f["name"].as_str().unwrap_or("").to_string(),
1026                                type_str: f["type"].clone().to_string().replace("\"", ""),
1027                                 required: f["required"].as_bool().unwrap_or(false),
1028                                 fields: Vec::new(),
1029                                 initial_default: None,
1030                                 write_default: None,
1031                                 indexes: Vec::new(),
1032                             }
1033                        }).collect()
1034                    }).unwrap_or_else(Vec::new),
1035                identifier_field_ids: Vec::new(),
1036            };
1037
1038            let iceberg_spec = meta.partition_specs.iter()
1039                .find(|s| s["spec-id"].as_i64().map(|id| id as i32) == Some(meta.default_spec_id))
1040                .and_then(|s| crate::core::iceberg::iceberg_partition_spec_to_hyperstream(s).ok())
1041                .unwrap_or_default();
1042
1043            let mut data_entries = Vec::new();
1044            let mut delete_files = Vec::new();
1045
1046            for entry in manifest {
1047                 if entry.status != 2 { // Not deleted
1048                     match crate::core::iceberg::convert_iceberg_to_object(&entry, &iceberg_schema, &iceberg_spec) {
1049                         Ok(crate::core::iceberg::IcebergManifestObject::Data(me)) => {
1050                             data_entries.push(me);
1051                         }
1052                         Ok(crate::core::iceberg::IcebergManifestObject::Delete(df)) => {
1053                             delete_files.push(df);
1054                         }
1055                         Err(e) => tracing::warn!("Error converting Iceberg entry: {}", e),
1056                     }
1057                 }
1058            }
1059
1060            // Associate deletes with data entries (simplification: same partition)
1061            for df in delete_files {
1062                for data in &mut data_entries {
1063                    if data.partition_values == df.partition_values {
1064                        data.delete_files.push(df.clone());
1065                    }
1066                }
1067            }
1068            all_entries.extend(data_entries);
1069        }
1070        
1071        // Commit as new manifest version in HyperStream
1072        let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
1073        manifest_manager.commit_imported_entries(all_entries).await?;
1074        
1075        Ok(())
1076    }

1089
1090        // Check if table already exists (has a manifest)
1091        let (_, version) = manifest_manager.load_latest().await?;
1092        if version > 0 {
1093            return Err(anyhow::anyhow!("Table already exists at {}", uri));
1094        }
1095

1106            2,

1110            PartitionSpec::default(), 
1111            SortOrder::default()
1112        );

1114
1115        // Return a table instance pointing to the now-initialized location
1116        TableBuilder::new(uri).with_index_all(false).build_async().await
1117    }

1212    /// Replace the table's sort order (Iceberg V2/V3 evolution)
1213    /// 
1214    /// # Arguments
1215    /// * `columns` - Column names to sort by
1216    /// * `ascending` - Whether each column should be sorted in ascending order
1217    /// 
1218    /// # Example
1219    /// ```no_run
1220    /// # use hyperstreamdb::core::table::Table;
1221    /// # fn example(table: &Table) -> anyhow::Result<()> {
1222    /// table.replace_sort_order(&["timestamp", "user_id"], &[false, true])?;
1223    /// # Ok(())
1224    /// # }
1225    /// ```
1226    pub fn replace_sort_order(&self, columns: &[&str], ascending: &[bool]) -> Result<()> {
1227        if columns.len() != ascending.len() {
1228            anyhow::bail!("columns and ascending arrays must have same length");
1229        }
1230        
1231        let fields: Vec<SortField> = columns.iter().zip(ascending.iter())
1232            .enumerate()
1233            .map(|(i, (_col, asc))| SortField {
1234                source_id: i as i32 + 1, // Field IDs typically start at 1
1235                transform: "identity".to_string(),
1236                direction: if *asc { SortDirection::Asc } else { SortDirection::Desc },
1237                null_order: if *asc { NullOrder::NullsFirst } else { NullOrder::NullsLast },
1238            })
1239            .collect();
1240        
1241        // Store column names for lookup during apply
1242        let order = SortOrder {
1243            order_id: 1,
1244            fields,
1245        };
1246        
1247        let mut guard = self.sort_order.write().unwrap();
1248        *guard = Some(order);
1249        
1250        // Also store column names for apply_sort_order
1251        self.sort_order_columns.write().unwrap().replace(columns.iter().map(|s| s.to_string()).collect());
1252        
1253        Ok(())
1254    }

1261    /// Apply sort order to a RecordBatch before writing
1262    fn apply_sort_order(&self, batch: &RecordBatch) -> Result<RecordBatch> {
1263        let guard = self.sort_order.read().unwrap();
1264        let order = match guard.as_ref() {
1265            Some(o) if !o.fields.is_empty() => o,
1266            _ => return Ok(batch.clone()), // No sort order configured
1267        };
1268        
1269        let columns_guard = self.sort_order_columns.read().unwrap();
1270        let column_names = match columns_guard.as_ref() {
1271            Some(names) => names,
1272            None => return Ok(batch.clone()),
1273        };
1274        
1275        // Build sort columns
1276        let mut sort_columns = Vec::new();
1277        for (field, col_name) in order.fields.iter().zip(column_names.iter()) {
1278            if let Some((idx, _)) = batch.schema().column_with_name(col_name) {
1279                let column = batch.column(idx).clone();
1280                let options = arrow::compute::SortOptions {
1281                    descending: matches!(field.direction, SortDirection::Desc),
1282                    nulls_first: matches!(field.null_order, NullOrder::NullsFirst),
1283                };
1284                sort_columns.push(arrow::compute::SortColumn { values: column, options: Some(options) });
1285            }
1286        }
1287        
1288        if sort_columns.is_empty() {
1289            return Ok(batch.clone());
1290        }
1291        
1292        // Compute sorted indices
1293        let indices = arrow::compute::lexsort_to_indices(&sort_columns, None)?;
1294        
1295        // Apply sort to all columns
1296        let sorted_columns: Vec<Arc<dyn arrow::array::Array>> = batch.columns()
1297            .iter()
1298            .map(|col| arrow::compute::take(col.as_ref(), &indices, None))
1299            .collect::<std::result::Result<Vec<_>, _>>()?;
1300        
1301        Ok(RecordBatch::try_new(batch.schema(), sorted_columns)?)
1302    }

1349                    return Arc::new(latest.to_arrow());
1350                } else if let Some(latest) = manifest.schemas.last() {
1351                    return Arc::new(latest.to_arrow());
1352                }
1353            }
1354            
1355            // If manifest exists but entries exist without schemas (Legacy)
1356            if !manifest.entries.is_empty() {

1359                    let file_path = &entry.file_path;

1366                        use futures::StreamExt;

1368                            return batch.schema();
1369                        }
1370                    }

1504    async fn backfill_indexes_async(&self, target_columns: Vec<String>) -> Result<()> {
1505        use futures::StreamExt;
1506        let manager = ManifestManager::new(self.store.clone(), "", &self.uri);
1507        let (_manifest, all_entries, _) = manager.load_latest_full().await?;
1508        
1509        if all_entries.is_empty() {
1510            return Ok(());
1511        }
1512
1513        let entries_results: Vec<Result<ManifestEntry>> = futures::future::join_all(all_entries.iter().map(|entry| {
1514            let entry = entry.clone();
1515            let table_uri = self.uri.clone();
1516            let store = self.store.clone();
1517            let data_store = self.data_store.clone().unwrap_or(self.store.clone());
1518            let target_cols = target_columns.clone();
1519            
1520            async move {
1521                let mut current_entry = entry.clone();
1522                let file_path_str = current_entry.file_path.clone();
1523                let segment_id = file_path_str.split('/').next_back().unwrap_or(&file_path_str)
1524                    .strip_suffix(".parquet").unwrap_or(&file_path_str);
1525
1526                let mut cols_to_index = self.indexing.index_columns.read().unwrap().clone();
1527                for col in target_cols {
1528                    if !cols_to_index.contains(&col) {
1529                        cols_to_index.push(col);
1530                    }
1531                }
1532
1533                let config = SegmentConfig::new(&table_uri, segment_id)
1534                    .with_parquet_path(current_entry.file_path.clone())
1535                    .with_data_store(data_store)
1536                    .with_index_all(self.indexing.index_all)
1537                    .with_columns_to_index(cols_to_index);
1538                
1539                let reader = HybridReader::new(config.clone(), store.clone(), &table_uri);
1540                let mut writer = HybridSegmentWriter::new(config)
1541                    .with_index_configs(self.indexing.index_configs.read().unwrap().clone())
1542                    .with_record_count(current_entry.record_count as usize)
1543                    .with_existing_stats(current_entry.column_stats.clone());
1544                writer.primary_key = self.primary_key.read().unwrap().clone();
1545                writer.set_store(store.clone());
1546                
1547                let mut stream = reader.stream_all(None as Option<arrow::datatypes::SchemaRef>).await?;
1548                let mut current_offset = 0;
1549                while let Some(batch) = stream.next().await {
1550                    let batch = batch?;
1551                    let batch_rows = batch.num_rows();
1552                    writer.build_indexes(&batch, current_offset)?;
1553                    current_offset += batch_rows;
1554                }
1555                
1556                writer.finish_indexing().await?;
1557                writer.upload_to_store().await?;
1558                
1559                // Invalidate the cache for this segment
1560                let cache_key = format!("{}/{}", table_uri, current_entry.file_path);
1561                crate::core::cache::PARQUET_META_CACHE.invalidate(&cache_key).await;
1562                
1563                let updated_entry = writer.to_manifest_entry();
1564                current_entry.index_files = updated_entry.index_files;
1565                
1566                Ok(current_entry)
1567            }
1568        })).await;
1569
1570        let mut updated_entries = Vec::new();
1571        for res in entries_results {
1572            updated_entries.push(res?);
1573        }
1574
1575        if !updated_entries.is_empty() {
1576            manager.commit_imported_entries(updated_entries).await?;
1577        }
1578
1579        Ok(())
1580    }

1650    /// Async implementation of delete
1651    pub async fn delete_async(&self, filter: &str) -> Result<()> {
1652        use futures::StreamExt;
1653        
1654        let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
1655        let (_manifest, all_entries, _) = manifest_manager.load_latest_full().await?;
1656        
1657        if all_entries.is_empty() {
1658            return Ok(());
1659        }
1660
1661        let planner = QueryPlanner::new();
1662        let arrow_schema = self.arrow_schema();
1663        let expr = FilterExpr::parse_sql(filter, arrow_schema).await.context("Failed to parse delete filter")?;
1664        
1665        let candidates = planner.prune_entries(&all_entries, Some(&expr), None);
1666        let candidate_paths: std::collections::HashSet<String> = candidates.iter().map(|(e, _)| e.file_path.clone()).collect();
1667        tracing::debug!("delete_async: filter='{}', potential candidates: {}", filter, candidates.len());
1668        
1669        let mut all_updated_entries = Vec::new();
1670
1671        for entry in all_entries {
1672            if !candidate_paths.contains(&entry.file_path) {
1673                // Preserve non-candidate segments as-is
1674                all_updated_entries.push(entry);
1675                continue;
1676            }
1677
1678            tracing::debug!("delete_async: processing segment {}", entry.file_path);
1679            let file_path_str = entry.file_path.clone();
1680            
1681            // Fix path resolution: find correct physical subdirectory
1682            let path = std::path::Path::new(&file_path_str);
1683            let rel_parent = path.parent().and_then(|p| p.to_str()).unwrap_or("");
1684            let full_base_path = if rel_parent.is_empty() {
1685                self.uri.clone()
1686            } else {
1687                format!("{}/{}", self.uri, rel_parent)
1688            };
1689
1690            let segment_id = file_path_str.split('/').next_back().unwrap_or(&file_path_str)
1691                .strip_suffix(".parquet").unwrap_or(&file_path_str);
1692            
1693            let config = SegmentConfig::new(&full_base_path, segment_id)
1694                .with_index_files(entry.index_files.clone())
1695                .with_record_count(entry.record_count as u64);
1696            let reader = HybridReader::new(config, self.store.clone(), &self.uri);
1697            
1698            let mut new_deletes = Vec::new();
1699            
1700            // OPTIMIZATION: Check for Sidecar Scalar Index (Inverted Index)
1701            let and_filters = expr.extract_and_conditions();
1702            let mut bitmap_opt: Option<roaring::RoaringBitmap> = None;
1703
1704            if !and_filters.is_empty() {
1705                for filter in and_filters {
1706                    if let Ok(Some(bm)) = reader.get_scalar_filter_bitmap(&filter).await {
1707                        if let Some(current) = bitmap_opt {
1708                            bitmap_opt = Some(current & bm);
1709                        } else {
1710                            bitmap_opt = Some(bm);
1711                        }
1712                    } else {
1713                        // If any part of the AND is MISSING an index, we might need a scan
1714                        // or we just skip this optimization.
1715                        bitmap_opt = None;
1716                        break;
1717                    }
1718                }
1719            }
1720
1721            if let Some(bitmap) = bitmap_opt {
1722                // Index HIT! We found the deleted rows instantly.
1723                for row_id in bitmap.iter() {
1724                    new_deletes.push(row_id as i64);
1725                }
1726            } else {
1727                // Index MISS: Full file scan (fallback)
1728                let mut stream = reader.stream_all(None).await?;
1729                let mut current_row_offset = 0;
1730                while let Some(batch_res) = stream.next().await {
1731                    let batch = batch_res?;
1732                    let num_rows = batch.num_rows();
1733                    let mask = planner.evaluate_expr(&batch, &expr)?;
1734                    for i in 0..num_rows {
1735                        if mask.value(i) {
1736                            new_deletes.push((current_row_offset + i) as i64);
1737                        }
1738                    }
1739                    current_row_offset += num_rows;
1740                }
1741            }
1742            
1743            if !new_deletes.is_empty() {
1744                // Generate NEW Position Delete File
1745                let mut file_paths = arrow::array::StringBuilder::new();
1746                let mut positions = arrow::array::Int64Builder::new();
1747                
1748                for &pos in &new_deletes {
1749                    file_paths.append_value(&entry.file_path);
1750                    positions.append_value(pos);
1751                }
1752                
1753                let file_path_array = file_paths.finish();
1754                let pos_array = positions.finish();
1755                
1756                let delete_writer = crate::core::iceberg::iceberg_delete::IcebergDeleteWriter::new(
1757                    self.uri.clone(),
1758                    2 // Format V2
1759                );
1760
1761                let partition_data = if !entry.partition_values.is_empty() {
1762                    let path = std::path::Path::new(&entry.file_path);
1763                    let rel_path = path.parent()
1764                        .and_then(|p| p.to_str())
1765                        .unwrap_or("")
1766                        .trim_start_matches('/')
1767                        .to_string();
1768                        
1769                    Some((rel_path, entry.partition_values.clone()))
1770                } else {
1771                    None
1772                };
1773
1774                let delete_file = delete_writer.write_position_delete(
1775                    partition_data, 
1776                    &file_path_array,
1777                    &pos_array
1778                ).await?;
1779                
1780                let mut new_entry = entry.clone();
1781                new_entry.delete_files.push(delete_file);
1782                all_updated_entries.push(new_entry);
1783            } else {
1784                all_updated_entries.push(entry.clone());
1785            }
1786        }
1787
1788        if !all_updated_entries.is_empty() {
1789            // Commit the entire updated state. 
1790            // In HyperStreamDB manifest management, this acts as an atomic swap of the segment list.
1791            manifest_manager.commit(&all_updated_entries, &[], crate::core::manifest::CommitMetadata::default()).await?;
1792        }
1793
1794        Ok(())
1795    }

1797    /// Check if any keys in the batch already exist in the table (Primary Key Enforcement)
1798    async fn check_primary_key_uniqueness_async(&self, batch: &RecordBatch, columns: &[String]) -> Result<()> {
1799        
1800        if batch.num_rows() == 0 { return Ok(()); }
1801        
1802        let schema = batch.schema();
1803        let col_indices: Vec<usize> = columns.iter()
1804            .map(|c| schema.index_of(c))
1805            .collect::<Result<Vec<usize>, _>>()?;
1806
1807        // OPTIMIZATION: Use IN clause for batches (efficient via Inverted Index)
1808        // For now, we take the first row as a sample check to avoid huge expression generation 
1809        // until we have a proper Row-Value In-List implementation.
1810        for i in 0..batch.num_rows().min(100) { // Limit samples for performance in MVP
1811            let mut filters_str_vec = Vec::new();
1812            for (col_name, col_idx) in columns.iter().zip(col_indices.iter()) {
1813                let col = batch.column(*col_idx);
1814                let val = if let Some(arr) = col.as_any().downcast_ref::<arrow::array::Int32Array>() {
1815                    format!("{}", arr.value(i))
1816                } else if let Some(arr) = col.as_any().downcast_ref::<arrow::array::Int64Array>() {
1817                    format!("{}", arr.value(i))
1818                } else if let Some(arr) = col.as_any().downcast_ref::<arrow::array::StringArray>() {
1819                    format!("'{}'", arr.value(i).replace("'", "''"))
1820                } else {
1821                    continue;
1822                };
1823                filters_str_vec.push(format!("{} = {}", col_name, val));
1824            }
1825            
1826            if !filters_str_vec.is_empty() {
1827                let filter_str = filters_str_vec.join(" AND ");
1828                let expr = FilterExpr::parse_sql(&filter_str, self.arrow_schema()).await?;
1829                
1830                // Check manifests
1831                let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
1832                let (_, all_entries, _) = manifest_manager.load_latest_full().await?;
1833                let planner = QueryPlanner::new();
1834                let candidates = planner.prune_entries(&all_entries, Some(&expr), None);
1835                
1836                if !candidates.is_empty() {
1837                    // Refine search within candidates (Index lookup)
1838                    for (entry, _) in candidates {
1839                        // Resolve partition-aware path for PK lookup
1840                        let path = std::path::Path::new(&entry.file_path);
1841                        let rel_parent = path.parent().and_then(|p| p.to_str()).unwrap_or("");
1842                        let full_base_path = if rel_parent.is_empty() {
1843                             self.uri.clone()
1844                        } else {
1845                             format!("{}/{}", self.uri, rel_parent)
1846                        };
1847                        
1848                        let seg_id = entry.file_path.split('/').next_back().unwrap_or(&entry.file_path)
1849                            .replace(".parquet", "");
1850
1851                        let config = SegmentConfig::new(&full_base_path, &seg_id)
1852                            .with_index_files(entry.index_files.clone())
1853                            .with_delete_files(entry.delete_files.clone());
1854                        let reader = HybridReader::new(config, self.store.clone(), &self.uri);
1855                        
1856                        let filters = expr.extract_and_conditions();
1857                        let mut bitmap_opt: Option<roaring::RoaringBitmap> = None;
1858                        
1859                        for f in filters {
1860                            if let Ok(Some(bm)) = reader.get_scalar_filter_bitmap(&f).await {
1861                                // Subtract logically deleted rows!
1862                                let deleted = reader.load_merged_deletes().await?;
1863                                let alive_bm = bm.clone() - deleted.clone();
1864                                
1865                                tracing::debug!("PK Check for {}: Index bits: {}, Deleted bits: {}, Alive bits: {}", 
1866                                    f.column, bm.len(), deleted.len(), alive_bm.len());
1867                                
1868                                if let Some(current) = bitmap_opt {
1869                                    bitmap_opt = Some(current & alive_bm);
1870                                } else {
1871                                    bitmap_opt = Some(alive_bm);
1872                                }
1873                            } else {
1874                                bitmap_opt = None;
1875                                break;
1876                            }
1877                        }
1878
1879                        if let Some(bm) = bitmap_opt {
1880                            if !bm.is_empty() {
1881                                let pk_val = columns.iter().zip(filters_str_vec.iter())
1882                                    .map(|(c, f)| format!("{}={}", c, f))
1883                                    .collect::<Vec<_>>().join(", ");
1884                                return Err(anyhow::anyhow!("Duplicate primary key error: {} already exists", pk_val));
1885                            }
1886                        }
1887                    }
1888                }
1889            }
1890        }
1891        
1892        Ok(())
1893    }

1909    fn merge_on_read(&self, batches: Vec<RecordBatch>, key_column: &str) -> Result<()> {
1910        let key_cols: Vec<&str> = key_column.split(',').collect();
1911        self.runtime().block_on(async {
1912            for batch in &batches {
1913                let schema = batch.schema();
1914                let col_indices: Vec<usize> = key_cols.iter()
1915                    .map(|&c| schema.index_of(c))
1916                    .collect::<Result<Vec<usize>, _>>()?;
1917                
1918                for i in 0..batch.num_rows() {
1919                    let mut filters = Vec::new();
1920                    for (&col_name, &col_idx) in key_cols.iter().zip(col_indices.iter()) {
1921                        let col = batch.column(col_idx);
1922                        let val = if let Some(arr) = col.as_any().downcast_ref::<arrow::array::Int32Array>() {
1923                            format!("{}", arr.value(i))
1924                        } else if let Some(arr) = col.as_any().downcast_ref::<arrow::array::Int64Array>() {
1925                            format!("{}", arr.value(i))
1926                        } else if let Some(arr) = col.as_any().downcast_ref::<arrow::array::StringArray>() {
1927                            format!("'{}'", arr.value(i))
1928                        } else {
1929                            // Fallback for other types
1930                            "".to_string()
1931                        };
1932                        
1933                        if !val.is_empty() {
1934                            filters.push(format!("{} = {}", col_name, val));
1935                        }
1936                    }
1937                    
1938                    if !filters.is_empty() {
1939                        let filter_expr = filters.join(" AND ");
1940                        self.delete_async(&filter_expr).await?;
1941                    }
1942                }
1943            }
1944            
1945            // Step 2: Write the new data (Append)
1946            self.write_async(batches).await?;
1947            Ok(())
1948        })
1949    }

1951    fn merge_on_write(&self, batches: Vec<RecordBatch>, key_column: &str) -> Result<()> {
1952        // MoW uses MergePlanner to rewrite segments
1953        use crate::core::merge::MergePlanner;
1954        
1955        if batches.is_empty() { return Ok(()); }
1956        
1957        let schema = batches[0].schema();
1958        let source_batch = arrow::compute::concat_batches(&schema, &batches)?;
1959        
1960        // Extract keys as JSON values for MergePlanner
1961        let mut source_keys = Vec::new();
1962        let col_idx = schema.index_of(key_column)?;
1963        let col = source_batch.column(col_idx);
1964        
1965        if let Some(arr) = col.as_any().downcast_ref::<arrow::array::Int32Array>() {
1966            for i in 0..arr.len() { source_keys.push(Value::Number(arr.value(i).into())); }
1967        } else if let Some(arr) = col.as_any().downcast_ref::<arrow::array::Int64Array>() {
1968            for i in 0..arr.len() { source_keys.push(Value::Number(arr.value(i).into())); }
1969        } else {
1970            return Err(anyhow::anyhow!("MoW currently only supports integer keys"));
1971        }
1972
1973        self.runtime().block_on(async {
1974            
1975            let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
1976            let (_manifest, all_entries, _) = manifest_manager.load_latest_full().await?;
1977            
1978            let segment_ids: Vec<String> = all_entries.iter().map(|e| {
1979                e.file_path.split('/').next_back().unwrap().replace(".parquet", "")
1980            }).collect();
1981            
1982            let planner = MergePlanner::new();
1983            let commit_actions = planner.execute_merge(
1984                &self.uri,
1985                key_column,
1986                &source_keys,
1987                &source_batch,
1988                &segment_ids,
1989                |_, _| Ok(None) // index_provider
1990            )?;
1991            
1992            let mut new_entries = Vec::new();
1993            let mut removed_paths = Vec::new();
1994            
1995            // Note: MergePlanner returns Segment IDs. We need ManifestEntries.
1996            // This part is a bit tricky because MergePlanner is legacy.
1997            // For now, let's assume we can reconstruct basic entries.
1998            // A better way would be to have MergePlanner work with ManifestEntries.
1999            
2000            for (old_seg, new_seg) in commit_actions {
2001                if let Some(old) = old_seg {
2002                    removed_paths.push(format!("{}.parquet", old));
2003                }
2004                
2005                // Create a dummy entry for the new segment (will be updated by compactor later)
2006                // In a perfect system, writer.to_manifest_entry() should be used.
2007                let entry = ManifestEntry {
2008                    file_path: format!("{}.parquet", new_seg),
2009                    ..Default::default()
2010                };
2011                new_entries.push(entry);
2012            }
2013            
2014            manifest_manager.commit(&new_entries, &removed_paths, crate::core::manifest::CommitMetadata::default()).await?;
2015            Ok(())
2016        })
2017    }

2658    async fn shuffle_batch_by_centroids(&self, batch: &RecordBatch, col_name: &str) -> Result<RecordBatch> {
2659        use crate::core::index::ivf::simple_kmeans;
2660        use arrow::array::{Array, Int32Array};
2661        
2662        let col_idx = batch.schema().index_of(col_name)?;
2663        let list_array = batch.column(col_idx).as_any().downcast_ref::<arrow::array::FixedSizeListArray>().unwrap();
2664        
2665        let n = list_array.len();
2666        if n < 1024 { return Ok(batch.clone()); } // Too small to benefit from shuffling
2667        
2668        // 1. Convert vectors to Vec<Vec<f32>> for K-Means (Training step)
2669        // Optimization: Port this to handle FixedSizeList directly to avoid copies
2670        let vectors: Vec<Vec<f32>> = (0..n).into_par_iter().step_by(n / 1000 + 1).map(|i| {
2671            list_array.value(i).as_any().downcast_ref::<arrow::array::Float32Array>().unwrap().values().to_vec()
2672        }).collect();
2673
2674        // 2. Train centroids (Sampled)
2675        let k = (n as f64).sqrt() as usize;
2676        let k = k.clamp(16, 1024);
2677        let (centroids, _) = simple_kmeans(&vectors, k, 3)?; // Fast 3-iter training
2678
2679        // 3. Assign all vectors (GPU Accelerated!)
2680        let _ = get_global_gpu_context().unwrap_or_else(crate::core::index::gpu::ComputeContext::auto_detect);
2681
2682
2683        let dim = list_array.value_length() as usize;
2684        let flat_vectors: Vec<f32> = (0..n).into_par_iter().flat_map(|i| {
2685            list_array.value(i).as_any().downcast_ref::<arrow::array::Float32Array>().unwrap().values().to_vec()
2686        }).collect();
2687        
2688        let flat_centroids: Vec<f32> = centroids.iter().flatten().copied().collect();
2689        
2690        let assignments = crate::core::index::gpu::compute_kmeans_assignment(&flat_vectors, &flat_centroids, dim)?;
2691        
2692        // 4. Sort batch by assignments
2693        let assignment_array = Int32Array::from(assignments.into_iter().map(|a| a as i32).collect::<Vec<i32>>());
2694        let sort_indices = arrow::compute::sort_to_indices(&assignment_array, None, None)?;
2695        
2696        let mut columns = Vec::new();
2697        for i in 0..batch.num_columns() {
2698            columns.push(arrow::compute::take(batch.column(i), &sort_indices, None)?);
2699        }
2700
