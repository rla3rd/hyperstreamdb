3use anyhow::Result;
4use arrow::record_batch::RecordBatch;
5use crate::core::planner::VectorSearchParams;
6use crate::core::index::gpu::{set_global_gpu_context, ComputeContext};
7
8use super::Table;
9
10pub struct TableQuery<'a> {
11    pub table: &'a Table,
12    pub filter_str: Option<String>,
13    pub vector_filter: Option<VectorSearchParams>,
14    pub columns: Option<Vec<String>>,
15    pub context: Option<ComputeContext>,
16    pub rrf_k: Option<f32>,
17}
18
19impl<'a> TableQuery<'a> {
20    pub fn new(table: &'a Table) -> Self {
21        Self {
22            table,
23            filter_str: None,
24            vector_filter: None,
25            columns: None,
26            context: None,
27            rrf_k: None,
28        }
29    }
30
31    pub fn filter(mut self, expr: &str) -> Self {
32        if let Some(ref mut f) = self.filter_str {
33            *f = format!("({}) AND ({})", f, expr);
34        } else {
35            self.filter_str = Some(expr.to_string());
36        }
37        self
38    }
39
40    pub fn vector_search(mut self, column: &str, query: crate::core::index::VectorValue, k: usize) -> Self {
41        self.vector_filter = Some(VectorSearchParams::new(column, query, k));
42        self
43    }
44
45    pub fn select(mut self, columns: Vec<String>) -> Self {
46        self.columns = Some(columns);
47        self
48    }
49
50    pub fn with_context(mut self, context: ComputeContext) -> Self {
51        self.context = Some(context);
52        self
53    }
54
55    pub fn rrf_k(mut self, k: f32) -> Self {
56        self.rrf_k = Some(k);
57        self
58    }
59
60    pub async fn to_batches(self) -> Result<Vec<RecordBatch>> {
61        let cols_refs: Option<Vec<&str>> = self.columns.as_ref().map(|c| c.iter().map(|s| s.as_str()).collect());
62        let cols_slice: Option<&[&str]> = cols_refs.as_deref();
63        
64        // Inject context if provided
65        if let Some(ctx) = self.context {
66            set_global_gpu_context(Some(ctx));
67        }
68        
69        let mut config = self.table.query_config().clone();
70        if let Some(k) = self.rrf_k {
71             config = config.with_rrf_k(k);
72        }
73        
74        self.table.read_with_config_async(self.filter_str.as_deref(), self.vector_filter, cols_slice, config).await
75    }
76}
