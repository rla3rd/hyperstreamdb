33    pub async fn sql(&self, query: &str) -> Result<Vec<RecordBatch>> {
34        use datafusion::prelude::SessionContext;
35        use crate::core::sql::HyperStreamTableProvider;
36
37        let mut ctx = SessionContext::new();        let _ = crate::core::sql::vector_operators::register_vector_operators(&mut ctx);
38        let provider = Arc::new(HyperStreamTableProvider::new(Arc::new(self.clone())));
39        ctx.register_table("t", provider)?;
40        let df = ctx.sql(query).await?;
41        Ok(df.collect().await?)
42    }

52    /// Generate a detailed execution plan with hit counts and pruning stats
53    pub async fn explain(&self, filter_str: Option<&str>, vector_param: Option<VectorSearchParams>) -> String {
54        let manifest_manager = crate::core::manifest::ManifestManager::new(self.store.clone(), "", &self.uri);
55        let (_manifest, all_entries, version) = manifest_manager.load_latest_full().await.unwrap_or((crate::core::manifest::Manifest::default(), Vec::new(), 0));
56        let total_rows_table: usize = all_entries.iter().map(|e| e.record_count as usize).sum();
57        let total_segments_table = all_entries.len();
58        
59        let mut plan = Vec::new();
60        let divider = "-".repeat(60);
61        plan.push(divider.clone());
62        plan.push(format!("HYPERSTREAM QUERY PLAN [Table: {}]", self.uri));
63        
64        // 1. Initial Pruning (Partition & Stats)
65        let expr = if let Some(f) = filter_str {
66            FilterExpr::parse_sql(f, self.arrow_schema()).await.ok()
67        } else {
68            None
69        };
70
71        let planner = QueryPlanner::new();
72        let pruned_entries: Vec<(ManifestEntry, Option<IndexFile>)> = if version > 0 {
73            planner.prune_entries(&all_entries, expr.as_ref(), vector_param.as_ref())
74        } else {
75            all_entries.iter().map(|e| (e.clone(), None)).collect()
76        };
77
78        let scanned_segments = pruned_entries.len();
79        let scanned_rows: usize = pruned_entries.iter().map(|(e, _)| e.record_count as usize).sum();
80        
81        plan.push(format!("Context Selection:"));
82        plan.push(format!("  -> Total Table Scope: {} rows in {} segments", total_rows_table, total_segments_table));
83        if scanned_segments < total_segments_table {
84            plan.push(format!("  -> Pruning Activity: {} segments pruned via Partition/Stats mapping", total_segments_table - scanned_segments));
85        }
86        plan.push(format!("  -> Execution Scope: {} rows in {} segments", scanned_rows, scanned_segments));
87        plan.push("".to_string());
88
89        // 2. Index Dry-run of Filters
90        let mut scalar_hits = scanned_rows;
91        let mut access_paths = Vec::new();
92
93        if let Some(ref e) = expr {
94            let sub_filters = e.extract_and_conditions();
95            let mut total_hits = 0;
96            let base_uri = self.uri.clone();
97            
98            // Convert file:// URI to filesystem path for directory listing
99            let fs_path = if base_uri.starts_with("file://") {
100                base_uri.strip_prefix("file://").unwrap_or(&base_uri).to_string()
101            } else {
102                base_uri.clone()
103            };
104            
105            // Map to track which index type was used for each sub-filter
106            let mut filter_index_types: HashMap<String, HashSet<&'static str>> = HashMap::new();
107
108            for (entry, _) in &pruned_entries {
109                let file_path_str = entry.file_path.clone();
110                let segment_id = file_path_str
111                   .split('/')
112                   .next_back()
113                   .unwrap_or(&file_path_str)
114                   .strip_suffix(".parquet")
115                   .unwrap_or(&file_path_str);
116
117                let config = SegmentConfig::new(&base_uri, segment_id)
118                   .with_parquet_path(entry.file_path.clone())
119                   .with_index_files(entry.index_files.clone())
120                   .with_delete_files(entry.delete_files.clone())
121                   .with_record_count(entry.record_count as u64);
122
123                let reader = HybridReader::new(config, self.store.clone(), &base_uri);
124                
125                let mut seg_bm: Option<roaring::RoaringBitmap> = None;
126                for sub_f in &sub_filters {
127                    // Detect access path by checking for actual index files on disk
128                    let path = {
129                        // Check for inverted index (.inv.parquet files)
130                        let inv_pattern = format!("{}.{}.inv.parquet", segment_id, sub_f.column);
131                        let has_inverted = std::fs::read_dir(&fs_path)
132                            .ok()
133                            .and_then(|dir| {
134                                dir.flatten()
135                                    .find(|e| e.file_name().to_string_lossy().contains(&inv_pattern))
136                            })
137                            .is_some();
138                        
139                        if has_inverted {
140                            "Inverted Index (Parquet)"
141                        } else {
142                            // Check for bitmap index (.idx files)
143                            let bitmap_pattern = format!("{}.{}.idx", segment_id, sub_f.column);
144                            let has_bitmap = std::fs::read_dir(&fs_path)
145                                .ok()
146                                .and_then(|dir| {
147                                    dir.flatten()
148                                        .find(|e| e.file_name().to_string_lossy().contains(&bitmap_pattern))
149                                })
150                                .is_some();
151                            
152                            if has_bitmap {
153                                "Bitmap Index (.idx)"
154                            } else {
155                                "Full Scan"
156                            }
157                        }
158                    };
159                    filter_index_types.entry(sub_f.column.clone()).or_default().insert(path);
160
161                    if let Ok(Some(bm)) = reader.get_scalar_filter_bitmap(sub_f).await {
162                        match seg_bm {
163                            Some(ref mut existing) => *existing &= bm,
164                            None => seg_bm = Some(bm),
165                        }
166                    }
167                }
168
169                if let Some(bm) = seg_bm {
170                    total_hits += bm.len() as usize;
171                } else {
172                    total_hits += entry.record_count as usize;
173                }
174            }
175            scalar_hits = total_hits;
176            
177            for sub_f in &sub_filters {
178                let paths: Vec<&'static str> = filter_index_types.get(&sub_f.column)
179                    .map(|s: &HashSet<&'static str>| s.iter().cloned().collect())
180                    .unwrap_or_else(|| vec!["Scan"]);
181                access_paths.push(format!("  -> Filter (col: {}, op: {}, access: {})", sub_f.column, sub_f.op_to_string(), paths.join(", ")));
182            }
183        }
184
185        // 3. Scalar Plan (First: Pre-filter the dataset)
186        if !access_paths.is_empty() {
187            plan.push("Scalar Execution:".to_string());
188            for path in access_paths {
189                plan.push(path);
190            }
191            let pct = if scanned_rows > 0 { (scalar_hits as f32 / scanned_rows as f32) * 100.0 } else { 0.0 };
192            plan.push(format!("     [Selectivity: {} / {} rows ({:.2}%)]", scalar_hits, scanned_rows, pct));
193            plan.push("".to_string());
194        }
195
196        // 4. Vector Plan (Second: Vector search on pre-filtered rows)
197        if let Some(ref vs) = vector_param {
198            plan.push("Vector Execution:".to_string());
199            plan.push(format!("  -> VectorSearch (col: {}, k: {}, metric: {:?})", vs.column, vs.k, vs.metric));
200            
201            // Check for vector index by detecting .hnsw.graph files on disk
202            let mut has_vector_index = false;
203            
204            // Convert file:// URI to filesystem path
205            let fs_path = if self.uri.starts_with("file://") {
206                self.uri.strip_prefix("file://").unwrap_or(&self.uri)
207            } else {
208                &self.uri
209            };
210            
211            for (entry, _) in &pruned_entries {
212                let segment_id = entry.file_path
213                    .split('/')
214                    .next_back()
215                    .unwrap_or(&entry.file_path)
216                    .strip_suffix(".parquet")
217                    .unwrap_or(&entry.file_path);
218                
219                // Look for HNSW graph files for this column: segment_id.{column_name}.cluster_*.hnsw.graph
220                let hnsw_pattern_prefix = format!("{}.{}.cluster_", segment_id, vs.column);
221                let hnsw_pattern_suffix = ".hnsw.graph";
222                
223                // Try to list files in the table directory to detect index files
224                if let Ok(dirs) = std::fs::read_dir(fs_path) {
225                    for entry in dirs.flatten() {
226                        if let Some(filename) = entry.file_name().to_str() {
227                            if filename.starts_with(&hnsw_pattern_prefix) && filename.ends_with(hnsw_pattern_suffix) {
228                                has_vector_index = true;
229                                break;
230                            }
231                        }
232                    }
233                }
234                if has_vector_index {
235                    break;
236                }
237            }
238            
239            let access_mode = if has_vector_index { "HNSW-IVF Cluster Index" } else { "Brute Force Scan (No Index)" };
240            plan.push(format!("     [Access: {}] [Eligibility: {} rows]", access_mode, scalar_hits));
241            plan.push("".to_string());
242        }
243        
244        plan.push(format!("Final Retrieval:"));
245        plan.push(format!("  -> ParallelRead (threads: {}, format: Parquet)", self.query_config.max_parallel_readers.unwrap_or(16)));
246        plan.push(divider);
247        
248        plan.join("\n")
249    }

255    pub async fn read_with_config_async(
256        &self, 
257        filter_str: Option<&str>, 
258        vector_filter: Option<VectorSearchParams>, 
259        columns: Option<&[&str]>,
260        config: QueryConfig
261    ) -> Result<Vec<RecordBatch>> {
262        let expr = match filter_str {
263            Some(f) => {
264                let schema = self.arrow_schema();
265                Some(FilterExpr::parse_sql(f, schema).await?)
266            }
267            _ => None,
268        };
269        self.read_expr_with_config_async(expr, vector_filter, columns, config, filter_str).await
270    }

272    pub async fn read_expr_with_config_async(
273        &self,
274        expr: Option<FilterExpr>,
275        vector_filter: Option<VectorSearchParams>,
276        columns: Option<&[&str]>,
277        config: QueryConfig,
278        filter_str: Option<&str>,
279    ) -> Result<Vec<RecordBatch>> {
280        use futures::StreamExt;
281
282        let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
283        let (_manifest, all_entries, version) = match manifest_manager.load_latest_full().await {
284            Ok((m, e, v)) => (m, e, v),
285            Err(_) => {
286                if manifest_manager.exists().await.unwrap_or(false) {
287                   (crate::core::manifest::Manifest::default(), Vec::new(), 0)
288                } else {
289                   let segments = self.list_segments_from_store().await.unwrap_or_default();
290                   (crate::core::manifest::Manifest::default(), segments, 0)
291                }
292            }
293        };        let entries_to_read = if version > 0 {
294            if expr.is_some() || vector_filter.is_some() {
295                let planner = QueryPlanner::new();
296                planner.prune_entries(&all_entries, expr.as_ref(), vector_filter.as_ref()).into_iter().map(|(e, _)| e).collect()
297            } else {
298                all_entries.clone()
299            }
300        } else {
301            // Version 0. Check if we want autodetection.
302            if manifest_manager.exists().await.unwrap_or(false) {
303                Vec::new() // Strictly follow the empty manifest
304            } else {
305                all_entries.clone() // Already contains discovered segments if in autodetection path
306            }
307        };
308        
309        // --- SMART HYBRID TRIGGER ---
310        // If we have both a vector filter AND a text filter on a BM25/Inverted indexed column,
311        // we switch to the Hybrid Coordinator path.
312        if let (Some(ref vs_params), Some(ref e)) = (&vector_filter, &expr) {
313             let manifest = self.manifest().await?;
314             let filtered_cols = e.get_referenced_columns();
315             
316             let current_schema = manifest.schemas.iter()
317                 .find(|s| s.schema_id == manifest.current_schema_id);
318             
319             let has_bm25_index = current_schema.map(|s| {
320                 s.fields.iter().any(|f| {
321                     let matches_col = filtered_cols.contains(&f.name);
322                     let has_index = f.indexes.iter().any(|idx| {
323                         matches!(idx, IndexAlgorithm::Bm25 { .. })
324                     });
325                     if matches_col {
326                         println!("Smart Trigger Check: Column '{}' has BM25 index: {}", f.name, has_index);
327                         if !has_index {
328                             println!("  Found indexes: {:?}", f.indexes);
329                         }
330                     }
331                     matches_col && has_index
332                 })
333             }).unwrap_or(false);
334
335             if has_bm25_index {
336                 tracing::info!("Smart Trigger: Executing Hybrid Search (RRF) for columns {:?}", filtered_cols);
337                 let coordinator = HybridSearchCoordinator::new();
338                 
339                 // Extract search terms from FilterExpr for the BM25 engine
340                 let extracted_query = {
341                     let conditions = e.extract_and_conditions();
342                     let mut terms = Vec::new();
343                     for f in conditions {
344                         if filtered_cols.contains(&f.column) {
345                            if let Some(v) = &f.min {
346                                if let Some(s) = v.as_str() { terms.push(s.to_string()); }
347                            }
348                            if let Some(vals) = &f.values {
349                                for v in vals {
350                                    if let Some(s) = v.as_str() { terms.push(s.to_string()); }
351                                }
352                            }
353                         }
354                     }
355                     if terms.is_empty() {
356                         filter_str.unwrap_or("").to_string()
357                     } else {
358                         terms.join(" ")
359                     }
360                 };
361
362                 let keyword_params = KeywordSearchParams {
363                     column: filtered_cols.iter().next().unwrap().clone(),
364                     query: extracted_query,
365                 };
366
367                 let scored_results = coordinator.execute_hybrid(
368                     self,
369                     filter_str,
370                     Some(vs_params.clone()),
371                     Some(keyword_params),
372                     vs_params.k,
373                     config.rrf_k,
374                 ).await?;
375
376                 // Convert ScoredResults back to RecordBatches by fetching from Parquet
377                 // This is a simplified version of the final row-fetcher
378                 return self.fetch_results_by_id(scored_results, columns).await;
379             }
380        }
381
382        // Handle standard vector search
383        if let Some(ref vs_params) = vector_filter {
384             // 1. Search Disk
385             let request = VectorSearchRequest::new(
386                 vs_params.column.clone(),
387                 vs_params.query.clone(),
388                 vs_params.k,
389                 vs_params.metric,
390             )
391             .with_filter(expr.clone())
392             .with_config(config.clone())
393             .with_ef_search(vs_params.ef_search)
394             .with_columns(columns.map(|c| c.iter().map(|s| s.to_string()).collect()));
395             
396             let mut results = execute_vector_search_with_config(
397                entries_to_read.clone(),
398                self.store.clone(),
399                self.data_store.clone(),
400                &self.uri,
401                request,
402            ).await?;
403
404             // 2. Search Memory
405             let memory_hits = {
406                 let idx = self.indexing.memory_index.read().unwrap();
407                 if let Some(mem_idx) = idx.as_ref() {
408                     let filter_bitmap = if let Some(ref e) = expr {
409                         let buffer = self.write_buffer.read().unwrap();
410                         let mut bitmap = RoaringBitmap::new();
411                          let mut offset = 0;
412                          let planner = QueryPlanner::new();
413                          for batch in buffer.iter() {
414                               if let Ok(mask) = planner.evaluate_expr(batch, e) {
415                                   for i in 0..batch.num_rows() {
416                                      if mask.value(i) {
417                                          bitmap.insert((offset + i) as u32);
418                                      }
419                                  }
420                              }
421                              offset += batch.num_rows();
422                          }
423                          Some(bitmap)
424                      } else {
425                          None
426                      };
427                      mem_idx.search(&vs_params.query, vs_params.k, filter_bitmap.as_ref())
428                  } else {
429                      vec![]
430                  }
431              };
432
433              if !memory_hits.is_empty() {
434                  let buffer = self.write_buffer.read().unwrap();
435                  if let Some(first) = buffer.first() {
436                      let schema = first.schema();
437                      let batch_offsets: Vec<usize> = buffer.iter().scan(0, |state, b| {
438                          let start = *state;
439                          *state += b.num_rows();
440                          Some(start)
441                      }).collect();
442                      
443                      let mut result_rows = Vec::new();
444                      for (id, _dist) in &memory_hits {
445                          for (i, offset) in batch_offsets.iter().enumerate().rev() {
446                              if *id >= *offset {
447                                  let row_idx = *id - offset;
448                                  if i < buffer.len() && row_idx < buffer[i].num_rows() {
449                                      result_rows.push(buffer[i].slice(row_idx, 1));
450                                  }
451                                  break;
452                              }
453                          }
454                      }
455                      
456                      if !result_rows.is_empty() {
457                          let mem_batch = arrow::compute::concat_batches(&schema, result_rows.iter().collect::<Vec<&RecordBatch>>())?;
458                          
459                          // Append _distance column to match disk search schema
460                          let mut new_fields = schema.fields().to_vec();
461                          new_fields.push(std::sync::Arc::new(arrow::datatypes::Field::new("distance", arrow::datatypes::DataType::Float32, false)));
462                          let new_schema = std::sync::Arc::new(arrow::datatypes::Schema::new(new_fields));
463                          
464                          let mut new_columns = mem_batch.columns().to_vec();
465                          let distance_array = arrow::array::Float32Array::from(memory_hits.iter().map(|(_, dist)| *dist).collect::<Vec<f32>>());
466                          new_columns.push(std::sync::Arc::new(distance_array));
467                          
468                          if let Ok(dist_batch) = RecordBatch::try_new(new_schema, new_columns) {
469                              results.push(dist_batch);
470                          }
471                      }
472                  }
473              }
474
475              return Ok(results);
476        }
477
478        // Extract Iceberg schema from the already-loaded manifest to avoid
479        // redundant manifest loads inside each per-segment read.
480        let iceberg_schema = _manifest.schemas.iter()
481            .find(|s| s.schema_id == _manifest.current_schema_id)
482            .cloned();
483        let iceberg_schema_arc = iceberg_schema.map(Arc::new);
484
485        // Capture current GPU context so it can be propagated into each async
486        // worker closure (thread_local is per-thread, not per-future).
487        let current_gpu_context = get_global_gpu_context();
488
489        let expr_arc = expr.map(Arc::new);
490        let concurrency = config.max_parallel_readers.unwrap_or_else(|| {
491            std::thread::available_parallelism().map(|n| n.get()).unwrap_or(4)
492        });
493        let stream = futures::stream::iter(entries_to_read)
494            .map(|entry| {
495                let expr_clone = expr_arc.clone();
496                let schema_clone = iceberg_schema_arc.clone();
497                let ctx = current_gpu_context.clone();
498                async move {
499                    if let Some(c) = ctx {
500                        crate::core::index::gpu::set_global_gpu_context(Some(c));
501                    }
502                    self.read_segment_expr(
503                        &entry, expr_clone.as_deref(), version, columns,
504                        schema_clone.as_deref(),
505                    ).await
506                }
507            })
508            .buffer_unordered(concurrency);
509
510        let results: Vec<Result<Vec<RecordBatch>>> = stream.collect().await;
511        let mut all_batches = Vec::new();
512        for (i, res) in results.into_iter().enumerate() {
513            match res {
514                Ok(b_vec) => {
515                     all_batches.extend(b_vec);
516                },
517                Err(e) => tracing::error!("Error reading batch {}: {}", i, e),
518            }
519        }
520
521        // --- Read from In-Memory Write Buffer ---
522        {
523            let buffer = self.write_buffer.read().unwrap();
524            if !buffer.is_empty() {
525                if let Some(ref e) = expr_arc {
526                    let planner = QueryPlanner::new();
527                    for batch in buffer.iter() {
528                         let batch_to_scan = if let Some(cols) = columns {
529                             let indices: Vec<usize> = cols.iter()
530                                .filter_map(|name| batch.schema().index_of(name).ok())
531                                .collect();
532                             batch.project(&indices).unwrap_or(batch.clone())
533                         } else {
534                             batch.clone()
535                         };
536
537                         if let Ok(filtered) = planner.filter_expr(&batch_to_scan, e) {
538                             if filtered.num_rows() > 0 {
539                                 all_batches.push(filtered);
540                             }
541                         }
542                    }
543                } else {
544                     for batch in buffer.iter() {
545                         if let Some(cols) = columns {
546                             let indices: Vec<usize> = cols.iter()
547                                .filter_map(|name| batch.schema().index_of(name).ok())
548                                .collect();
549                             if let Ok(projected) = batch.project(&indices) {
550                                 all_batches.push(projected);
551                             }
552                         } else {
553                             all_batches.push(batch.clone());
554                         }
555                     }
556                }
557            }
558        }
559        
560
561        Ok(all_batches)
562    }

584    pub async fn read_segment_expr(
585        &self,
586        entry: &ManifestEntry,
587        expr: Option<&FilterExpr>,
588        manifest_version: u64,
589        columns: Option<&[&str]>,
590        cached_iceberg_schema: Option<&crate::core::manifest::Schema>,
591    ) -> Result<Vec<RecordBatch>> {
592        let file_path_str = entry.file_path.clone();
593        let segment_id = file_path_str.split('/').next_back().unwrap_or(&file_path_str)
594            .strip_suffix(".parquet").unwrap_or(&file_path_str);
595
596        // Use cached schema if provided by caller; otherwise fall back to manifest load.
597        let iceberg_schema = if let Some(schema) = cached_iceberg_schema {
598            Some(schema.clone())
599        } else {
600            let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
601            let (manifest, _, _) = manifest_manager.load_latest_full().await.unwrap_or_default();
602            manifest.schemas.iter().find(|s| s.schema_id == manifest.current_schema_id).cloned()
603        };
604
605        // Resolve partition-aware path
606        let path = std::path::Path::new(&file_path_str);
607        let rel_parent = path.parent().and_then(|p| p.to_str()).unwrap_or("");
608        let full_base_path = if rel_parent.is_empty() {
609             self.uri.clone()
610        } else {
611             format!("{}/{}", self.uri, rel_parent)
612        };
613
614        let config = SegmentConfig::new(&full_base_path, segment_id)
615            .with_parquet_path(entry.file_path.clone())
616            .with_data_store(self.data_store.clone().unwrap_or(self.store.clone()))
617            .with_delete_files(entry.delete_files.clone())
618            .with_index_files(entry.index_files.clone())
619            .with_file_size(entry.file_size_bytes as u64)
620            .with_index_all(self.indexing.index_all)
621            .with_columns_to_index(self.indexing.index_columns.read().unwrap().clone());
622
623        let mut reader = HybridReader::new(config, self.store.clone(), &self.uri);
624        if let Some(s) = &iceberg_schema {
625            reader = reader.with_iceberg_schema(s.clone());
626        }
627
628        let full_schema = if let Some(schema) = &iceberg_schema {
629             Arc::new(schema.to_arrow())
630        } else {
631             self.arrow_schema()
632        };
633
634        let target_schema = if let Some(cols) = columns {
635             let fields: Vec<arrow::datatypes::Field> = cols.iter()
636                 .filter_map(|name| full_schema.field_with_name(name).ok().cloned())
637                 .collect();
638             Some(Arc::new(Schema::new(fields)))
639        } else {
640             Some(full_schema)
641        };
642
643        if manifest_version == 0 || expr.is_none() {
644            let mut stream = reader.stream_all(target_schema).await?;
645            let mut batches = Vec::new();
646            while let Some(batch_result) = stream.next().await {
647                batches.push(batch_result?);
648            }
649            return Ok(batches);
650        }
651
652        let expr = expr.unwrap();
653        let and_filters = expr.extract_and_conditions();
654
655        // Try to use index for the FIRST filter that has one
656        let mut batches = Vec::new();
657        let mut index_used = false;
658
659        for filter in &and_filters {
660             if let Ok(indexed_batches) = reader.query_index_first(filter, target_schema.clone()).await {
661                 batches = indexed_batches;
662                 index_used = true;
663                 break;
664             } 
665        }
666
667        if !index_used {
668            // Point Selection Optimization: Check Bloom Filters before scanning
669            for filter in &and_filters {
670                if let Some(vals) = &filter.values {
671                    if vals.len() == 1 {
672                        if !reader.check_bloom_filter(&filter.column, &vals[0]).await.unwrap_or(true) {
673                            tracing::debug!("Bloom Filter Pruned segment: {} for col: {}", segment_id, filter.column);
674                            return Ok(vec![]);
675                        }
676                    }
677                } else if let (Some(min), Some(max)) = (&filter.min, &filter.max) {
678                    if min == max {
679                        if !reader.check_bloom_filter(&filter.column, min).await.unwrap_or(true) {
680                            tracing::debug!("Bloom Filter Pruned segment: {} for col: {}", segment_id, filter.column);
681                            return Ok(vec![]);
682                        }
683                    }
684                }
685            }
686            
687            let mut stream = reader.stream_all(target_schema).await?;
688            while let Some(batch_result) = stream.next().await {
689                batches.push(batch_result?);
690            }
691        }
692
693        let planner = QueryPlanner::new();
694        let mut filtered_batches = Vec::new();
695        for batch in batches {
696            match planner.filter_expr(&batch, expr) {
697                Ok(filtered) => {
698                    if filtered.num_rows() > 0 {
699                         filtered_batches.push(filtered);
700                    }
701                }
702                Err(e) => {
703                    tracing::error!("Failed to evaluate filter expression on batch: {}", e);
704                }
705            }
706        }
707        Ok(filtered_batches)
708    }

741    async fn list_segments_from_store(&self) -> Result<Vec<ManifestEntry>> {
742        use futures::StreamExt;
743        let mut entries = Vec::new();
744        let mut stream = self.store.list(None);
745        while let Some(res) = stream.next().await {
746            if let Ok(meta) = res {
747                let path = meta.location.to_string();
748                if (!path.contains("/") || path.contains("data/")) && path.ends_with(".parquet") 
749                   && !path.contains(".inv.parquet") && !path.contains(".hnsw.") {
750                       entries.push(ManifestEntry {
751                           file_path: path,
752                           file_size_bytes: meta.size as i64,
753                           ..Default::default()
754                       });
755                }
756            }
757        }
758        Ok(entries)
759    }

761    pub async fn execute_vector_search_as_scored(
762        &self,
763        params: VectorSearchParams,
764    ) -> Result<Vec<ScoredResult>> {
765        let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
766        let (_, all_entries, _) = manifest_manager.load_latest_full().await?;
767        
768        let request = VectorSearchRequest::new(
769            params.column.clone(),
770            params.query.clone(),
771            params.k,
772            params.metric,
773        ).with_ef_search(params.ef_search);
774
775        // For now, we reuse the existing vector search and convert RecordBatches to ScoredResults
776        // In a future optimization, we'll return ScoredResults directly from the reader to avoid Parquet I/O if possible
777        let batches = execute_vector_search_with_config(
778            all_entries,
779            self.store.clone(),
780            self.data_store.clone(),
781            &self.uri,
782            request,
783        ).await?;
784
785        let mut scored_results = Vec::new();
786        for batch in batches {
787            // RecordBatch results from vector search include a "distance" column
788            let dist_col = batch.column(batch.num_columns() - 1)
789                .as_any()
790                .downcast_ref::<arrow::array::Float32Array>()
791                .ok_or_else(|| anyhow::anyhow!("Missing distance column in vector search result"))?;
792            
793            // Note: Parallel vector search currently doesn't preserve segment/row IDs in the final RecordBatch
794            // We'll need to add those to the schema in Phase 3. 
795            // For Phase 2, we use a synthetic segment ID "vector_path"
796            for i in 0..batch.num_rows() {
797                scored_results.push(ScoredResult {
798                    segment_id: "vector_path".to_string(),
799                    row_id: i as u32,
800                    score: dist_col.value(i),
801                });
802            }
803        }
804        Ok(scored_results)
805    }

807    pub async fn execute_keyword_search_as_scored(
808        &self,
809        params: KeywordSearchParams,
810    ) -> Result<Vec<ScoredResult>> {
811        let manifest = self.manifest().await?;
812        let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
813        let all_entries = manifest_manager.load_all_entries(&manifest).await?;
814        
815        let mut all_scored = Vec::new();
816        for entry in all_entries {
817            let file_path_str = entry.file_path.clone();
818            let segment_id = file_path_str
819                .split('/')
820                .next_back()
821                .unwrap_or(&file_path_str)
822                .strip_suffix(".parquet")
823                .unwrap_or(&file_path_str);
824
825            let config = SegmentConfig::new(&self.uri, segment_id)
826                .with_parquet_path(entry.file_path.clone())
827                .with_index_files(entry.index_files.clone())
828                .with_record_count(entry.record_count as u64);
829
830            let reader = HybridReader::new(config, self.store.clone(), &self.uri);
831            let matches = reader.keyword_search_index(&params.column, &params.query, 1000, None).await?;
832            
833            for (row_id, score) in matches {
834                all_scored.push(ScoredResult {
835                    segment_id: segment_id.to_string(),
836                    row_id: row_id as u32,
837                    score,
838                });
839            }
840        }
841        Ok(all_scored)
842    }

844    /// Helper to fetch full RecordBatches for a set of fused IDs
845    pub async fn fetch_results_by_id(
846        &self,
847        results: Vec<ScoredResult>,
848        columns: Option<&[&str]>,
849    ) -> Result<Vec<RecordBatch>> {
850        if results.is_empty() {
851            return Ok(vec![]);
852        }
853
854        // Group by segment to minimize I/O
855        let mut by_segment: HashMap<String, Vec<(u32, f32)>> = HashMap::new();
856        for r in results {
857            by_segment.entry(r.segment_id).or_default().push((r.row_id, r.score));
858        }
859
860        let mut final_batches = Vec::new();
861        let manifest = self.manifest().await?;
862        let manifest_manager = ManifestManager::new(self.store.clone(), "", &self.uri);
863        let all_entries = manifest_manager.load_all_entries(&manifest).await?;
864        
865        for (seg_id, rows) in by_segment {
866            if seg_id == "vector_path" { continue; } 
867
868            // Find segment in all entries
869            let entry = all_entries.iter()
870                .find(|e| e.file_path.contains(&seg_id))
871                .ok_or_else(|| anyhow::anyhow!("Segment {} not found in manifest", seg_id))?;
872
873            let config = SegmentConfig::new(&self.uri, &seg_id)
874                .with_parquet_path(entry.file_path.clone())
875                .with_index_files(entry.index_files.clone());
876
877            let reader = HybridReader::new(config, self.store.clone(), &self.uri);
878            
879            // Convert fused row IDs and scores to RecordBatch
880            let batch = reader.read_rows_by_id(rows, columns).await?;
881            final_batches.push(batch);
882        }
883
884        Ok(final_batches)
885    }
