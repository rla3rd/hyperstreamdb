3use pyo3::prelude::*;
4use crate::core::index::gpu::{ComputeContext, ComputeBackend};
5use std::sync::{Arc, Mutex};
6
7/// Performance statistics for GPU operations
8#[derive(Debug, Clone, Default)]
9pub struct GPUStats {
10    pub total_kernel_launches: u64,
11    pub total_gpu_time_ms: f64,
12    pub total_cpu_time_ms: f64,
13    pub total_vectors_processed: u64,
14    pub memory_transfers_mb: f64,
15}
16
17/// Python wrapper for Device with performance monitoring
18#[pyclass(name = "Device")]
19pub struct PyDevice {
20    pub(crate) context: ComputeContext,
21    pub(crate) stats: Arc<Mutex<GPUStats>>,
22}
23
24#[pymethods]
25impl PyDevice {
26    /// Create a new Device instance.
27    #[new]
28    #[pyo3(signature = (device, index=None))]
29    fn new(device: &str, index: Option<i32>) -> PyResult<Self> {
30        let (backend_str, device_id) = if device.contains(':') {
31             let parts: Vec<&str> = device.split(':').collect();
32             if parts.len() != 2 {
33                 return Err(pyo3::exceptions::PyValueError::new_err("Device string must be in 'type:index' format (e.g., 'cuda:0')"));
34             }
35             let b = parts[0];
36             let i = parts[1].parse::<i32>().map_err(|_| pyo3::exceptions::PyValueError::new_err("Invalid device index"))?;
37             (b, i)
38        } else {
39             (device, index.unwrap_or(if device == "cpu" { -1 } else { 0 }))
40        };
41
42        let backend_enum = match backend_str.to_lowercase().as_str() {
43            "cpu" => ComputeBackend::Cpu,
44            "cuda" => {
45                #[cfg(feature = "cuda")]
46                { ComputeBackend::Cuda }
47                #[cfg(not(feature = "cuda"))]
48                { return Err(pyo3::exceptions::PyRuntimeError::new_err(
49                    "CUDA backend not available in this build. Ensure NVIDIA drivers are installed and try 'pip install hyperstreamdb'. See GPU_SETUP_GUIDE.md for detailed troubleshooting."
50                )); }
51            }
52            "mps" | "metal" => {
53                #[cfg(target_os = "macos")]
54                { ComputeBackend::Mps }
55                #[cfg(not(target_os = "macos"))]
56                { return Err(pyo3::exceptions::PyRuntimeError::new_err("Backend 'mps' is only available on macOS.")); }
57            }
58            "intel" | "graphics" => {
59                #[cfg(target_os = "linux")]
60                { ComputeBackend::Intel }
61                #[cfg(not(target_os = "linux"))]
62                { return Err(pyo3::exceptions::PyRuntimeError::new_err("Backend 'intel' (XPU) is only supported on Linux.")); }
63            }
64            "rocm" => {
65                #[cfg(target_os = "linux")]
66                { ComputeBackend::Rocm }
67                #[cfg(not(target_os = "linux"))]
68                { return Err(pyo3::exceptions::PyRuntimeError::new_err("Backend 'rocm' is only supported on Linux.")); }
69            }
70            "auto" | "gpu" => {
71                let context = crate::core::index::gpu::ComputeContext::auto_detect();
72                return Ok(Self {
73                    context,
74                    stats: Arc::new(Mutex::new(GPUStats::default())),
75                });
76            }
77            _ => {
78                return Err(pyo3::exceptions::PyValueError::new_err(format!("Unknown device type '{}'. Valid types: 'cpu', 'cuda', 'mps', 'intel', 'rocm', 'gpu', 'auto'", backend_str)));
79            }
80        };
81
82        Ok(Self {
83            context: ComputeContext {
84                backend: backend_enum,
85                device_id,
86                implementation: None,
87            },
88            stats: Arc::new(Mutex::new(GPUStats::default())),
89        })
90    }
91
92    /// Check if a specific device type is available on this system.
93    #[staticmethod]
94    fn is_available(device_type: &str) -> bool {
95        let backend = match device_type.to_lowercase().as_str() {
96            "cpu" => ComputeBackend::Cpu,
97            "cuda" => ComputeBackend::Cuda,
98            "mps" | "metal" => ComputeBackend::Mps,
99            "intel" | "graphics" => ComputeBackend::Intel,
100            "rocm" => ComputeBackend::Rocm,
101            _ => return false,
102        };
103        ComputeContext { backend, device_id: 0, implementation: None }.is_available()
104    }
105
106    #[getter]
107    fn type_name(&self) -> String {
108        match self.context.backend {
109            ComputeBackend::Cpu => "cpu".to_string(),
110            ComputeBackend::Cuda => "cuda".to_string(),
111            ComputeBackend::Rocm => "rocm".to_string(),
112            ComputeBackend::Mps => "mps".to_string(),
113            ComputeBackend::Intel => "intel".to_string(),
114        }
115    }
116
117    #[getter]
118    fn index(&self) -> i32 {
119        self.context.device_id
120    }
121
122    fn get_stats(&self, py: Python<'_>) -> PyResult<Py<PyAny>> {
123        let stats = self.stats.lock().unwrap();
124        let dict = pyo3::types::PyDict::new(py);
125        dict.set_item("total_kernel_launches", stats.total_kernel_launches)?;
126        dict.set_item("total_gpu_time_ms", stats.total_gpu_time_ms)?;
127        dict.set_item("total_cpu_time_ms", stats.total_cpu_time_ms)?;
128        dict.set_item("total_vectors_processed", stats.total_vectors_processed)?;
129        dict.set_item("memory_transfers_mb", stats.memory_transfers_mb)?;
130        Ok(dict.into())
131    }
132
133    fn reset_stats(&self) {
134        let mut stats = self.stats.lock().unwrap();
135        *stats = GPUStats::default();
136    }
137
138    fn activate(&self) {
139        crate::core::index::gpu::set_global_gpu_context(Some(self.context.clone()));
140    }
141
142    #[staticmethod]
143    fn deactivate() {
144        crate::core::index::gpu::set_global_gpu_context(None);
145    }
146
147    fn __repr__(&self) -> String {
148        format!("Device(type='{}', index={})", self.type_name(), self.index())
149    }
150
151    #[pyo3(signature = (vectors, k, max_iters=10))]
152    fn kmeans(
153        &self,
154        py: Python<'_>,
155        vectors: Vec<Vec<f32>>,
156        k: usize,
157        max_iters: usize,
158    ) -> PyResult<(Vec<Vec<f32>>, Vec<usize>)> {
159        #[allow(deprecated)]
160        py.allow_threads(|| {
161            crate::core::index::ivf::simple_kmeans(&vectors, k, max_iters)
162        }).map_err(|e| pyo3::exceptions::PyRuntimeError::new_err(e.to_string()))
163    }
164}
165
166impl PyDevice {
167    pub fn get_context(&self) -> &ComputeContext {
168        &self.context
169    }
170
171    pub fn get_stats_tracker(&self) -> Arc<Mutex<GPUStats>> {
172        self.stats.clone()
173    }
174}
