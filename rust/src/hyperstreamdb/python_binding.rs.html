35// Helper function to parse metric string to VectorMetric enum
36// Uses native Rust names: L2, Cosine, InnerProduct, L1, Hamming, Jaccard
37// Also accepts lowercase aliases for backward compatibility
38
39fn parse_metric(metric_str: &str) -> PyResult<VectorMetric> {
40    match metric_str {
41        "l2" | "L2" => Ok(VectorMetric::L2),
42        "cosine" | "Cosine" => Ok(VectorMetric::Cosine),
43        "innerproduct" | "inner_product" | "InnerProduct" => Ok(VectorMetric::InnerProduct),
44        "l1" | "L1" => Ok(VectorMetric::L1),
45        "hamming" | "Hamming" => Ok(VectorMetric::Hamming),
46        "jaccard" | "Jaccard" => Ok(VectorMetric::Jaccard),
47        _ => Err(pyo3::exceptions::PyValueError::new_err(format!("Invalid metric '{}'", metric_str))),
48    }
49}

51fn parse_index_algorithm(val: Bound<'_, PyAny>) -> PyResult<IndexAlgorithm> {
52    if let Ok(s) = val.extract::<String>() {
53        match s.to_lowercase().as_str() {
54            "hnsw" => Ok(IndexAlgorithm::Hnsw { 
55                metric: "l2".to_string(), 
56                complexity: 16, 
57                quality: 200, 
58                build_device: None, 
59                search_device: None 
60            }),
61            "hnsw_pq" | "pq" => Ok(IndexAlgorithm::HnswPq { 
62                metric: "l2".to_string(), 
63                complexity: 16, 
64                quality: 200, 
65                compression: 8 
66            }),
67            "hnsw_tq4" | "tq4" => Ok(IndexAlgorithm::HnswTq4 { 
68                metric: "l2".to_string(), 
69                complexity: 16, 
70                quality: 200 
71            }),
72            "hnsw_tq8" | "tq8" => Ok(IndexAlgorithm::HnswTq8 { 
73                metric: "l2".to_string(), 
74                complexity: 16, 
75                quality: 200 
76            }),
77            "bm25" => Ok(IndexAlgorithm::Bm25 {
78                k1: 1.2,
79                b: 0.75,
80                tokenizer: "default".to_string(),
81            }),
82            "bloom" => Ok(IndexAlgorithm::Bloom { fpr: 0.05 }),
83            "bitmap" | "inverted" => Ok(IndexAlgorithm::Bitmap),
84            _ => Err(pyo3::exceptions::PyValueError::new_err(format!("Unknown index type: {}", s))),
85        }
86    } else if let Ok(dict) = val.downcast::<PyDict>() {
87        let type_str: String = dict.get_item("type")?.ok_or_else(|| pyo3::exceptions::PyKeyError::new_err("Missing 'type' key in index config"))?.extract()?;
88        match type_str.to_lowercase().as_str() {
89            "hnsw" => {
90                let metric = dict.get_item("metric")?.and_then(|v| v.extract::<String>().ok()).unwrap_or_else(|| "l2".to_string());
91                let complexity = dict.get_item("complexity")?.and_then(|v| v.extract::<usize>().ok())
92                    .or_else(|| dict.get_item("m").ok().flatten().and_then(|v| v.extract::<usize>().ok()))
93                    .unwrap_or(16);
94                let quality = dict.get_item("quality")?.and_then(|v| v.extract::<usize>().ok())
95                    .or_else(|| dict.get_item("ef_construction").ok().flatten().and_then(|v| v.extract::<usize>().ok()))
96                    .unwrap_or(200);
97                let build_device = dict.get_item("build_device")?.and_then(|v| v.extract::<String>().ok());
98                let search_device = dict.get_item("search_device")?.and_then(|v| v.extract::<String>().ok());
99                Ok(IndexAlgorithm::Hnsw { metric, complexity, quality, build_device, search_device })
100            },
101            "hnsw_pq" | "pq" => {
102                let metric = dict.get_item("metric")?.and_then(|v| v.extract::<String>().ok()).unwrap_or_else(|| "l2".to_string());
103                let compression = dict.get_item("compression")?.and_then(|v| v.extract::<usize>().ok())
104                    .or_else(|| dict.get_item("subspaces").ok().flatten().and_then(|v| v.extract::<usize>().ok()))
105                    .unwrap_or(8);
106                let complexity = dict.get_item("complexity")?.and_then(|v| v.extract::<usize>().ok())
107                    .or_else(|| dict.get_item("m").ok().flatten().and_then(|v| v.extract::<usize>().ok()))
108                    .unwrap_or(16);
109                let quality = dict.get_item("quality")?.and_then(|v| v.extract::<usize>().ok())
110                    .or_else(|| dict.get_item("ef_construction").ok().flatten().and_then(|v| v.extract::<usize>().ok()))
111                    .unwrap_or(200);
112                Ok(IndexAlgorithm::HnswPq { metric, complexity, quality, compression })
113            },
114            "hnsw_tq4" | "tq4" => {
115                let metric = dict.get_item("metric")?.and_then(|v| v.extract::<String>().ok()).unwrap_or_else(|| "l2".to_string());
116                let complexity = dict.get_item("complexity")?.and_then(|v| v.extract::<usize>().ok())
117                    .or_else(|| dict.get_item("m").ok().flatten().and_then(|v| v.extract::<usize>().ok()))
118                    .unwrap_or(16);
119                let quality = dict.get_item("quality")?.and_then(|v| v.extract::<usize>().ok())
120                    .or_else(|| dict.get_item("ef_construction").ok().flatten().and_then(|v| v.extract::<usize>().ok()))
121                    .unwrap_or(200);
122                Ok(IndexAlgorithm::HnswTq4 { metric, complexity, quality })
123            },
124            "hnsw_tq8" | "tq8" => {
125                let metric = dict.get_item("metric")?.and_then(|v| v.extract::<String>().ok()).unwrap_or_else(|| "l2".to_string());
126                let complexity = dict.get_item("complexity")?.and_then(|v| v.extract::<usize>().ok())
127                    .or_else(|| dict.get_item("m").ok().flatten().and_then(|v| v.extract::<usize>().ok()))
128                    .unwrap_or(16);
129                let quality = dict.get_item("quality")?.and_then(|v| v.extract::<usize>().ok())
130                    .or_else(|| dict.get_item("ef_construction").ok().flatten().and_then(|v| v.extract::<usize>().ok()))
131                    .unwrap_or(200);
132                Ok(IndexAlgorithm::HnswTq8 { metric, complexity, quality })
133            },
134            "bm25" => {
135                let k1 = dict.get_item("k1")?.and_then(|v| v.extract().ok()).unwrap_or(1.2);
136                let b = dict.get_item("b")?.and_then(|v| v.extract().ok()).unwrap_or(0.75);
137                let tokenizer = dict.get_item("tokenizer")?.and_then(|v| v.extract().ok()).unwrap_or_else(|| "default".to_string());
138                Ok(IndexAlgorithm::Bm25 { k1, b, tokenizer })
139            },
140            "bloom" => {
141                let fpr = dict.get_item("fpr")?.and_then(|v| v.extract().ok()).unwrap_or(0.05);
142                Ok(IndexAlgorithm::Bloom { fpr })
143            },
144            "bitmap" | "inverted" => Ok(IndexAlgorithm::Bitmap),
145            _ => Err(pyo3::exceptions::PyValueError::new_err(format!("Unknown index type: {}", type_str))),
146        }
147    } else {
148        Err(pyo3::exceptions::PyTypeError::new_err("Index algorithm must be a string or a dict"))
149    }
150}

665    /// This allows the Rust core to trigger vectorization even without Python GIL

683            name.clone(),

714    /// Read table to PyArrow Table with optional filtering and column projection
715    /// 
716    /// Args:
717    ///     filter: Optional SQL-like filter string (e.g., "age > 25 AND city = 'NYC'")
718    ///     vector_filter: Optional dict for vector search:
719    ///         - column: str (required) - vector column name
720    ///         - query: list (required) - query vector
721    ///         - k: int (required) - number of results
722    ///         - metric: str (optional) - 'l2'|'cosine'|'innerproduct'|'l1'|'hamming'|'jaccard' (default: l2)
723    ///         - ef_search: int (optional) - HNSW ef parameter for search quality tuning
724    ///         - probes: int (optional) - IVF probes parameter for search speed tuning
725    ///     columns: Optional list of column names to read (skips others like embeddings)
726    /// 
727    /// Returns:
728    ///     PyArrow Table (via Arrow C Data Interface)
729    /// 
730    /// Example:
731    ///     # Vector search with cosine metric
732    ///     df = table.to_pandas(vector_filter={"column": "embedding", "query": [1.0, 2.0], "k": 3, "metric": "cosine"})
733    #[pyo3(signature = (filter=None, vector_filter=None, columns=None, device=None))]
734    fn to_arrow(
735        &self,
736        py: Python<'_>,
737        filter: Option<String>,
738        vector_filter: Option<Bound<'_, PyDict>>,
739        columns: Option<Vec<String>>,
740        device: Option<Py<PyDevice>>,
741    ) -> PyResult<Py<PyAny>> {
742        let _pyarrow = py.import("pyarrow")?;
743        // Parse vector filter if provided
744        let vs_params = if let Some(ref vf) = vector_filter {
745            let column: String = vf.get_item("column")?
746                .ok_or_else(|| pyo3::exceptions::PyValueError::new_err("vector_filter requires 'column' key"))?
747                .extract()?;
748            let k: usize = vf.get_item("k")?
749                .ok_or_else(|| pyo3::exceptions::PyValueError::new_err("vector_filter requires 'k' key"))?
750                .extract()?;
751            let query_obj = vf.get_item("query")?
752                .ok_or_else(|| pyo3::exceptions::PyValueError::new_err("vector_filter requires 'query' key"))?;
753            let query: Vec<f32> = query_obj.extract()?;
754            let mut params = VectorSearchParams::new(&column, crate::core::index::VectorValue::Float32(query), k);
755            
756            // Parse optional metric parameter
757            if let Ok(Some(metric_obj)) = vf.get_item("metric") {
758                if let Ok(metric_str) = metric_obj.extract::<String>() {
759                    params = params.with_metric(parse_metric(&metric_str)?);
760                }
761            }
762            
763            // Parse optional ef_search parameter (for HNSW)
764            if let Ok(Some(ef_obj)) = vf.get_item("ef_search") {
765                if let Ok(ef_search) = ef_obj.extract::<usize>() {
766                    params = params.with_ef_search(ef_search);
767                }
768            }
769            
770            // Parse optional probes parameter (for IVF)
771            if let Ok(Some(probes_obj)) = vf.get_item("probes") {
772                if let Ok(probes) = probes_obj.extract::<usize>() {
773                    params = params.with_probes(probes);
774                }
775            }
776            
777            // Parse optional rrf_k parameter
778            let rrf_k = if let Ok(Some(rrf_obj)) = vf.get_item("rrf_k") {
779                rrf_obj.extract::<f32>().ok()
780            } else {
781                None
782            };
783            
784            Some((params, rrf_k))
785        } else {
786            None
787        };
788
789        // Clone for closure
790        let filter_str = filter.clone();
791        let vs_params_combined = vs_params.clone();
792        let columns_clone = columns.clone();
793
794        let table_schema = self.table.arrow_schema();
795        let projected_schema = if let Some(cols) = &columns_clone {
796            let mut fields = Vec::new();
797            for c in cols {
798                if let Some((_, field)) = table_schema.column_with_name(c) {
799                    fields.push(std::sync::Arc::new(field.clone()));
800                }
801            }
802            std::sync::Arc::new(arrow::datatypes::Schema::new(fields))
803        } else {
804            table_schema
805        };
806
807        // Determine which device to use: per-call device or table-level default
808        let ctx = device.as_ref().or(self.device.as_ref()).map(|c| c.clone_ref(py));
809        let rust_context = if let Some(py_ctx) = ctx {
810            let ctx_borrow = py_ctx.bind(py).borrow();
811            Some(ctx_borrow.context.clone())
812        } else {
813            None
814        };
815
816        // Call core Table API with column projection, releasing the GIL
817        let batches = py.allow_threads(move || {
818            if let Some(c) = rust_context {
819                crate::core::index::gpu::set_global_gpu_context(Some(c));
820            }
821            
822            let mut config = self.table.query_config().clone();
823            let (vs_params, _rrf_k) = if let Some((p, k)) = vs_params_combined {
824                if let Some(val) = k {
825                    config = config.with_rrf_k(val);
826                }
827                (Some(p), k)
828            } else {
829                (None, None)
830            };
831
832            TOKIO_RUNTIME.block_on(self.table.read_with_config_async(
833                filter_str.as_deref(), 
834                vs_params, 
835                columns_clone.as_ref().map(|c| c.iter().map(|s| s.as_str()).collect::<Vec<&str>>()).as_deref(), 
836                config
837            ))
838        }).map_err(|e| pyo3::exceptions::PyRuntimeError::new_err((e.to_string(), )))?;
839        
840        let final_schema = if vs_params.is_some() {
841            // Include distance column in schema if it was a vector search
842            let mut fields: Vec<arrow::datatypes::Field> = projected_schema.fields().iter().map(|f| f.as_ref().clone()).collect();
843            if !fields.iter().any(|f| f.name() == "distance") {
844                fields.push(arrow::datatypes::Field::new("distance", arrow::datatypes::DataType::Float32, false));
845            }
846            std::sync::Arc::new(arrow::datatypes::Schema::new(fields))
847        } else {
848            projected_schema
849        };
850        
851        // Cast all batches to the final schema to support evolution (e.g. promoting Int32 to Int64)
852        let casted_batches: Vec<RecordBatch> = batches.into_iter()
853            .map(|b| {
854                final_schema.fields().iter()
855                    .map(|field| {
856                        if let Some(col) = b.column_by_name(field.name()) {
857                            if col.data_type() != field.data_type() {
858                                // Automatically cast types if they evolved (e.g. Int32 -> Int64)
859                                arrow::compute::cast(col, field.data_type())
860                                    .map_err(|e| pyo3::exceptions::PyRuntimeError::new_err(format!("Schema evolution cast failed for column '{}': {}", field.name(), e)))
861                            } else {
862                                Ok(col.clone())
863                            }
864                        } else {
865                            // Handle missing columns in older segments (NULL fill)
866                            Ok(arrow::array::new_null_array(field.data_type(), b.num_rows()))
867                        }
868                    })
869                    .collect::<Result<Vec<_>, pyo3::PyErr>>()
870                    .and_then(|cols| {
871                        RecordBatch::try_new(final_schema.clone(), cols)
872                            .map_err(|e| pyo3::exceptions::PyRuntimeError::new_err(format!("Failed to create batch with evolved schema: {}", e)))
873                    })
874            })
875            .collect::<Result<Vec<RecordBatch>, pyo3::PyErr>>()?;
876
877        // Convert to Arrow C Data Interface
878        arrow_batches_to_pyarrow(py, casted_batches, final_schema)
879    }

907    /// Write data to table.
908    /// Supports:
909    /// - List of PyArrow RecordBatches
910    /// - PyArrow Table
911    /// - Pandas DataFrame
912    #[pyo3(signature = (data, device=None))]
913    fn write(&self, py: Python<'_>, data: Bound<'_, PyAny>, device: Option<Py<PyDevice>>) -> PyResult<()> {
914        let ctx = device.as_ref().or(self.device.as_ref()).map(|c| c.clone_ref(py));
915        let rust_context = if let Some(py_ctx) = ctx {
916            let ctx_borrow = py_ctx.bind(py).borrow();
917            Some(ctx_borrow.context.clone())
918        } else {
919            None
920        };
921
922        // 1. Check if it's a list (List[RecordBatch])
923        if let Ok(list) = data.downcast::<pyo3::types::PyList>() {
924             let mut batches = Vec::new();
925             for item in list {
926                 // Export PyArrow RecordBatch to C Interface
927                 let mut array = FFI_ArrowArray::empty();
928                 let mut schema = FFI_ArrowSchema::empty();
929                 
930                 let array_ptr = &mut array as *mut _ as Py_uintptr_t;
931                 let schema_ptr = &mut schema as *mut _ as Py_uintptr_t;
932                 
933                 item.call_method1("_export_to_c", (array_ptr, schema_ptr))?;
934                 
935                 // Import as Rust RecordBatch
936                 // Safety: We just exported it from PyArrow, so it should be valid.
937                 let batch = unsafe { import_record_batch_from_c(array, &schema) }
938                     .map_err(|e: arrow::error::ArrowError| pyo3::exceptions::PyRuntimeError::new_err((e.to_string(), )))?;
939                     
940                 batches.push(batch);
941             }
942             
943             // Call core Table API, releasing the GIL
944             py.allow_threads(move || {
945                if let Some(c) = rust_context {
946                    crate::core::index::gpu::set_global_gpu_context(Some(c));
947                }
948                self.table.write(batches)
949             }).map_err(|e: anyhow::Error| pyo3::exceptions::PyRuntimeError::new_err((e.to_string(), )))
950             
951        } else {
952             // Fallback to Table/DataFrame handling
953             let obj: Py<PyAny> = data.unbind();
954             
955             let context_clone = device.as_ref().map(|c| c.clone_ref(py));
956             // Let's try arrow first as it's lighter
957             if self.write_arrow(py, obj.clone_ref(py), context_clone.as_ref().map(|c| c.clone_ref(py))).is_ok() {
958                 Ok(())
959             } else {
960                 // Try pandas
961                 self.write_pandas(py, obj, context_clone)
962             }
963        }
964    }

1010    /// Parallel vector search - runs multiple queries in parallel in Rust (bypasses Python GIL)
1011    /// 
1012    /// This method submits all queries to a dedicated Rust thread pool, allowing
1013    /// true parallelism that bypasses Python's GIL limitations.
1014    /// 
1015    /// Args:
1016    ///     queries: List of tuples (column, query_vector, k, filter_optional)
1017    ///              Each tuple is (str, List[float], int, Optional[str])
1018    /// 
1019    /// Returns:
1020    ///     List of PyArrow Tables (one per query)
1021    /// 
1022    /// Example:
1023    ///     queries = [
1024    ///         ("embedding", [0.1, 0.2, ...], 10, None),
1025    ///         ("embedding", [0.3, 0.4, ...], 10, "user_id < 100"),
1026    ///     ]
1027    ///     results = table.search_parallel(queries)
1028    ///     # All queries run in parallel in Rust - no GIL blocking!
1029    #[pyo3(signature = (queries))]
1030    fn search_parallel(
1031        &self,
1032        py: Python<'_>,
1033        queries: Vec<(String, Vec<f32>, usize, Option<String>)>,
1034    ) -> PyResult<Py<PyAny>> {
1035        use futures::future::join_all;
1036        
1037        let table = Arc::new(self.table.clone());
1038        
1039        // Release GIL and run all queries in parallel in Rust
1040        let results: Result<Vec<Vec<RecordBatch>>, anyhow::Error> = py.allow_threads(move || {
1041            TOKIO_RUNTIME.block_on(async {
1042                // Spawn all queries as concurrent tasks
1043                let tasks: Vec<_> = queries.into_iter().map(|(column, query, k, filter): (String, Vec<f32>, usize, Option<String>)| {
1044                    let table_clone = table.clone();
1045                    tokio::spawn(async move {
1046                        let vf_params = VectorSearchParams::new(&column, crate::core::index::VectorValue::Float32(query), k);
1047                        table_clone.read_async(filter.as_deref(), Some(vf_params), None).await
1048                    })
1049                }).collect();
1050                
1051                // Wait for all queries to complete in parallel
1052                let join_results = join_all(tasks).await;
1053                
1054                // Collect results, handling errors
1055                let mut all_results = Vec::new();
1056                for result in join_results {
1057                    match result {
1058                        Ok(Ok(batches)) => all_results.push(batches),
1059                        Ok(Err(e)) => return Err(anyhow::anyhow!("Query failed: {}", e)),
1060                        Err(e) => return Err(anyhow::anyhow!("Task join failed: {}", e)),
1061                    }
1062                }
1063                Ok(all_results)
1064            })
1065        });
1066        
1067        // Convert results to PyArrow tables
1068        match results {
1069            Ok(batch_vecs) => {
1070                let schema = self.table.arrow_schema();
1071                let mut py_tables = Vec::new();
1072                for batches in batch_vecs {
1073                    // Convert RecordBatches to Py<PyAny> (PyArrow Table)
1074                    let py_table = arrow_batches_to_pyarrow(py, batches, schema.clone())?;
1075                    py_tables.push(py_table);
1076                }
1077                // Manually convert to PyList to avoid PyO3 ambiguity
1078                let list = pyo3::types::PyList::new(py, py_tables)?;
1079                Ok(list.into())
1080            }
1081            Err(e) => {
1082                let msg = format!("{}", e);
1083                Err(pyo3::exceptions::PyRuntimeError::new_err(msg))
1084            },
1085        }
1086    }

1317    /// Execute SQL query against the table.
1318    /// The table is registered as 't'.
1319    /// 
1320    /// Example:
1321    ///     table.sql("SELECT * FROM t WHERE id > 10")
1322    fn execute_sql(&self, py: Python<'_>, query: String) -> PyResult<Py<PyAny>> {
1323        let query = sanitize_sql(&query);
1324        let rt = self.table.runtime();
1325        
1326        let batch_result: Result<(Vec<RecordBatch>, arrow::datatypes::SchemaRef), String> = rt.block_on(async {
1327            use datafusion::prelude::SessionContext;
1328            let mut ctx = SessionContext::new();
1329            
1330            // Register standard functions and aggregates
1331            datafusion_functions::register_all(&mut ctx).map_err(|e| e.to_string())?;
1332            datafusion_functions_aggregate::register_all(&mut ctx).map_err(|e| e.to_string())?;
1333            
1334            let _ = crate::core::sql::vector_operators::register_vector_operators(&mut ctx);
1335            
1336            // Register table as 't' (short alias, safe from keywords)
1337            let provider = Arc::new(crate::core::sql::HyperStreamTableProvider::new(Arc::new(self.table.clone())));
1338            ctx.register_table("t", provider).map_err(|e| e.to_string())?;
1339            
1340            // Register vector UDFs (dist_l2, dist_cosine, etc.)
1341            for udf in crate::core::sql::vector_udf::all_vector_udfs() {
1342                ctx.register_udf(udf);
1343            }
1344            
1345            // Register Vector Aggregate functions (Additive in DF 52)
1346            for udf in crate::core::sql::vector_udf::all_vector_aggregates() {
1347                ctx.register_udaf(udf);
1348            }
1349            
1350            // Execute
1351            let df = ctx.sql(&query).await.map_err(|e| e.to_string())?;
1352            let schema: arrow::datatypes::SchemaRef = std::sync::Arc::new(df.schema().as_arrow().clone());
1353            let batches = df.collect().await.map_err(|e| e.to_string())?;
1354            Ok((batches, schema))
1355        });
1356        
1357        match batch_result {
1358            Ok((batches, schema)) => arrow_batches_to_pyarrow(py, batches, schema),
1359            Err(e) => Err(pyo3::exceptions::PyRuntimeError::new_err(e)),
1360        }
1361    }

1497    /// 
1498    /// Args:
1499    ///     filter: Optional SQL-like filter string
1500    ///     vector_filter: Optional dict for vector search (see to_arrow for parameter details)
1501    /// 
1502    /// Returns:
1503    ///     String explaining the query execution plan with index coverage
1504    #[pyo3(signature = (filter=None, vector_filter=None))]
1505    fn explain(&self, filter: Option<String>, vector_filter: Option<Bound<'_, PyDict>>) -> PyResult<String> {
1506        let vs_params = if let Some(ref vf) = vector_filter {
1507            let column: String = vf.get_item("column")?
1508                .ok_or_else(|| pyo3::exceptions::PyValueError::new_err("vector_filter requires 'column' key"))?
1509                .extract()?;
1510            let k: usize = vf.get_item("k")?
1511                .ok_or_else(|| pyo3::exceptions::PyValueError::new_err("vector_filter requires 'k' key"))?
1512                .extract()?;
1513            let query_obj = vf.get_item("query")?
1514                .ok_or_else(|| pyo3::exceptions::PyValueError::new_err("vector_filter requires 'query' key"))?;
1515            let query: Vec<f32> = query_obj.extract()?;
1516            let mut params = VectorSearchParams::new(&column, crate::core::index::VectorValue::Float32(query), k);
1517            
1518            // Parse optional metric parameter
1519            if let Ok(Some(metric_obj)) = vf.get_item("metric") {
1520                if let Ok(metric_str) = metric_obj.extract::<String>() {
1521                    params = params.with_metric(parse_metric(&metric_str)?);
1522                }
1523            }
1524            
1525            // Parse optional ef_search parameter (for HNSW)
1526            if let Ok(Some(ef_obj)) = vf.get_item("ef_search") {
1527                if let Ok(ef_search) = ef_obj.extract::<usize>() {
1528                    params = params.with_ef_search(ef_search);
1529                }
1530            }
1531            
1532            // Parse optional probes parameter (for IVF)
1533            if let Ok(Some(probes_obj)) = vf.get_item("probes") {
1534                if let Ok(probes) = probes_obj.extract::<usize>() {
1535                    params = params.with_probes(probes);
1536                }
1537            }
1538            
1539            Some(params)
1540        } else {
1541            None
1542        };
1543
1544        Ok(TOKIO_RUNTIME.block_on(self.table.explain(filter.as_deref(), vs_params)))
1545    }

1813fn arrow_schema_to_pyarrow(py: Python<'_>, schema: arrow::datatypes::SchemaRef) -> PyResult<Py<PyAny>> {
1814    let mut ffi_schema = FFI_ArrowSchema::try_from(schema.as_ref())
1815        .map_err(|e| pyo3::exceptions::PyRuntimeError::new_err((e.to_string(), )))?;
1816    
1817    let schema_ptr = &mut ffi_schema as *mut _ as Py_uintptr_t;
1818    let pyarrow = py.import("pyarrow")?;
1819    let schema_class = pyarrow.getattr("Schema")?;
1820    let py_schema = schema_class.call_method1("_import_from_c", (schema_ptr,))?.unbind();
1821    
1822    Ok(py_schema)
1823}

1825fn arrow_batches_to_pyarrow(py: Python<'_>, batches: Vec<RecordBatch>, schema: arrow::datatypes::SchemaRef) -> PyResult<Py<PyAny>> {
1826    // Use Arrow C Stream Interface for efficient transfer
1827    let batch_iter = RecordBatchIterator::new(batches.into_iter().map(Ok), schema);
1828    
1829    // Export to C Stream
1830    let stream = FFI_ArrowArrayStream::new(Box::new(batch_iter));
1831    let stream_ptr = Box::into_raw(Box::new(stream)) as Py_uintptr_t;
1832    
1833    // Import in Python via PyArrow
1834    let pyarrow = py.import("pyarrow")?;
1835    let reader_class = pyarrow.getattr("RecordBatchReader")?;
1836    let table = reader_class.call_method1("_import_from_c", (stream_ptr,))?
1837        .call_method0("read_all")?
1838        .unbind();
1839    
1840    Ok(table)
1841}

2042fn pyarrow_to_arrow_batches(py: Python<'_>, table: Py<PyAny>) -> PyResult<Vec<RecordBatch>> {
2043    // Convert PyArrow Table to batches via C Stream Interface
2044    let _pyarrow = py.import("pyarrow")?;
2045    
2046    // Get RecordBatchReader
2047    let reader = table.call_method0(py, "to_reader")?;
2048    
2049    // Create struct to hold the exported stream
2050    let mut stream = FFI_ArrowArrayStream::empty();
2051    let stream_ptr = &mut stream as *mut FFI_ArrowArrayStream as Py_uintptr_t;
2052
2053    // Export to C Stream (pass pointer to python)
2054    reader.call_method1(py, "_export_to_c", (stream_ptr,))?;
2055    
2056    // Import from C Stream
2057    let stream_reader = ArrowArrayStreamReader::try_new(stream)
2058        .map_err(|e| pyo3::exceptions::PyRuntimeError::new_err((e.to_string(), )))?;
2059    
2060    let mut batches = Vec::new();
2061    for batch_result in stream_reader {

2092fn extract_partition_spec(spec_obj: Bound<'_, PyAny>) -> PyResult<crate::core::manifest::PartitionSpec> {
2093    let dict = spec_obj.downcast::<pyo3::types::PyDict>()
2094        .map_err(|_| pyo3::exceptions::PyTypeError::new_err("partition_spec must be a dictionary"))?;
2095
2096    let fields_obj = dict.get_item("fields")?
2097        .ok_or_else(|| pyo3::exceptions::PyValueError::new_err("partition_spec must contain 'fields'"))?;
2098    let fields_list = fields_obj.downcast::<pyo3::types::PyList>()
2099        .map_err(|_| pyo3::exceptions::PyTypeError::new_err("'fields' must be a list"))?;
2100
2101    let mut fields = Vec::new();
2102    for item in fields_list {
2103        let f_dict = item.downcast::<pyo3::types::PyDict>()
2104            .map_err(|_| pyo3::exceptions::PyTypeError::new_err("Each partition field must be a dictionary"))?;
2105
2106        let name = f_dict.get_item("name")?
2107            .ok_or_else(|| pyo3::exceptions::PyValueError::new_err("Missing 'name' in partition field"))?
2108            .extract::<String>()?;
2109        let transform = f_dict.get_item("transform")?
2110            .ok_or_else(|| pyo3::exceptions::PyValueError::new_err("Missing 'transform' in partition field"))?
2111            .extract::<String>()?;
2112        
2113        let source_id = f_dict.get_item("source_id")?.and_then(|i| i.extract::<i32>().ok());
2114        let field_id = f_dict.get_item("field_id")?.and_then(|i| i.extract::<i32>().ok());
2115
2116        fields.push(crate::core::manifest::PartitionField {
2117            source_ids: source_id.map(|id| vec![id]).unwrap_or_default(),
2118            source_id,
2119            field_id,
2120            name,
2121            transform,
2122        });
2123    }
2124
2125    Ok(crate::core::manifest::PartitionSpec {
2126        spec_id: 0,
2127        fields,
2128    })
